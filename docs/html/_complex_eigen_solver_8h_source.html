251  ComplexEigenSolver<MatrixType>&
252  ComplexEigenSolver<MatrixType>::compute(const EigenBase<InputType>& matrix, bool computeEigenvectors) {
253  check_template_parameters();
254 
255  // this code is inspired from Jampack
256  eigen_assert(matrix.cols() == matrix.rows());
257 
258  // Do a complex Schur decomposition, A = U T U^*
259  // The eigenvalues are on the diagonal of T.
260  m_schur.compute(matrix.derived(), computeEigenvectors);
261 
262  if (m_schur.info() == Success) {
263  m_eivalues = m_schur.matrixT().diagonal();
264  if (computeEigenvectors)
265  doComputeEigenvectors(m_schur.matrixT().norm());
266  sortEigenvalues(computeEigenvectors);
267  }
268 
269  m_isInitialized = true;
270  m_eigenvectorsOk = computeEigenvectors;
271  return *this;
272  }

274  template<typename MatrixType>
275  void ComplexEigenSolver<MatrixType>::doComputeEigenvectors(RealScalar matrixnorm) {
276  const Index n = m_eivalues.size();
277 
278  matrixnorm = numext::maxi(matrixnorm, (std::numeric_limits<RealScalar>::min)());
279 
280  // Compute X such that T = X D X^(-1), where D is the diagonal of T.
281  // The matrix X is unit triangular.
282  m_matX = EigenvectorType::Zero(n, n);
283  for (Index k = n - 1; k >= 0; k--) {
284  m_matX.coeffRef(k, k) = ComplexScalar(1.0, 0.0);
285  // Compute X(i,k) using the (i,k) entry of the equation X T = D X
286  for (Index i = k - 1; i >= 0; i--) {
287  m_matX.coeffRef(i, k) = -m_schur.matrixT().coeff(i, k);
288  if (k - i - 1 > 0)
289  m_matX.coeffRef(i, k) -= (m_schur.matrixT().row(i).segment(i + 1, k - i - 1) * m_matX.col(k).segment(i + 1, k - i - 1)).value();
290  ComplexScalar z = m_schur.matrixT().coeff(i, i) - m_schur.matrixT().coeff(k, k);
291  if (z == ComplexScalar(0)) {
292  // If the i-th and k-th eigenvalue are equal, then z equals 0.
293  // Use a small value instead, to prevent division by zero.
294  numext::real_ref(z) = NumTraits<RealScalar>::epsilon() * matrixnorm;
295  }
296  m_matX.coeffRef(i, k) = m_matX.coeff(i, k) / z;
297  }
298  }
299 
300  // Compute V as V = U X; now A = U T U^* = U X D X^(-1) U^* = V D V^(-1)
301  m_eivec.noalias() = m_schur.matrixU() * m_matX;
302  // .. and normalize the eigenvectors
303  for (Index k = 0; k < n; k++) {
304  m_eivec.col(k).normalize();
305  }
306  }

308  template<typename MatrixType>
309  void ComplexEigenSolver<MatrixType>::sortEigenvalues(bool computeEigenvectors) {
310  const Index n = m_eivalues.size();
311  for (Index i = 0; i < n; i++) {
312  Index k;
313  m_eivalues.cwiseAbs().tail(n - i).minCoeff(&k);
314  if (k != 0) {
315  k += i;
316  std::swap(m_eivalues[k], m_eivalues[i]);
317  if (computeEigenvectors)
318  m_eivec.col(i).swap(m_eivec.col(k));
319  }
320  }
321  }
