24  template<typename Scalar, typename Index, typename UType, typename VType>
25  struct selfadjoint_rank2_update_selector<Scalar, Index, UType, VType, Lower> {
26 
27  static void run(Scalar* mat, Index stride, const UType& u, const VType& v, const Scalar& alpha) {
28  const Index size = u.size();
29  for (Index i = 0; i < size; ++i) {
30  Map<Matrix<Scalar, Dynamic, 1> >(mat + stride * i + i, size - i) +=
31  (numext::conj(alpha) * numext::conj(u.coeff(i))) * v.tail(size - i)
32  + (alpha * numext::conj(v.coeff(i))) * u.tail(size - i);
33  }
34  }
35  };

37  template<typename Scalar, typename Index, typename UType, typename VType>
38  struct selfadjoint_rank2_update_selector<Scalar, Index, UType, VType, Upper> {
39 
40  static void run(Scalar* mat, Index stride, const UType& u, const VType& v, const Scalar& alpha) {
41  const Index size = u.size();
42  for (Index i = 0; i < size; ++i)
43  Map<Matrix<Scalar, Dynamic, 1> >(mat + stride * i, i + 1) +=
44  (numext::conj(alpha) * numext::conj(u.coeff(i))) * v.head(i + 1)
45  + (alpha * numext::conj(v.coeff(i))) * u.head(i + 1);
46  }
47  };

56  template<typename MatrixType, unsigned int UpLo>
57  template<typename DerivedU, typename DerivedV>
58  SelfAdjointView<MatrixType, UpLo>& SelfAdjointView<MatrixType, UpLo>
59  ::rankUpdate(const MatrixBase<DerivedU>& u, const MatrixBase<DerivedV>& v, const Scalar& alpha) {
60  typedef internal::blas_traits<DerivedU> UBlasTraits;
61  typedef typename UBlasTraits::DirectLinearAccessType ActualUType;
62  typedef typename internal::remove_all<ActualUType>::type _ActualUType;
63  typename internal::add_const_on_value_type<ActualUType>::type actualU = UBlasTraits::extract(u.derived());
64 
65  typedef internal::blas_traits<DerivedV> VBlasTraits;
66  typedef typename VBlasTraits::DirectLinearAccessType ActualVType;
67  typedef typename internal::remove_all<ActualVType>::type _ActualVType;
68  typename internal::add_const_on_value_type<ActualVType>::type actualV = VBlasTraits::extract(v.derived());
69 
70  // If MatrixType is row major, then we use the routine for lower triangular in the upper triangular case and
71  // vice versa, and take the complex conjugate of all coefficients and vector entries.
72 
73  enum {
74  IsRowMajor = (internal::traits<MatrixType>::Flags&RowMajorBit) ? 1 : 0
75  };
76  Scalar actualAlpha = alpha * UBlasTraits::extractScalarFactor(u.derived())
77  * numext::conj(VBlasTraits::extractScalarFactor(v.derived()));
78  if (IsRowMajor)
79  actualAlpha = numext::conj(actualAlpha);
80 
81  typedef typename internal::remove_all<typename internal::conj_expr_if < IsRowMajor ^ UBlasTraits::NeedToConjugate, _ActualUType>::type>::type UType;
82  typedef typename internal::remove_all<typename internal::conj_expr_if < IsRowMajor ^ VBlasTraits::NeedToConjugate, _ActualVType>::type>::type VType;
83  internal::selfadjoint_rank2_update_selector<Scalar, Index, UType, VType,
84  (IsRowMajor ? int(UpLo == Upper ? Lower : Upper) : UpLo)>
85  ::run(_expression().const_cast_derived().data(), _expression().outerStride(), UType(actualU), VType(actualV), actualAlpha);
86 
87  return *this;
88  }
