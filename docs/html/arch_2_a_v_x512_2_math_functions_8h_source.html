32  // Natural logarithm
33  // Computes log(x) as log(2^e * m) = C*e + log(m), where the constant C =log(2)
34  // and m is in the range [sqrt(1/2),sqrt(2)). In this range, the logarithm can
35  // be easily approximated by a polynomial centered on m=1 for stability.
36 #if defined(EIGEN_VECTORIZE_AVX512DQ)
37 
38 template <>
39  EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS EIGEN_UNUSED Packet16f
40  plog<Packet16f>(const Packet16f& _x) {
41  Packet16f x = _x;
42  _EIGEN_DECLARE_CONST_Packet16f(1, 1.0f);
43  _EIGEN_DECLARE_CONST_Packet16f(half, 0.5f);
44  _EIGEN_DECLARE_CONST_Packet16f(126f, 126.0f);
45 
46  _EIGEN_DECLARE_CONST_Packet16f_FROM_INT(inv_mant_mask, ~0x7f800000);
47 
48  // The smallest non denormalized float number.
49  _EIGEN_DECLARE_CONST_Packet16f_FROM_INT(min_norm_pos, 0x00800000);
50  _EIGEN_DECLARE_CONST_Packet16f_FROM_INT(minus_inf, 0xff800000);
51  _EIGEN_DECLARE_CONST_Packet16f_FROM_INT(nan, 0x7fc00000);
52 
53  // Polynomial coefficients.
54  _EIGEN_DECLARE_CONST_Packet16f(cephes_SQRTHF, 0.707106781186547524f);
55  _EIGEN_DECLARE_CONST_Packet16f(cephes_log_p0, 7.0376836292E-2f);
56  _EIGEN_DECLARE_CONST_Packet16f(cephes_log_p1, -1.1514610310E-1f);
57  _EIGEN_DECLARE_CONST_Packet16f(cephes_log_p2, 1.1676998740E-1f);
58  _EIGEN_DECLARE_CONST_Packet16f(cephes_log_p3, -1.2420140846E-1f);
59  _EIGEN_DECLARE_CONST_Packet16f(cephes_log_p4, +1.4249322787E-1f);
60  _EIGEN_DECLARE_CONST_Packet16f(cephes_log_p5, -1.6668057665E-1f);
61  _EIGEN_DECLARE_CONST_Packet16f(cephes_log_p6, +2.0000714765E-1f);
62  _EIGEN_DECLARE_CONST_Packet16f(cephes_log_p7, -2.4999993993E-1f);
63  _EIGEN_DECLARE_CONST_Packet16f(cephes_log_p8, +3.3333331174E-1f);
64  _EIGEN_DECLARE_CONST_Packet16f(cephes_log_q1, -2.12194440e-4f);
65  _EIGEN_DECLARE_CONST_Packet16f(cephes_log_q2, 0.693359375f);
66 
67  // invalid_mask is set to true when x is NaN
68  __mmask16 invalid_mask =
69  _mm512_cmp_ps_mask(x, _mm512_setzero_ps(), _CMP_NGE_UQ);
70  __mmask16 iszero_mask =
71  _mm512_cmp_ps_mask(x, _mm512_setzero_ps(), _CMP_EQ_UQ);
72 
73  // Truncate input values to the minimum positive normal.
74  x = pmax(x, p16f_min_norm_pos);
75 
76  // Extract the shifted exponents.
77  Packet16f emm0 = _mm512_cvtepi32_ps(_mm512_srli_epi32((__m512i) x, 23));
78  Packet16f e = _mm512_sub_ps(emm0, p16f_126f);
79 
80  // Set the exponents to -1, i.e. x are in the range [0.5,1).
81  x = _mm512_and_ps(x, p16f_inv_mant_mask);
82  x = _mm512_or_ps(x, p16f_half);
83 
84  // part2: Shift the inputs from the range [0.5,1) to [sqrt(1/2),sqrt(2))
85  // and shift by -1. The values are then centered around 0, which improves
86  // the stability of the polynomial evaluation.
87  // if( x < SQRTHF ) {
88  // e -= 1;
89  // x = x + x - 1.0;
90  // } else { x = x - 1.0; }
91  __mmask16 mask = _mm512_cmp_ps_mask(x, p16f_cephes_SQRTHF, _CMP_LT_OQ);
92  Packet16f tmp = _mm512_mask_blend_ps(mask, _mm512_setzero_ps(), x);
93  x = psub(x, p16f_1);
94  e = psub(e, _mm512_mask_blend_ps(mask, _mm512_setzero_ps(), p16f_1));
95  x = padd(x, tmp);
96 
97  Packet16f x2 = pmul(x, x);
98  Packet16f x3 = pmul(x2, x);
99 
100  // Evaluate the polynomial approximant of degree 8 in three parts, probably
101  // to improve instruction-level parallelism.
102  Packet16f y, y1, y2;
103  y = pmadd(p16f_cephes_log_p0, x, p16f_cephes_log_p1);
104  y1 = pmadd(p16f_cephes_log_p3, x, p16f_cephes_log_p4);
105  y2 = pmadd(p16f_cephes_log_p6, x, p16f_cephes_log_p7);
106  y = pmadd(y, x, p16f_cephes_log_p2);
107  y1 = pmadd(y1, x, p16f_cephes_log_p5);
108  y2 = pmadd(y2, x, p16f_cephes_log_p8);
109  y = pmadd(y, x3, y1);
110  y = pmadd(y, x3, y2);
111  y = pmul(y, x3);
112 
113  // Add the logarithm of the exponent back to the result of the interpolation.
114  y1 = pmul(e, p16f_cephes_log_q1);
115  tmp = pmul(x2, p16f_half);
116  y = padd(y, y1);
117  x = psub(x, tmp);
118  y2 = pmul(e, p16f_cephes_log_q2);
119  x = padd(x, y);
120  x = padd(x, y2);
121 
122  // Filter out invalid inputs, i.e. negative arg will be NAN, 0 will be -INF.
123  return _mm512_mask_blend_ps(iszero_mask,
124  _mm512_mask_blend_ps(invalid_mask, x, p16f_nan),
125  p16f_minus_inf);
126  }

129  // Exponential function. Works by writing "x = m*log(2) + r" where
130  // "m = floor(x/log(2)+1/2)" and "r" is the remainder. The result is then
131  // "exp(x) = 2^m*exp(r)" where exp(r) is in the range [-1,1).
132 
133  template <>
134  EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS EIGEN_UNUSED Packet16f
135  pexp<Packet16f>(const Packet16f& _x) {
136  _EIGEN_DECLARE_CONST_Packet16f(1, 1.0f);
137  _EIGEN_DECLARE_CONST_Packet16f(half, 0.5f);
138  _EIGEN_DECLARE_CONST_Packet16f(127, 127.0f);
139 
140  _EIGEN_DECLARE_CONST_Packet16f(exp_hi, 88.3762626647950f);
141  _EIGEN_DECLARE_CONST_Packet16f(exp_lo, -88.3762626647949f);
142 
143  _EIGEN_DECLARE_CONST_Packet16f(cephes_LOG2EF, 1.44269504088896341f);
144 
145  _EIGEN_DECLARE_CONST_Packet16f(cephes_exp_p0, 1.9875691500E-4f);
146  _EIGEN_DECLARE_CONST_Packet16f(cephes_exp_p1, 1.3981999507E-3f);
147  _EIGEN_DECLARE_CONST_Packet16f(cephes_exp_p2, 8.3334519073E-3f);
148  _EIGEN_DECLARE_CONST_Packet16f(cephes_exp_p3, 4.1665795894E-2f);
149  _EIGEN_DECLARE_CONST_Packet16f(cephes_exp_p4, 1.6666665459E-1f);
150  _EIGEN_DECLARE_CONST_Packet16f(cephes_exp_p5, 5.0000001201E-1f);
151 
152  // Clamp x.
153  Packet16f x = pmax(pmin(_x, p16f_exp_hi), p16f_exp_lo);
154 
155  // Express exp(x) as exp(m*ln(2) + r), start by extracting
156  // m = floor(x/ln(2) + 0.5).
157  Packet16f m = _mm512_floor_ps(pmadd(x, p16f_cephes_LOG2EF, p16f_half));
158 
159  // Get r = x - m*ln(2). Note that we can do this without losing more than one
160  // ulp precision due to the FMA instruction.
161  _EIGEN_DECLARE_CONST_Packet16f(nln2, -0.6931471805599453f);
162  Packet16f r = _mm512_fmadd_ps(m, p16f_nln2, x);
163  Packet16f r2 = pmul(r, r);
164 
165  // TODO(gonnet): Split into odd/even polynomials and try to exploit
166  // instruction-level parallelism.
167  Packet16f y = p16f_cephes_exp_p0;
168  y = pmadd(y, r, p16f_cephes_exp_p1);
169  y = pmadd(y, r, p16f_cephes_exp_p2);
170  y = pmadd(y, r, p16f_cephes_exp_p3);
171  y = pmadd(y, r, p16f_cephes_exp_p4);
172  y = pmadd(y, r, p16f_cephes_exp_p5);
173  y = pmadd(y, r2, r);
174  y = padd(y, p16f_1);
175 
176  // Build emm0 = 2^m.
177  Packet16i emm0 = _mm512_cvttps_epi32(padd(m, p16f_127));
178  emm0 = _mm512_slli_epi32(emm0, 23);
179 
180  // Return 2^m * exp(r).
181  return pmax(pmul(y, _mm512_castsi512_ps(emm0)), _x);
182  }

253  // Functions for sqrt.
254  // The EIGEN_FAST_MATH version uses the _mm_rsqrt_ps approximation and one step
255  // of Newton's method, at a cost of 1-2 bits of precision as opposed to the
256  // exact solution. The main advantage of this approach is not just speed, but
257  // also the fact that it can be inlined and pipelined with other computations,
258  // further reducing its effective latency.
259 #if EIGEN_FAST_MATH
260 
261 template <>
262  EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS EIGEN_UNUSED Packet16f
263  psqrt<Packet16f>(const Packet16f& _x) {
264  _EIGEN_DECLARE_CONST_Packet16f(one_point_five, 1.5f);
265  _EIGEN_DECLARE_CONST_Packet16f(minus_half, -0.5f);
266  _EIGEN_DECLARE_CONST_Packet16f_FROM_INT(flt_min, 0x00800000);
267 
268  Packet16f neg_half = pmul(_x, p16f_minus_half);
269 
270  // select only the inverse sqrt of positive normal inputs (denormals are
271  // flushed to zero and cause infs as well).
272  __mmask16 non_zero_mask = _mm512_cmp_ps_mask(_x, p16f_flt_min, _CMP_GE_OQ);
273  Packet16f x = _mm512_mask_blend_ps(non_zero_mask, _mm512_setzero_ps(), _mm512_rsqrt14_ps(_x));
274 
275  // Do a single step of Newton's iteration.
276  x = pmul(x, pmadd(neg_half, pmul(x, x), p16f_one_point_five));
277 
278  // Multiply the original _x by it's reciprocal square root to extract the
279  // square root.
280  return pmul(_x, x);
281  }

284  EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS EIGEN_UNUSED Packet8d
285  psqrt<Packet8d>(const Packet8d& _x) {
286  _EIGEN_DECLARE_CONST_Packet8d(one_point_five, 1.5);
287  _EIGEN_DECLARE_CONST_Packet8d(minus_half, -0.5);
288  _EIGEN_DECLARE_CONST_Packet8d_FROM_INT64(dbl_min, 0x0010000000000000LL);
289 
290  Packet8d neg_half = pmul(_x, p8d_minus_half);
291 
292  // select only the inverse sqrt of positive normal inputs (denormals are
293  // flushed to zero and cause infs as well).
294  __mmask8 non_zero_mask = _mm512_cmp_pd_mask(_x, p8d_dbl_min, _CMP_GE_OQ);
295  Packet8d x = _mm512_mask_blend_pd(non_zero_mask, _mm512_setzero_pd(), _mm512_rsqrt14_pd(_x));
296 
297  // Do a first step of Newton's iteration.
298  x = pmul(x, pmadd(neg_half, pmul(x, x), p8d_one_point_five));
299 
300  // Do a second step of Newton's iteration.
301  x = pmul(x, pmadd(neg_half, pmul(x, x), p8d_one_point_five));
302 
303  // Multiply the original _x by it's reciprocal square root to extract the
304  // square root.
305  return pmul(_x, x);
306  }

320  // Functions for rsqrt.
321  // Almost identical to the sqrt routine, just leave out the last multiplication
322  // and fill in NaN/Inf where needed. Note that this function only exists as an
323  // iterative version for doubles since there is no instruction for diretly
324  // computing the reciprocal square root in AVX-512.
325 #ifdef EIGEN_FAST_MATH
326 
327 template <>
328  EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS EIGEN_UNUSED Packet16f
329  prsqrt<Packet16f>(const Packet16f& _x) {
330  _EIGEN_DECLARE_CONST_Packet16f_FROM_INT(inf, 0x7f800000);
331  _EIGEN_DECLARE_CONST_Packet16f_FROM_INT(nan, 0x7fc00000);
332  _EIGEN_DECLARE_CONST_Packet16f(one_point_five, 1.5f);
333  _EIGEN_DECLARE_CONST_Packet16f(minus_half, -0.5f);
334  _EIGEN_DECLARE_CONST_Packet16f_FROM_INT(flt_min, 0x00800000);
335 
336  Packet16f neg_half = pmul(_x, p16f_minus_half);
337 
338  // select only the inverse sqrt of positive normal inputs (denormals are
339  // flushed to zero and cause infs as well).
340  __mmask16 le_zero_mask = _mm512_cmp_ps_mask(_x, p16f_flt_min, _CMP_LT_OQ);
341  Packet16f x = _mm512_mask_blend_ps(le_zero_mask, _mm512_rsqrt14_ps(_x), _mm512_setzero_ps());
342 
343  // Fill in NaNs and Infs for the negative/zero entries.
344  __mmask16 neg_mask = _mm512_cmp_ps_mask(_x, _mm512_setzero_ps(), _CMP_LT_OQ);
345  Packet16f infs_and_nans = _mm512_mask_blend_ps(
346  neg_mask, _mm512_mask_blend_ps(le_zero_mask, _mm512_setzero_ps(), p16f_inf), p16f_nan);
347 
348  // Do a single step of Newton's iteration.
349  x = pmul(x, pmadd(neg_half, pmul(x, x), p16f_one_point_five));
350 
351  // Insert NaNs and Infs in all the right places.
352  return _mm512_mask_blend_ps(le_zero_mask, x, infs_and_nans);
353  }
