25  template<typename T>
26  T generic_fast_tanh_float(const T& a_x) {
27  // Clamp the inputs to the range [-9, 9] since anything outside
28  // this range is +/-1.0f in single-precision.
29  const T plus_9 = pset1<T>(9.f);
30  const T minus_9 = pset1<T>(-9.f);
31  // NOTE GCC prior to 6.3 might improperly optimize this max/min
32  // step such that if a_x is nan, x will be either 9 or -9,
33  // and tanh will return 1 or -1 instead of nan.
34  // This is supposed to be fixed in gcc6.3,
35  // see: https://gcc.gnu.org/bugzilla/show_bug.cgi?id=72867
36  const T x = pmax(minus_9, pmin(plus_9, a_x));
37  // The monomial coefficients of the numerator polynomial (odd).
38  const T alpha_1 = pset1<T>(4.89352455891786e-03f);
39  const T alpha_3 = pset1<T>(6.37261928875436e-04f);
40  const T alpha_5 = pset1<T>(1.48572235717979e-05f);
41  const T alpha_7 = pset1<T>(5.12229709037114e-08f);
42  const T alpha_9 = pset1<T>(-8.60467152213735e-11f);
43  const T alpha_11 = pset1<T>(2.00018790482477e-13f);
44  const T alpha_13 = pset1<T>(-2.76076847742355e-16f);
45 
46  // The monomial coefficients of the denominator polynomial (even).
47  const T beta_0 = pset1<T>(4.89352518554385e-03f);
48  const T beta_2 = pset1<T>(2.26843463243900e-03f);
49  const T beta_4 = pset1<T>(1.18534705686654e-04f);
50  const T beta_6 = pset1<T>(1.19825839466702e-06f);
51 
52  // Since the polynomials are odd/even, we need x^2.
53  const T x2 = pmul(x, x);
54 
55  // Evaluate the numerator polynomial p.
56  T p = pmadd(x2, alpha_13, alpha_11);
57  p = pmadd(x2, p, alpha_9);
58  p = pmadd(x2, p, alpha_7);
59  p = pmadd(x2, p, alpha_5);
60  p = pmadd(x2, p, alpha_3);
61  p = pmadd(x2, p, alpha_1);
62  p = pmul(x, p);
63 
64  // Evaluate the denominator polynomial p.
65  T q = pmadd(x2, beta_6, beta_4);
66  q = pmadd(x2, q, beta_2);
67  q = pmadd(x2, q, beta_0);
68 
69  // Divide the numerator by the denominator.
70  return pdiv(p, q);
71  }

73  template<typename RealScalar>
74  EIGEN_STRONG_INLINE
75  RealScalar positive_real_hypot(const RealScalar& x, const RealScalar& y) {
76  EIGEN_USING_STD_MATH(sqrt);
77  RealScalar p, qp;
78  p = numext::maxi(x, y);
79  if (p == RealScalar(0)) return RealScalar(0);
80  qp = numext::mini(y, x) / p;
81  return p * sqrt(RealScalar(1) + qp * qp);
82  }
