17  template<typename ExpressionType, typename Scalar>
18  inline void stable_norm_kernel(const ExpressionType& bl, Scalar& ssq, Scalar& scale, Scalar& invScale) {
19  Scalar maxCoeff = bl.cwiseAbs().maxCoeff();
20 
21  if (maxCoeff > scale) {
22  ssq = ssq * numext::abs2(scale / maxCoeff);
23  Scalar tmp = Scalar(1) / maxCoeff;
24  if (tmp > NumTraits<Scalar>::highest()) {
25  invScale = NumTraits<Scalar>::highest();
26  scale = Scalar(1) / invScale;
27  } else if (maxCoeff > NumTraits<Scalar>::highest()) // we got a INF
28  {
29  invScale = Scalar(1);
30  scale = maxCoeff;
31  } else {
32  scale = maxCoeff;
33  invScale = tmp;
34  }
35  } else if (maxCoeff != maxCoeff) // we got a NaN
36  {
37  scale = maxCoeff;
38  }
39 
40  // TODO if the maxCoeff is much much smaller than the current scale,
41  // then we can neglect this sub vector
42  if (scale > Scalar(0)) // if scale==0, then bl is 0
43  ssq += (bl * invScale).squaredNorm();
44  }

46  template<typename Derived>
47  inline typename NumTraits<typename traits<Derived>::Scalar>::Real
48  blueNorm_impl(const EigenBase<Derived>& _vec) {
49  typedef typename Derived::RealScalar RealScalar;
50  using std::pow;
51  using std::sqrt;
52  using std::abs;
53  const Derived & vec(_vec.derived());
54  static bool initialized = false;
55  static RealScalar b1, b2, s1m, s2m, rbig, relerr;
56  if (!initialized) {
57  int ibeta, it, iemin, iemax, iexp;
58  RealScalar eps;
59  // This program calculates the machine-dependent constants
60  // bl, b2, slm, s2m, relerr overfl
61  // from the "basic" machine-dependent numbers
62  // nbig, ibeta, it, iemin, iemax, rbig.
63  // The following define the basic machine-dependent constants.
64  // For portability, the PORT subprograms "ilmaeh" and "rlmach"
65  // are used. For any specific computer, each of the assignment
66  // statements can be replaced
67  ibeta = std::numeric_limits<RealScalar>::radix; // base for floating-point numbers
68  it = std::numeric_limits<RealScalar>::digits; // number of base-beta digits in mantissa
69  iemin = std::numeric_limits<RealScalar>::min_exponent; // minimum exponent
70  iemax = std::numeric_limits<RealScalar>::max_exponent; // maximum exponent
71  rbig = (std::numeric_limits<RealScalar>::max)(); // largest floating-point number
72 
73  iexp = -((1 - iemin) / 2);
74  b1 = RealScalar(pow(RealScalar(ibeta), RealScalar(iexp))); // lower boundary of midrange
75  iexp = (iemax + 1 - it) / 2;
76  b2 = RealScalar(pow(RealScalar(ibeta), RealScalar(iexp))); // upper boundary of midrange
77 
78  iexp = (2 - iemin) / 2;
79  s1m = RealScalar(pow(RealScalar(ibeta), RealScalar(iexp))); // scaling factor for lower range
80  iexp = -((iemax + it) / 2);
81  s2m = RealScalar(pow(RealScalar(ibeta), RealScalar(iexp))); // scaling factor for upper range
82 
83  eps = RealScalar(pow(double(ibeta), 1 - it));
84  relerr = sqrt(eps); // tolerance for neglecting asml
85  initialized = true;
86  }
87  Index n = vec.size();
88  RealScalar ab2 = b2 / RealScalar(n);
89  RealScalar asml = RealScalar(0);
90  RealScalar amed = RealScalar(0);
91  RealScalar abig = RealScalar(0);
92  for (typename Derived::InnerIterator it(vec, 0); it; ++it) {
93  RealScalar ax = abs(it.value());
94  if (ax > ab2) abig += numext::abs2(ax * s2m);
95  else if (ax < b1) asml += numext::abs2(ax * s1m);
96  else amed += numext::abs2(ax);
97  }
98  if (amed != amed)
99  return amed; // we got a NaN
100  if (abig > RealScalar(0)) {
101  abig = sqrt(abig);
102  if (abig > rbig) // overflow, or *this contains INF values
103  return abig; // return INF
104  if (amed > RealScalar(0)) {
105  abig = abig / s2m;
106  amed = sqrt(amed);
107  } else
108  return abig / s2m;
109  } else if (asml > RealScalar(0)) {
110  if (amed > RealScalar(0)) {
111  abig = sqrt(amed);
112  amed = sqrt(asml) / s1m;
113  } else
114  return sqrt(asml) / s1m;
115  } else
116  return sqrt(amed);
117  asml = numext::mini(abig, amed);
118  abig = numext::maxi(abig, amed);
119  if (asml <= abig * relerr)
120  return abig;
121  else
122  return abig * sqrt(RealScalar(1) + numext::abs2(asml / abig));
123  }

137  template<typename Derived>
138  inline typename NumTraits<typename internal::traits<Derived>::Scalar>::Real
139  MatrixBase<Derived>::stableNorm() const {
140  using std::sqrt;
141  using std::abs;
142  const Index blockSize = 4096;
143  RealScalar scale(0);
144  RealScalar invScale(1);
145  RealScalar ssq(0); // sum of square
146 
147  typedef typename internal::nested_eval<Derived, 2>::type DerivedCopy;
148  typedef typename internal::remove_all<DerivedCopy>::type DerivedCopyClean;
149  const DerivedCopy copy(derived());
150 
151  enum {
152  CanAlign = ((int(DerivedCopyClean::Flags) & DirectAccessBit)
153  || (int(internal::evaluator<DerivedCopyClean>::Alignment) > 0) // FIXME Alignment)>0 might not be enough
154  ) && (blockSize * sizeof (Scalar)*2 < EIGEN_STACK_ALLOCATION_LIMIT)
155  && (EIGEN_MAX_STATIC_ALIGN_BYTES > 0) // if we cannot allocate on the stack, then let's not bother about this optimization
156  };
157  typedef typename internal::conditional<CanAlign, Ref<const Matrix<Scalar, Dynamic, 1, 0, blockSize, 1>, internal::evaluator<DerivedCopyClean>::Alignment>,
158  typename DerivedCopyClean::ConstSegmentReturnType>::type SegmentWrapper;
159  Index n = size();
160 
161  if (n == 1)
162  return abs(this->coeff(0));
163 
164  Index bi = internal::first_default_aligned(copy);
165  if (bi > 0)
166  internal::stable_norm_kernel(copy.head(bi), ssq, scale, invScale);
167  for (; bi < n; bi += blockSize)
168  internal::stable_norm_kernel(SegmentWrapper(copy.segment(bi, numext::mini(blockSize, n - bi))), ssq, scale, invScale);
169  return scale * sqrt(ssq);
170  }
