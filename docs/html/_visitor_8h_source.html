48  visitor(mat.coeff(i, 0), i, 0);

52  }

112  template<typename Derived>
113  template<typename Visitor>
114  EIGEN_DEVICE_FUNC
115  void DenseBase<Derived>::visit(Visitor& visitor) const {
116  typedef typename internal::visitor_evaluator<Derived> ThisEvaluator;
117  ThisEvaluator thisEval(derived());
118 
119  enum {
120  unroll = SizeAtCompileTime != Dynamic
121  && SizeAtCompileTime * ThisEvaluator::CoeffReadCost + (SizeAtCompileTime - 1) * internal::functor_traits<Visitor>::Cost <= EIGEN_UNROLLING_LIMIT
122  };
123  return internal::visitor_impl<Visitor, ThisEvaluator, unroll ? int(SizeAtCompileTime) : Dynamic>::run(thisEval, visitor);
124  }
