472  template<typename MatrixType, typename DiagType, typename SubDiagType>
473  ComputationInfo computeFromTridiagonal_impl(DiagType& diag, SubDiagType& subdiag, const Index maxIterations, bool computeEigenvectors, MatrixType& eivec) {
474  using std::abs;
475 
476  ComputationInfo info;
477  typedef typename MatrixType::Scalar Scalar;
478 
479  Index n = diag.size();
480  Index end = n - 1;
481  Index start = 0;
482  Index iter = 0; // total number of iterations
483 
484  typedef typename DiagType::RealScalar RealScalar;
485  const RealScalar considerAsZero = (std::numeric_limits<RealScalar>::min)();
486  const RealScalar precision = RealScalar(2) * NumTraits<RealScalar>::epsilon();
487 
488  while (end > 0) {
489  for (Index i = start; i < end; ++i)
490  if (internal::isMuchSmallerThan(abs(subdiag[i]), (abs(diag[i]) + abs(diag[i + 1])), precision) || abs(subdiag[i]) <= considerAsZero)
491  subdiag[i] = 0;
492 
493  // find the largest unreduced block
494  while (end > 0 && subdiag[end - 1] == RealScalar(0)) {
495  end--;
496  }
497  if (end <= 0)
498  break;
499 
500  // if we spent too many iterations, we give up
501  iter++;
502  if (iter > maxIterations * n) break;
503 
504  start = end - 1;
505  while (start > 0 && subdiag[start - 1] != 0)
506  start--;
507 
508  internal::tridiagonal_qr_step < MatrixType::Flags & RowMajorBit ? RowMajor : ColMajor > (diag.data(), subdiag.data(), start, end, computeEigenvectors ? eivec.data() : (Scalar*) 0, n);
509  }
510  if (iter <= maxIterations * n)
511  info = Success;
512  else
513  info = NoConvergence;
514 
515  // Sort eigenvalues and corresponding vectors.
516  // TODO make the sort optional ?
517  // TODO use a better sort algorithm !!
518  if (info == Success) {
519  for (Index i = 0; i < n - 1; ++i) {
520  Index k;
521  diag.segment(i, n - i).minCoeff(&k);
522  if (k > 0) {
523  std::swap(diag[i], diag[k + i]);
524  if (computeEigenvectors)
525  eivec.col(i).swap(eivec.col(k + i));
526  }
527  }
528  }
529  return info;
530  }

550  EIGEN_DEVICE_FUNC
551  static inline void computeRoots(const MatrixType& m, VectorType& roots) {
552  EIGEN_USING_STD_MATH(sqrt)
553  EIGEN_USING_STD_MATH(atan2)
554  EIGEN_USING_STD_MATH(cos)
555  EIGEN_USING_STD_MATH(sin)
556  const Scalar s_inv3 = Scalar(1) / Scalar(3);
557  const Scalar s_sqrt3 = sqrt(Scalar(3));
558 
559  // The characteristic equation is x^3 - c2*x^2 + c1*x - c0 = 0. The
560  // eigenvalues are the roots to this equation, all guaranteed to be
561  // real-valued, because the matrix is symmetric.
562  Scalar c0 = m(0, 0) * m(1, 1) * m(2, 2) + Scalar(2) * m(1, 0) * m(2, 0) * m(2, 1) - m(0, 0) * m(2, 1) * m(2, 1) - m(1, 1) * m(2, 0) * m(2, 0) - m(2, 2) * m(1, 0) * m(1, 0);
563  Scalar c1 = m(0, 0) * m(1, 1) - m(1, 0) * m(1, 0) + m(0, 0) * m(2, 2) - m(2, 0) * m(2, 0) + m(1, 1) * m(2, 2) - m(2, 1) * m(2, 1);
564  Scalar c2 = m(0, 0) + m(1, 1) + m(2, 2);
565 
566  // Construct the parameters used in classifying the roots of the equation
567  // and in solving the equation for the roots in closed form.
568  Scalar c2_over_3 = c2*s_inv3;
569  Scalar a_over_3 = (c2 * c2_over_3 - c1) * s_inv3;
570  a_over_3 = numext::maxi(a_over_3, Scalar(0));
571 
572  Scalar half_b = Scalar(0.5)*(c0 + c2_over_3 * (Scalar(2) * c2_over_3 * c2_over_3 - c1));
573 
574  Scalar q = a_over_3 * a_over_3 * a_over_3 - half_b*half_b;
575  q = numext::maxi(q, Scalar(0));
576 
577  // Compute the eigenvalues by solving for the roots of the polynomial.
578  Scalar rho = sqrt(a_over_3);
579  Scalar theta = atan2(sqrt(q), half_b) * s_inv3; // since sqrt(q) > 0, atan2 is in [0, pi] and theta is in [0, pi/3]
580  Scalar cos_theta = cos(theta);
581  Scalar sin_theta = sin(theta);
582  // roots are already sorted, since cos is monotonically decreasing on [0, pi]
583  roots(0) = c2_over_3 - rho * (cos_theta + s_sqrt3 * sin_theta); // == 2*rho*cos(theta+2pi/3)
584  roots(1) = c2_over_3 - rho * (cos_theta - s_sqrt3 * sin_theta); // == 2*rho*cos(theta+ pi/3)
585  roots(2) = c2_over_3 + Scalar(2) * rho*cos_theta;
586  }

588  EIGEN_DEVICE_FUNC
589  static inline bool extract_kernel(MatrixType& mat, Ref<VectorType> res, Ref<VectorType> representative) {
590  using std::abs;
591  Index i0;
592  // Find non-zero column i0 (by construction, there must exist a non zero coefficient on the diagonal):
593  mat.diagonal().cwiseAbs().maxCoeff(&i0);
594  // mat.col(i0) is a good candidate for an orthogonal vector to the current eigenvector,
595  // so let's save it:
596  representative = mat.col(i0);
597  Scalar n0, n1;
598  VectorType c0, c1;
599  n0 = (c0 = representative.cross(mat.col((i0 + 1) % 3))).squaredNorm();
600  n1 = (c1 = representative.cross(mat.col((i0 + 2) % 3))).squaredNorm();
601  if (n0 > n1) res = c0 / std::sqrt(n0);
602  else res = c1 / std::sqrt(n1);
603 
604  return true;
605  }

607  EIGEN_DEVICE_FUNC
608  static inline void run(SolverType& solver, const MatrixType& mat, int options) {
609  eigen_assert(mat.cols() == 3 && mat.cols() == mat.rows());
610  eigen_assert((options&~(EigVecMask | GenEigMask)) == 0
611  && (options & EigVecMask) != EigVecMask
612  && "invalid option parameter");
613  bool computeEigenvectors = (options & ComputeEigenvectors) == ComputeEigenvectors;
614 
615  EigenvectorsType& eivecs = solver.m_eivec;
616  VectorType& eivals = solver.m_eivalues;
617 
618  // Shift the matrix to the mean eigenvalue and map the matrix coefficients to [-1:1] to avoid over- and underflow.
619  Scalar shift = mat.trace() / Scalar(3);
620  // TODO Avoid this copy. Currently it is necessary to suppress bogus values when determining maxCoeff and for computing the eigenvectors later
621  MatrixType scaledMat = mat.template selfadjointView<Lower>();
622  scaledMat.diagonal().array() -= shift;
623  Scalar scale = scaledMat.cwiseAbs().maxCoeff();
624  if (scale > 0) scaledMat /= scale; // TODO for scale==0 we could save the remaining operations
625 
626  // compute the eigenvalues
627  computeRoots(scaledMat, eivals);
628 
629  // compute the eigenvectors
630  if (computeEigenvectors) {
631  if ((eivals(2) - eivals(0)) <= Eigen::NumTraits<Scalar>::epsilon()) {
632  // All three eigenvalues are numerically the same
633  eivecs.setIdentity();
634  } else {
635  MatrixType tmp;
636  tmp = scaledMat;
637 
638  // Compute the eigenvector of the most distinct eigenvalue
639  Scalar d0 = eivals(2) - eivals(1);
640  Scalar d1 = eivals(1) - eivals(0);
641  Index k(0), l(2);
642  if (d0 > d1) {
643  numext::swap(k, l);
644  d0 = d1;
645  }
646 
647  // Compute the eigenvector of index k
648  {
649  tmp.diagonal().array() -= eivals(k);
650  // By construction, 'tmp' is of rank 2, and its kernel corresponds to the respective eigenvector.
651  extract_kernel(tmp, eivecs.col(k), eivecs.col(l));
652  }
653 
654  // Compute eigenvector of index l
655  if (d0 <= 2 * Eigen::NumTraits<Scalar>::epsilon() * d1) {
656  // If d0 is too small, then the two other eigenvalues are numerically the same,
657  // and thus we only have to ortho-normalize the near orthogonal vector we saved above.
658  eivecs.col(l) -= eivecs.col(k).dot(eivecs.col(l)) * eivecs.col(l);
659  eivecs.col(l).normalize();
660  } else {
661  tmp = scaledMat;
662  tmp.diagonal().array() -= eivals(l);
663 
664  VectorType dummy;
665  extract_kernel(tmp, eivecs.col(l), dummy);
666  }
667 
668  // Compute last eigenvector from the other two
669  eivecs.col(1) = eivecs.col(2).cross(eivecs.col(0)).normalized();
670  }
671  }
672 
673  // Rescale back to the original size.
674  eivals *= scale;
675  eivals.array() += shift;
676 
677  solver.m_info = Success;
678  solver.m_isInitialized = true;
679  solver.m_eigenvectorsOk = computeEigenvectors;
680  }
681  };

692  EIGEN_DEVICE_FUNC
693  static inline void computeRoots(const MatrixType& m, VectorType& roots) {
694  using std::sqrt;
695  const Scalar t0 = Scalar(0.5) * sqrt(numext::abs2(m(0, 0) - m(1, 1)) + Scalar(4) * numext::abs2(m(1, 0)));
696  const Scalar t1 = Scalar(0.5) * (m(0, 0) + m(1, 1));
697  roots(0) = t1 - t0;
698  roots(1) = t1 + t0;
699  }
700 
701  EIGEN_DEVICE_FUNC
702  static inline void run(SolverType& solver, const MatrixType& mat, int options) {
703  EIGEN_USING_STD_MATH(sqrt);
704  EIGEN_USING_STD_MATH(abs);
705 
706  eigen_assert(mat.cols() == 2 && mat.cols() == mat.rows());
707  eigen_assert((options&~(EigVecMask | GenEigMask)) == 0
708  && (options & EigVecMask) != EigVecMask
709  && "invalid option parameter");
710  bool computeEigenvectors = (options & ComputeEigenvectors) == ComputeEigenvectors;
711 
712  EigenvectorsType& eivecs = solver.m_eivec;
713  VectorType& eivals = solver.m_eivalues;
714 
715  // Shift the matrix to the mean eigenvalue and map the matrix coefficients to [-1:1] to avoid over- and underflow.
716  Scalar shift = mat.trace() / Scalar(2);
717  MatrixType scaledMat = mat;
718  scaledMat.coeffRef(0, 1) = mat.coeff(1, 0);
719  scaledMat.diagonal().array() -= shift;
720  Scalar scale = scaledMat.cwiseAbs().maxCoeff();
721  if (scale > Scalar(0))
722  scaledMat /= scale;
723 
724  // Compute the eigenvalues
725  computeRoots(scaledMat, eivals);
726 
727  // compute the eigen vectors
728  if (computeEigenvectors) {
729  if ((eivals(1) - eivals(0)) <= abs(eivals(1)) * Eigen::NumTraits<Scalar>::epsilon()) {
730  eivecs.setIdentity();
731  } else {
732  scaledMat.diagonal().array() -= eivals(1);
733  Scalar a2 = numext::abs2(scaledMat(0, 0));
734  Scalar c2 = numext::abs2(scaledMat(1, 1));
735  Scalar b2 = numext::abs2(scaledMat(1, 0));
736  if (a2 > c2) {
737  eivecs.col(1) << -scaledMat(1, 0), scaledMat(0, 0);
738  eivecs.col(1) /= sqrt(a2 + b2);
739  } else {
740  eivecs.col(1) << -scaledMat(1, 1), scaledMat(1, 0);
741  eivecs.col(1) /= sqrt(c2 + b2);
742  }
743 
744  eivecs.col(0) << eivecs.col(1).unitOrthogonal();
745  }
746  }
747 
748  // Rescale back to the original size.
749  eivals *= scale;
750  eivals.array() += shift;
751 
752  solver.m_info = Success;
753  solver.m_isInitialized = true;
754  solver.m_eigenvectorsOk = computeEigenvectors;
755  }
756  };

770  template<int StorageOrder, typename RealScalar, typename Scalar, typename Index>
771  EIGEN_DEVICE_FUNC
772  static void tridiagonal_qr_step(RealScalar* diag, RealScalar* subdiag, Index start, Index end, Scalar* matrixQ, Index n) {
773  using std::abs;
774  RealScalar td = (diag[end - 1] - diag[end]) * RealScalar(0.5);
775  RealScalar e = subdiag[end - 1];
776  // Note that thanks to scaling, e^2 or td^2 cannot overflow, however they can still
777  // underflow thus leading to inf/NaN values when using the following commented code:
778  // RealScalar e2 = numext::abs2(subdiag[end-1]);
779  // RealScalar mu = diag[end] - e2 / (td + (td>0 ? 1 : -1) * sqrt(td*td + e2));
780  // This explain the following, somewhat more complicated, version:
781  RealScalar mu = diag[end];
782  if (td == RealScalar(0))
783  mu -= abs(e);
784  else {
785  RealScalar e2 = numext::abs2(subdiag[end - 1]);
786  RealScalar h = numext::hypot(td, e);
787  if (e2 == RealScalar(0)) mu -= (e / (td + (td > RealScalar(0) ? RealScalar(1) : RealScalar(-1)))) * (e / h);
788  else mu -= e2 / (td + (td > RealScalar(0) ? h : -h));
789  }
790 
791  RealScalar x = diag[start] - mu;
792  RealScalar z = subdiag[start];
793  for (Index k = start; k < end; ++k) {
794  JacobiRotation<RealScalar> rot;
795  rot.makeGivens(x, z);
796 
797  // do T = G' T G
798  RealScalar sdk = rot.s() * diag[k] + rot.c() * subdiag[k];
799  RealScalar dkp1 = rot.s() * subdiag[k] + rot.c() * diag[k + 1];
800 
801  diag[k] = rot.c() * (rot.c() * diag[k] - rot.s() * subdiag[k]) - rot.s() * (rot.c() * subdiag[k] - rot.s() * diag[k + 1]);
802  diag[k + 1] = rot.s() * sdk + rot.c() * dkp1;
803  subdiag[k] = rot.c() * sdk - rot.s() * dkp1;
804 
805 
806  if (k > start)
807  subdiag[k - 1] = rot.c() * subdiag[k - 1] - rot.s() * z;
808 
809  x = subdiag[k];
810 
811  if (k < end - 1) {
812  z = -rot.s() * subdiag[k + 1];
813  subdiag[k + 1] = rot.c() * subdiag[k + 1];
814  }
815
