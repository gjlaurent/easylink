21  inline Packet8i pshiftleft(Packet8i v, int n) {
22 #ifdef EIGEN_VECTORIZE_AVX2
23  return _mm256_slli_epi32(v, n);
24 #else
25  __m128i lo = _mm_slli_epi32(_mm256_extractf128_si256(v, 0), n);
26  __m128i hi = _mm_slli_epi32(_mm256_extractf128_si256(v, 1), n);
27  return _mm256_insertf128_si256(_mm256_castsi128_si256(lo), (hi), 1);
28 #endif
29  }

31  inline Packet8f pshiftright(Packet8f v, int n) {
32 #ifdef EIGEN_VECTORIZE_AVX2
33  return _mm256_cvtepi32_ps(_mm256_srli_epi32(_mm256_castps_si256(v), n));
34 #else
35  __m128i lo = _mm_srli_epi32(_mm256_extractf128_si256(_mm256_castps_si256(v), 0), n);
36  __m128i hi = _mm_srli_epi32(_mm256_extractf128_si256(_mm256_castps_si256(v), 1), n);
37  return _mm256_cvtepi32_ps(_mm256_insertf128_si256(_mm256_castsi128_si256(lo), (hi), 1));
38 #endif
39  }

41  // Sine function
42  // Computes sin(x) by wrapping x to the interval [-Pi/4,3*Pi/4] and
43  // evaluating interpolants in [-Pi/4,Pi/4] or [Pi/4,3*Pi/4]. The interpolants
44  // are (anti-)symmetric and thus have only odd/even coefficients
45 
46  template <>
47  EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS EIGEN_UNUSED Packet8f
48  psin<Packet8f>(const Packet8f& _x) {
49  Packet8f x = _x;
50 
51  // Some useful values.
52  _EIGEN_DECLARE_CONST_Packet8i(one, 1);
53  _EIGEN_DECLARE_CONST_Packet8f(one, 1.0f);
54  _EIGEN_DECLARE_CONST_Packet8f(two, 2.0f);
55  _EIGEN_DECLARE_CONST_Packet8f(one_over_four, 0.25f);
56  _EIGEN_DECLARE_CONST_Packet8f(one_over_pi, 3.183098861837907e-01f);
57  _EIGEN_DECLARE_CONST_Packet8f(neg_pi_first, -3.140625000000000e+00f);
58  _EIGEN_DECLARE_CONST_Packet8f(neg_pi_second, -9.670257568359375e-04f);
59  _EIGEN_DECLARE_CONST_Packet8f(neg_pi_third, -6.278329571784980e-07f);
60  _EIGEN_DECLARE_CONST_Packet8f(four_over_pi, 1.273239544735163e+00f);
61 
62  // Map x from [-Pi/4,3*Pi/4] to z in [-1,3] and subtract the shifted period.
63  Packet8f z = pmul(x, p8f_one_over_pi);
64  Packet8f shift = _mm256_floor_ps(padd(z, p8f_one_over_four));
65  x = pmadd(shift, p8f_neg_pi_first, x);
66  x = pmadd(shift, p8f_neg_pi_second, x);
67  x = pmadd(shift, p8f_neg_pi_third, x);
68  z = pmul(x, p8f_four_over_pi);
69 
70  // Make a mask for the entries that need flipping, i.e. wherever the shift
71  // is odd.
72  Packet8i shift_ints = _mm256_cvtps_epi32(shift);
73  Packet8i shift_isodd = _mm256_castps_si256(_mm256_and_ps(_mm256_castsi256_ps(shift_ints), _mm256_castsi256_ps(p8i_one)));
74  Packet8i sign_flip_mask = pshiftleft(shift_isodd, 31);
75 
76  // Create a mask for which interpolant to use, i.e. if z > 1, then the mask
77  // is set to ones for that entry.
78  Packet8f ival_mask = _mm256_cmp_ps(z, p8f_one, _CMP_GT_OQ);
79 
80  // Evaluate the polynomial for the interval [1,3] in z.
81  _EIGEN_DECLARE_CONST_Packet8f(coeff_right_0, 9.999999724233232e-01f);
82  _EIGEN_DECLARE_CONST_Packet8f(coeff_right_2, -3.084242535619928e-01f);
83  _EIGEN_DECLARE_CONST_Packet8f(coeff_right_4, 1.584991525700324e-02f);
84  _EIGEN_DECLARE_CONST_Packet8f(coeff_right_6, -3.188805084631342e-04f);
85  Packet8f z_minus_two = psub(z, p8f_two);
86  Packet8f z_minus_two2 = pmul(z_minus_two, z_minus_two);
87  Packet8f right = pmadd(p8f_coeff_right_6, z_minus_two2, p8f_coeff_right_4);
88  right = pmadd(right, z_minus_two2, p8f_coeff_right_2);
89  right = pmadd(right, z_minus_two2, p8f_coeff_right_0);
90 
91  // Evaluate the polynomial for the interval [-1,1] in z.
92  _EIGEN_DECLARE_CONST_Packet8f(coeff_left_1, 7.853981525427295e-01f);
93  _EIGEN_DECLARE_CONST_Packet8f(coeff_left_3, -8.074536727092352e-02f);
94  _EIGEN_DECLARE_CONST_Packet8f(coeff_left_5, 2.489871967827018e-03f);
95  _EIGEN_DECLARE_CONST_Packet8f(coeff_left_7, -3.587725841214251e-05f);
96  Packet8f z2 = pmul(z, z);
97  Packet8f left = pmadd(p8f_coeff_left_7, z2, p8f_coeff_left_5);
98  left = pmadd(left, z2, p8f_coeff_left_3);
99  left = pmadd(left, z2, p8f_coeff_left_1);
100  left = pmul(left, z);
101 
102  // Assemble the results, i.e. select the left and right polynomials.
103  left = _mm256_andnot_ps(ival_mask, left);
104  right = _mm256_and_ps(ival_mask, right);
105  Packet8f res = _mm256_or_ps(left, right);
106 
107  // Flip the sign on the odd intervals and return the result.
108  res = _mm256_xor_ps(res, _mm256_castsi256_ps(sign_flip_mask));
109  return res;
110  }

112  // Natural logarithm
113  // Computes log(x) as log(2^e * m) = C*e + log(m), where the constant C =log(2)
114  // and m is in the range [sqrt(1/2),sqrt(2)). In this range, the logarithm can
115  // be easily approximated by a polynomial centered on m=1 for stability.
116  // TODO(gonnet): Further reduce the interval allowing for lower-degree
117  // polynomial interpolants -> ... -> profit!
118 
119  template <>
120  EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS EIGEN_UNUSED Packet8f
121  plog<Packet8f>(const Packet8f& _x) {
122  Packet8f x = _x;
123  _EIGEN_DECLARE_CONST_Packet8f(1, 1.0f);
124  _EIGEN_DECLARE_CONST_Packet8f(half, 0.5f);
125  _EIGEN_DECLARE_CONST_Packet8f(126f, 126.0f);
126 
127  _EIGEN_DECLARE_CONST_Packet8f_FROM_INT(inv_mant_mask, ~0x7f800000);
128 
129  // The smallest non denormalized float number.
130  _EIGEN_DECLARE_CONST_Packet8f_FROM_INT(min_norm_pos, 0x00800000);
131  _EIGEN_DECLARE_CONST_Packet8f_FROM_INT(minus_inf, 0xff800000);
132 
133  // Polynomial coefficients.
134  _EIGEN_DECLARE_CONST_Packet8f(cephes_SQRTHF, 0.707106781186547524f);
135  _EIGEN_DECLARE_CONST_Packet8f(cephes_log_p0, 7.0376836292E-2f);
136  _EIGEN_DECLARE_CONST_Packet8f(cephes_log_p1, -1.1514610310E-1f);
137  _EIGEN_DECLARE_CONST_Packet8f(cephes_log_p2, 1.1676998740E-1f);
138  _EIGEN_DECLARE_CONST_Packet8f(cephes_log_p3, -1.2420140846E-1f);
139  _EIGEN_DECLARE_CONST_Packet8f(cephes_log_p4, +1.4249322787E-1f);
140  _EIGEN_DECLARE_CONST_Packet8f(cephes_log_p5, -1.6668057665E-1f);
141  _EIGEN_DECLARE_CONST_Packet8f(cephes_log_p6, +2.0000714765E-1f);
142  _EIGEN_DECLARE_CONST_Packet8f(cephes_log_p7, -2.4999993993E-1f);
143  _EIGEN_DECLARE_CONST_Packet8f(cephes_log_p8, +3.3333331174E-1f);
144  _EIGEN_DECLARE_CONST_Packet8f(cephes_log_q1, -2.12194440e-4f);
145  _EIGEN_DECLARE_CONST_Packet8f(cephes_log_q2, 0.693359375f);
146 
147  Packet8f invalid_mask = _mm256_cmp_ps(x, _mm256_setzero_ps(), _CMP_NGE_UQ); // not greater equal is true if x is NaN
148  Packet8f iszero_mask = _mm256_cmp_ps(x, _mm256_setzero_ps(), _CMP_EQ_OQ);
149 
150  // Truncate input values to the minimum positive normal.
151  x = pmax(x, p8f_min_norm_pos);
152 
153  Packet8f emm0 = pshiftright(x, 23);
154  Packet8f e = _mm256_sub_ps(emm0, p8f_126f);
155 
156  // Set the exponents to -1, i.e. x are in the range [0.5,1).
157  x = _mm256_and_ps(x, p8f_inv_mant_mask);
158  x = _mm256_or_ps(x, p8f_half);
159 
160  // part2: Shift the inputs from the range [0.5,1) to [sqrt(1/2),sqrt(2))
161  // and shift by -1. The values are then centered around 0, which improves
162  // the stability of the polynomial evaluation.
163  // if( x < SQRTHF ) {
164  // e -= 1;
165  // x = x + x - 1.0;
166  // } else { x = x - 1.0; }
167  Packet8f mask = _mm256_cmp_ps(x, p8f_cephes_SQRTHF, _CMP_LT_OQ);
168  Packet8f tmp = _mm256_and_ps(x, mask);
169  x = psub(x, p8f_1);
170  e = psub(e, _mm256_and_ps(p8f_1, mask));
171  x = padd(x, tmp);
172 
173  Packet8f x2 = pmul(x, x);
174  Packet8f x3 = pmul(x2, x);
175 
176  // Evaluate the polynomial approximant of degree 8 in three parts, probably
177  // to improve instruction-level parallelism.
178  Packet8f y, y1, y2;
179  y = pmadd(p8f_cephes_log_p0, x, p8f_cephes_log_p1);
180  y1 = pmadd(p8f_cephes_log_p3, x, p8f_cephes_log_p4);
181  y2 = pmadd(p8f_cephes_log_p6, x, p8f_cephes_log_p7);
182  y = pmadd(y, x, p8f_cephes_log_p2);
183  y1 = pmadd(y1, x, p8f_cephes_log_p5);
184  y2 = pmadd(y2, x, p8f_cephes_log_p8);
185  y = pmadd(y, x3, y1);
186  y = pmadd(y, x3, y2);
187  y = pmul(y, x3);
188 
189  // Add the logarithm of the exponent back to the result of the interpolation.
190  y1 = pmul(e, p8f_cephes_log_q1);
191  tmp = pmul(x2, p8f_half);
192  y = padd(y, y1);
193  x = psub(x, tmp);
194  y2 = pmul(e, p8f_cephes_log_q2);
195  x = padd(x, y);
196  x = padd(x, y2);
197 
198  // Filter out invalid inputs, i.e. negative arg will be NAN, 0 will be -INF.
199  return _mm256_or_ps(
200  _mm256_andnot_ps(iszero_mask, _mm256_or_ps(x, invalid_mask)),
201  _mm256_and_ps(iszero_mask, p8f_minus_inf));
202  }

204  // Exponential function. Works by writing "x = m*log(2) + r" where
205  // "m = floor(x/log(2)+1/2)" and "r" is the remainder. The result is then
206  // "exp(x) = 2^m*exp(r)" where exp(r) is in the range [-1,1).
207 
208  template <>
209  EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS EIGEN_UNUSED Packet8f
210  pexp<Packet8f>(const Packet8f& _x) {
211  _EIGEN_DECLARE_CONST_Packet8f(1, 1.0f);
212  _EIGEN_DECLARE_CONST_Packet8f(half, 0.5f);
213  _EIGEN_DECLARE_CONST_Packet8f(127, 127.0f);
214 
215  _EIGEN_DECLARE_CONST_Packet8f(exp_hi, 88.3762626647950f);
216  _EIGEN_DECLARE_CONST_Packet8f(exp_lo, -88.3762626647949f);
217 
218  _EIGEN_DECLARE_CONST_Packet8f(cephes_LOG2EF, 1.44269504088896341f);
219 
220  _EIGEN_DECLARE_CONST_Packet8f(cephes_exp_p0, 1.9875691500E-4f);
221  _EIGEN_DECLARE_CONST_Packet8f(cephes_exp_p1, 1.3981999507E-3f);
222  _EIGEN_DECLARE_CONST_Packet8f(cephes_exp_p2, 8.3334519073E-3f);
223  _EIGEN_DECLARE_CONST_Packet8f(cephes_exp_p3, 4.1665795894E-2f);
224  _EIGEN_DECLARE_CONST_Packet8f(cephes_exp_p4, 1.6666665459E-1f);
225  _EIGEN_DECLARE_CONST_Packet8f(cephes_exp_p5, 5.0000001201E-1f);
226 
227  // Clamp x.
228  Packet8f x = pmax(pmin(_x, p8f_exp_hi), p8f_exp_lo);
229 
230  // Express exp(x) as exp(m*ln(2) + r), start by extracting
231  // m = floor(x/ln(2) + 0.5).
232  Packet8f m = _mm256_floor_ps(pmadd(x, p8f_cephes_LOG2EF, p8f_half));
233 
234  // Get r = x - m*ln(2). If no FMA instructions are available, m*ln(2) is
235  // subtracted out in two parts, m*C1+m*C2 = m*ln(2), to avoid accumulating
236  // truncation errors. Note that we don't use the "pmadd" function here to
237  // ensure that a precision-preserving FMA instruction is used.
238 #ifdef EIGEN_VECTORIZE_FMA
239  _EIGEN_DECLARE_CONST_Packet8f(nln2, -0.6931471805599453f);
240  Packet8f r = _mm256_fmadd_ps(m, p8f_nln2, x);
241 #else
242  _EIGEN_DECLARE_CONST_Packet8f(cephes_exp_C1, 0.693359375f);
243  _EIGEN_DECLARE_CONST_Packet8f(cephes_exp_C2, -2.12194440e-4f);
244  Packet8f r = psub(x, pmul(m, p8f_cephes_exp_C1));
245  r = psub(r, pmul(m, p8f_cephes_exp_C2));
246 #endif
247 
248  Packet8f r2 = pmul(r, r);
249 
250  // TODO(gonnet): Split into odd/even polynomials and try to exploit
251  // instruction-level parallelism.
252  Packet8f y = p8f_cephes_exp_p0;
253  y = pmadd(y, r, p8f_cephes_exp_p1);
254  y = pmadd(y, r, p8f_cephes_exp_p2);
255  y = pmadd(y, r, p8f_cephes_exp_p3);
256  y = pmadd(y, r, p8f_cephes_exp_p4);
257  y = pmadd(y, r, p8f_cephes_exp_p5);
258  y = pmadd(y, r2, r);
259  y = padd(y, p8f_1);
260 
261  // Build emm0 = 2^m.
262  Packet8i emm0 = _mm256_cvttps_epi32(padd(m, p8f_127));
263  emm0 = pshiftleft(emm0, 23);
264 
265  // Return 2^m * exp(r).
266  return pmax(pmul(y, _mm256_castsi256_ps(emm0)), _x);
267  }

277  template <>
278  EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS EIGEN_UNUSED Packet4d
279  pexp<Packet4d>(const Packet4d& _x) {
280  Packet4d x = _x;
281 
282  _EIGEN_DECLARE_CONST_Packet4d(1, 1.0);
283  _EIGEN_DECLARE_CONST_Packet4d(2, 2.0);
284  _EIGEN_DECLARE_CONST_Packet4d(half, 0.5);
285 
286  _EIGEN_DECLARE_CONST_Packet4d(exp_hi, 709.437);
287  _EIGEN_DECLARE_CONST_Packet4d(exp_lo, -709.436139303);
288 
289  _EIGEN_DECLARE_CONST_Packet4d(cephes_LOG2EF, 1.4426950408889634073599);
290 
291  _EIGEN_DECLARE_CONST_Packet4d(cephes_exp_p0, 1.26177193074810590878e-4);
292  _EIGEN_DECLARE_CONST_Packet4d(cephes_exp_p1, 3.02994407707441961300e-2);
293  _EIGEN_DECLARE_CONST_Packet4d(cephes_exp_p2, 9.99999999999999999910e-1);
294 
295  _EIGEN_DECLARE_CONST_Packet4d(cephes_exp_q0, 3.00198505138664455042e-6);
296  _EIGEN_DECLARE_CONST_Packet4d(cephes_exp_q1, 2.52448340349684104192e-3);
297  _EIGEN_DECLARE_CONST_Packet4d(cephes_exp_q2, 2.27265548208155028766e-1);
298  _EIGEN_DECLARE_CONST_Packet4d(cephes_exp_q3, 2.00000000000000000009e0);
299 
300  _EIGEN_DECLARE_CONST_Packet4d(cephes_exp_C1, 0.693145751953125);
301  _EIGEN_DECLARE_CONST_Packet4d(cephes_exp_C2, 1.42860682030941723212e-6);
302  _EIGEN_DECLARE_CONST_Packet4i(1023, 1023);
303 
304  Packet4d tmp, fx;
305 
306  // clamp x
307  x = pmax(pmin(x, p4d_exp_hi), p4d_exp_lo);
308  // Express exp(x) as exp(g + n*log(2)).
309  fx = pmadd(p4d_cephes_LOG2EF, x, p4d_half);
310 
311  // Get the integer modulus of log(2), i.e. the "n" described above.
312  fx = _mm256_floor_pd(fx);
313 
314  // Get the remainder modulo log(2), i.e. the "g" described above. Subtract
315  // n*log(2) out in two steps, i.e. n*C1 + n*C2, C1+C2=log2 to get the last
316  // digits right.
317  tmp = pmul(fx, p4d_cephes_exp_C1);
318  Packet4d z = pmul(fx, p4d_cephes_exp_C2);
319  x = psub(x, tmp);
320  x = psub(x, z);
321 
322  Packet4d x2 = pmul(x, x);
323 
324  // Evaluate the numerator polynomial of the rational interpolant.
325  Packet4d px = p4d_cephes_exp_p0;
326  px = pmadd(px, x2, p4d_cephes_exp_p1);
327  px = pmadd(px, x2, p4d_cephes_exp_p2);
328  px = pmul(px, x);
329 
330  // Evaluate the denominator polynomial of the rational interpolant.
331  Packet4d qx = p4d_cephes_exp_q0;
332  qx = pmadd(qx, x2, p4d_cephes_exp_q1);
333  qx = pmadd(qx, x2, p4d_cephes_exp_q2);
334  qx = pmadd(qx, x2, p4d_cephes_exp_q3);
335 
336  // I don't really get this bit, copied from the SSE2 routines, so...
337  // TODO(gonnet): Figure out what is going on here, perhaps find a better
338  // rational interpolant?
339  x = _mm256_div_pd(px, psub(qx, px));
340  x = pmadd(p4d_2, x, p4d_1);
341 
342  // Build e=2^n by constructing the exponents in a 128-bit vector and
343  // shifting them to where they belong in double-precision values.
344  __m128i emm0 = _mm256_cvtpd_epi32(fx);
345  emm0 = _mm_add_epi32(emm0, p4i_1023);
346  emm0 = _mm_shuffle_epi32(emm0, _MM_SHUFFLE(3, 1, 2, 0));
347  __m128i lo = _mm_slli_epi64(emm0, 52);
348  __m128i hi = _mm_slli_epi64(_mm_srli_epi64(emm0, 32), 52);
349  __m256i e = _mm256_insertf128_si256(_mm256_setzero_si256(), lo, 0);
350  e = _mm256_insertf128_si256(e, hi, 1);
351 
352  // Construct the result 2^n * exp(g) = e * x. The max is used to catch
353  // non-finite values in the input.
354  return pmax(pmul(x, _mm256_castsi256_pd(e)), _x);
355  }

357  // Functions for sqrt.
358  // The EIGEN_FAST_MATH version uses the _mm_rsqrt_ps approximation and one step
359  // of Newton's method, at a cost of 1-2 bits of precision as opposed to the
360  // exact solution. It does not handle +inf, or denormalized numbers correctly.
361  // The main advantage of this approach is not just speed, but also the fact that
362  // it can be inlined and pipelined with other computations, further reducing its
363  // effective latency. This is similar to Quake3's fast inverse square root.
364  // For detail see here: http://www.beyond3d.com/content/articles/8/
365 #if EIGEN_FAST_MATH
366 
367 template <>
368  EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS EIGEN_UNUSED Packet8f
369  psqrt<Packet8f>(const Packet8f& _x) {
370  Packet8f half = pmul(_x, pset1<Packet8f>(.5f));
371  Packet8f denormal_mask = _mm256_and_ps(
372  _mm256_cmp_ps(_x, pset1<Packet8f>((std::numeric_limits<float>::min)()),
373  _CMP_LT_OQ),
374  _mm256_cmp_ps(_x, _mm256_setzero_ps(), _CMP_GE_OQ));
375 
376  // Compute approximate reciprocal sqrt.
377  Packet8f x = _mm256_rsqrt_ps(_x);
378  // Do a single step of Newton's iteration.
379  x = pmul(x, psub(pset1<Packet8f>(1.5f), pmul(half, pmul(x, x))));
380  // Flush results for denormals to zero.
381  return _mm256_andnot_ps(denormal_mask, pmul(_x, x));
382  }
383 #else

397  template<> EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS EIGEN_UNUSED
398  Packet8f prsqrt<Packet8f>(const Packet8f& _x) {
399  _EIGEN_DECLARE_CONST_Packet8f_FROM_INT(inf, 0x7f800000);
400  _EIGEN_DECLARE_CONST_Packet8f_FROM_INT(nan, 0x7fc00000);
401  _EIGEN_DECLARE_CONST_Packet8f(one_point_five, 1.5f);
402  _EIGEN_DECLARE_CONST_Packet8f(minus_half, -0.5f);
403  _EIGEN_DECLARE_CONST_Packet8f_FROM_INT(flt_min, 0x00800000);
404 
405  Packet8f neg_half = pmul(_x, p8f_minus_half);
406 
407  // select only the inverse sqrt of positive normal inputs (denormals are
408  // flushed to zero and cause infs as well).
409  Packet8f le_zero_mask = _mm256_cmp_ps(_x, p8f_flt_min, _CMP_LT_OQ);
410  Packet8f x = _mm256_andnot_ps(le_zero_mask, _mm256_rsqrt_ps(_x));
411 
412  // Fill in NaNs and Infs for the negative/zero entries.
413  Packet8f neg_mask = _mm256_cmp_ps(_x, _mm256_setzero_ps(), _CMP_LT_OQ);
414  Packet8f zero_mask = _mm256_andnot_ps(neg_mask, le_zero_mask);
415  Packet8f infs_and_nans = _mm256_or_ps(_mm256_and_ps(neg_mask, p8f_nan),
416  _mm256_and_ps(zero_mask, p8f_inf));
417 
418  // Do a single step of Newton's iteration.
419  x = pmul(x, pmadd(neg_half, pmul(x, x), p8f_one_point_five));
420 
421  // Insert NaNs and Infs in all the right places.
422  return _mm256_or_ps(x, infs_and_nans);
423  }
