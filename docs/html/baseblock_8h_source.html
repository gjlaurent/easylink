200  static void checkParameters()
201  {
202  for (int i=0;i<PARAMETER_NUMBER;i++)
203  {
204  if ((parameterType[i]==2)||(parameterType[i]==3))
205  {
206  // check if it is really a string
207  if (!mxIsChar(ssGetSFcnParam(simStruct,i)))
208  throw runtime_error("easyLink error: Parameter port " + toString(i) + " must be a string.");
209 
210  if (parameterType[i]==3)
211  {
212  // check if it is really a identifier
213  string str=getParameterString(i);
214  if (!isIdentifier(str))
215  throw runtime_error("easyLink error: Parameter port " + toString(i) + " must be a valid identifier.");
216  }
217  }
218  else
219  {
220  // check if it is a scalar or a vector
221  if (mxIsSparse(ssGetSFcnParam(simStruct,i))||!mxIsDouble(ssGetSFcnParam(simStruct,i)))
222  throw runtime_error("easyLink error: Parameter port " + toString(i) + " must be a scalar or an array of real values.");
223 
224  // check the number of rows
225  if ( (parameterRows[i]>0) && (mxGetM(ssGetSFcnParam(simStruct,i))!=parameterRows[i]) )
226  throw runtime_error("easyLink error: Parameter port " + toString(i) + " must have " + toString(parameterRows[i]) + " rows.");
227 
228  // check the number of cols
229  if ( (parameterCols[i]>0) && (mxGetN(ssGetSFcnParam(simStruct,i))!=parameterCols[i]) )
230  throw runtime_error("easyLink error: Parameter port " + toString(i) + " must have " + toString(parameterCols[i]) + " cols.");
231  }
232  }
233  }
