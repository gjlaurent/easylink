94  template<> EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS EIGEN_UNUSED
95  Packet4f plog<Packet4f>(const Packet4f& _x) {
96  Packet4f x = _x;
97 
98  Packet4i emm0;
99 
100  /* isvalid_mask is 0 if x < 0 or x is NaN. */
101  Packet4ui isvalid_mask = reinterpret_cast<Packet4ui> (vec_cmpge(x, p4f_ZERO));
102  Packet4ui iszero_mask = reinterpret_cast<Packet4ui> (vec_cmpeq(x, p4f_ZERO));
103 
104  x = pmax(x, p4f_min_norm_pos); /* cut off denormalized stuff */
105  emm0 = vec_sr(reinterpret_cast<Packet4i> (x),
106  reinterpret_cast<Packet4ui> (p4i_23));
107 
108  /* keep only the fractional part */
109  x = pand(x, p4f_inv_mant_mask);
110  x = por(x, p4f_half);
111 
112  emm0 = psub(emm0, p4i_0x7f);
113  Packet4f e = padd(vec_ctf(emm0, 0), p4f_1);
114 
115  /* part2:
116  if( x < SQRTHF ) {
117  e -= 1;
118  x = x + x - 1.0;
119  } else { x = x - 1.0; }
120  */
121  Packet4f mask = reinterpret_cast<Packet4f> (vec_cmplt(x, p4f_cephes_SQRTHF));
122  Packet4f tmp = pand(x, mask);
123  x = psub(x, p4f_1);
124  e = psub(e, pand(p4f_1, mask));
125  x = padd(x, tmp);
126 
127  Packet4f x2 = pmul(x, x);
128  Packet4f x3 = pmul(x2, x);
129 
130  Packet4f y, y1, y2;
131  y = pmadd(p4f_cephes_log_p0, x, p4f_cephes_log_p1);
132  y1 = pmadd(p4f_cephes_log_p3, x, p4f_cephes_log_p4);
133  y2 = pmadd(p4f_cephes_log_p6, x, p4f_cephes_log_p7);
134  y = pmadd(y, x, p4f_cephes_log_p2);
135  y1 = pmadd(y1, x, p4f_cephes_log_p5);
136  y2 = pmadd(y2, x, p4f_cephes_log_p8);
137  y = pmadd(y, x3, y1);
138  y = pmadd(y, x3, y2);
139  y = pmul(y, x3);
140 
141  y1 = pmul(e, p4f_cephes_log_q1);
142  tmp = pmul(x2, p4f_half);
143  y = padd(y, y1);
144  x = psub(x, tmp);
145  y2 = pmul(e, p4f_cephes_log_q2);
146  x = padd(x, y);
147  x = padd(x, y2);
148  // negative arg will be NAN, 0 will be -INF
149  x = vec_sel(x, p4f_minus_inf, iszero_mask);
150  x = vec_sel(p4f_minus_nan, x, isvalid_mask);
151  return x;
152  }

154  template<> EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS EIGEN_UNUSED
155  Packet4f pexp<Packet4f>(const Packet4f& _x) {
156  Packet4f x = _x;
157 
158  Packet4f tmp, fx;
159  Packet4i emm0;
160 
161  // clamp x
162  x = pmax(pmin(x, p4f_exp_hi), p4f_exp_lo);
163 
164  // express exp(x) as exp(g + n*log(2))
165  fx = pmadd(x, p4f_cephes_LOG2EF, p4f_half);
166 
167  fx = pfloor(fx);
168 
169  tmp = pmul(fx, p4f_cephes_exp_C1);
170  Packet4f z = pmul(fx, p4f_cephes_exp_C2);
171  x = psub(x, tmp);
172  x = psub(x, z);
173 
174  z = pmul(x, x);
175 
176  Packet4f y = p4f_cephes_exp_p0;
177  y = pmadd(y, x, p4f_cephes_exp_p1);
178  y = pmadd(y, x, p4f_cephes_exp_p2);
179  y = pmadd(y, x, p4f_cephes_exp_p3);
180  y = pmadd(y, x, p4f_cephes_exp_p4);
181  y = pmadd(y, x, p4f_cephes_exp_p5);
182  y = pmadd(y, z, x);
183  y = padd(y, p4f_1);
184 
185  // build 2^n
186  emm0 = vec_cts(fx, 0);
187  emm0 = vec_add(emm0, p4i_0x7f);
188  emm0 = vec_sl(emm0, reinterpret_cast<Packet4ui> (p4i_23));
189 
190  // Altivec's max & min operators just drop silent NaNs. Check NaNs in
191  // inputs and return them unmodified.
192  Packet4ui isnumber_mask = reinterpret_cast<Packet4ui> (vec_cmpeq(_x, _x));
193  return vec_sel(_x, pmax(pmul(y, reinterpret_cast<Packet4f> (emm0)), _x),
194  isnumber_mask);
195  }

224  // VSX support varies between different compilers and even different
225  // versions of the same compiler. For gcc version >= 4.9.3, we can use
226  // vec_cts to efficiently convert Packet2d to Packet2l. Otherwise, use
227  // a slow version that works with older compilers.
228  // Update: apparently vec_cts/vec_ctf intrinsics for 64-bit doubles
229  // are buggy, https://gcc.gnu.org/bugzilla/show_bug.cgi?id=70963
230 
231  static inline Packet2l ConvertToPacket2l(const Packet2d& x) {
232 #if EIGEN_GNUC_AT_LEAST(5, 4) || \
233  (EIGEN_GNUC_AT(6, 1) && __GNUC_PATCHLEVEL__ >= 1)
234  return vec_cts(x, 0); // TODO: check clang version.
235 #else
236  double tmp[2];
237  memcpy(tmp, &x, sizeof (tmp));
238  Packet2l l = {static_cast<long long> (tmp[0]),
239  static_cast<long long> (tmp[1])};
240  return l;
241 #endif
242  }

244  template<> EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS EIGEN_UNUSED
245  Packet2d pexp<Packet2d>(const Packet2d& _x) {
246  Packet2d x = _x;
247 
248  Packet2d tmp, fx;
249  Packet2l emm0;
250 
251  // clamp x
252  x = pmax(pmin(x, p2d_exp_hi), p2d_exp_lo);
253 
254  /* express exp(x) as exp(g + n*log(2)) */
255  fx = pmadd(x, p2d_cephes_LOG2EF, p2d_half);
256 
257  fx = pfloor(fx);
258 
259  tmp = pmul(fx, p2d_cephes_exp_C1);
260  Packet2d z = pmul(fx, p2d_cephes_exp_C2);
261  x = psub(x, tmp);
262  x = psub(x, z);
263 
264  Packet2d x2 = pmul(x, x);
265 
266  Packet2d px = p2d_cephes_exp_p0;
267  px = pmadd(px, x2, p2d_cephes_exp_p1);
268  px = pmadd(px, x2, p2d_cephes_exp_p2);
269  px = pmul(px, x);
270 
271  Packet2d qx = p2d_cephes_exp_q0;
272  qx = pmadd(qx, x2, p2d_cephes_exp_q1);
273  qx = pmadd(qx, x2, p2d_cephes_exp_q2);
274  qx = pmadd(qx, x2, p2d_cephes_exp_q3);
275 
276  x = pdiv(px, psub(qx, px));
277  x = pmadd(p2d_2, x, p2d_1);
278 
279  // build 2^n
280  emm0 = ConvertToPacket2l(fx);
281 
282 #ifdef __POWER8_VECTOR__
283  emm0 = vec_add(emm0, p2l_1023);
284  emm0 = vec_sl(emm0, p2ul_52);
285 #else
286  // Code is a bit complex for POWER7. There is actually a
287  // vec_xxsldi intrinsic but it is not supported by some gcc versions.
288  // So we shift (52-32) bits and do a word swap with zeros.
289  _EIGEN_DECLARE_CONST_Packet4i(1023, 1023);
290  _EIGEN_DECLARE_CONST_Packet4i(20, 20); // 52 - 32
291 
292  Packet4i emm04i = reinterpret_cast<Packet4i> (emm0);
293  emm04i = vec_add(emm04i, p4i_1023);
294  emm04i = vec_sl(emm04i, reinterpret_cast<Packet4ui> (p4i_20));
295  static const Packet16uc perm = {
296  0x14, 0x15, 0x16, 0x17, 0x00, 0x01, 0x02, 0x03,
297  0x1c, 0x1d, 0x1e, 0x1f, 0x08, 0x09, 0x0a, 0x0b
298  };
299 #ifdef _BIG_ENDIAN
300  emm0 = reinterpret_cast<Packet2l> (vec_perm(p4i_ZERO, emm04i, perm));
301 #else
302  emm0 = reinterpret_cast<Packet2l> (vec_perm(emm04i, p4i_ZERO, perm));
303 #endif
304 
305 #endif
306 
307  // Altivec's max & min operators just drop silent NaNs. Check NaNs in
308  // inputs and return them unmodified.
309  Packet2ul isnumber_mask = reinterpret_cast<Packet2ul> (vec_cmpeq(_x, _x));
310  return vec_sel(_x, pmax(pmul(x, reinterpret_cast<Packet2d> (emm0)), _x),
311  isnumber_mask);
312  }
313 #endif
