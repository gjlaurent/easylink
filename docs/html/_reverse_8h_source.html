145  template<typename Derived>
146  inline void DenseBase<Derived>::reverseInPlace() {
147  if (cols() > rows()) {
148  Index half = cols() / 2;
149  leftCols(half).swap(rightCols(half).reverse());
150  if ((cols() % 2) == 1) {
151  Index half2 = rows() / 2;
152  col(half).head(half2).swap(col(half).tail(half2).reverse());
153  }
154  } else {
155  Index half = rows() / 2;
156  topRows(half).swap(bottomRows(half).reverse());
157  if ((rows() % 2) == 1) {
158  Index half2 = cols() / 2;
159  row(half).head(half2).swap(row(half).tail(half2).reverse());
160  }
161  }
162  }
