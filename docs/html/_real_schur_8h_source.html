243  RealSchur<MatrixType>& RealSchur<MatrixType>::compute(const EigenBase<InputType>& matrix, bool computeU) {
244  const Scalar considerAsZero = (std::numeric_limits<Scalar>::min)();
245 
246  eigen_assert(matrix.cols() == matrix.rows());
247  Index maxIters = m_maxIters;
248  if (maxIters == -1)
249  maxIters = m_maxIterationsPerRow * matrix.rows();
250 
251  Scalar scale = matrix.derived().cwiseAbs().maxCoeff();
252  if (scale < considerAsZero) {
253  m_matT.setZero(matrix.rows(), matrix.cols());
254  if (computeU)
255  m_matU.setIdentity(matrix.rows(), matrix.cols());
256  m_info = Success;
257  m_isInitialized = true;
258  m_matUisUptodate = computeU;
259  return *this;
260  }
261 
262  // Step 1. Reduce to Hessenberg form
263  m_hess.compute(matrix.derived() / scale);
264 
265  // Step 2. Reduce to real Schur form
266  computeFromHessenberg(m_hess.matrixH(), m_hess.matrixQ(), computeU);
267 
268  m_matT *= scale;
269 
270  return *this;
271  }

275  RealSchur<MatrixType>& RealSchur<MatrixType>::computeFromHessenberg(const HessMatrixType& matrixH, const OrthMatrixType& matrixQ, bool computeU) {
276  using std::abs;
277 
278  m_matT = matrixH;
279  if (computeU)
280  m_matU = matrixQ;
281 
282  Index maxIters = m_maxIters;
283  if (maxIters == -1)
284  maxIters = m_maxIterationsPerRow * matrixH.rows();
285  m_workspaceVector.resize(m_matT.cols());
286  Scalar* workspace = &m_workspaceVector.coeffRef(0);
287 
288  // The matrix m_matT is divided in three parts.
289  // Rows 0,...,il-1 are decoupled from the rest because m_matT(il,il-1) is zero.
290  // Rows il,...,iu is the part we are working on (the active window).
291  // Rows iu+1,...,end are already brought in triangular form.
292  Index iu = m_matT.cols() - 1;
293  Index iter = 0; // iteration count for current eigenvalue
294  Index totalIter = 0; // iteration count for whole matrix
295  Scalar exshift(0); // sum of exceptional shifts
296  Scalar norm = computeNormOfT();
297 
298  if (norm != Scalar(0)) {
299  while (iu >= 0) {
300  Index il = findSmallSubdiagEntry(iu);
301 
302  // Check for convergence
303  if (il == iu) // One root found
304  {
305  m_matT.coeffRef(iu, iu) = m_matT.coeff(iu, iu) + exshift;
306  if (iu > 0)
307  m_matT.coeffRef(iu, iu - 1) = Scalar(0);
308  iu--;
309  iter = 0;
310  } else if (il == iu - 1) // Two roots found
311  {
312  splitOffTwoRows(iu, computeU, exshift);
313  iu -= 2;
314  iter = 0;
315  } else // No convergence yet
316  {
317  // The firstHouseholderVector vector has to be initialized to something to get rid of a silly GCC warning (-O1 -Wall -DNDEBUG )
318  Vector3s firstHouseholderVector = Vector3s::Zero(), shiftInfo;
319  computeShift(iu, iter, exshift, shiftInfo);
320  iter = iter + 1;
321  totalIter = totalIter + 1;
322  if (totalIter > maxIters) break;
323  Index im;
324  initFrancisQRStep(il, iu, shiftInfo, im, firstHouseholderVector);
325  performFrancisQRStep(il, im, iu, computeU, firstHouseholderVector, workspace);
326  }
327  }
328  }
329  if (totalIter <= maxIters)
330  m_info = Success;
331  else
332  m_info = NoConvergence;
333 
334  m_isInitialized = true;
335  m_matUisUptodate = computeU;
336  return *this;
337  }

341  inline typename MatrixType::Scalar RealSchur<MatrixType>::computeNormOfT() {
342  const Index size = m_matT.cols();
343  // FIXME to be efficient the following would requires a triangular reduxion code
344  // Scalar norm = m_matT.upper().cwiseAbs().sum()
345  // + m_matT.bottomLeftCorner(size-1,size-1).diagonal().cwiseAbs().sum();
346  Scalar norm(0);
347  for (Index j = 0; j < size; ++j)
348  norm += m_matT.col(j).segment(0, (std::min)(size, j + 2)).cwiseAbs().sum();
349  return norm;

353  template<typename MatrixType>
354  inline Index RealSchur<MatrixType>::findSmallSubdiagEntry(Index iu) {
355  using std::abs;
356  Index res = iu;
357  while (res > 0) {
358  Scalar s = abs(m_matT.coeff(res - 1, res - 1)) + abs(m_matT.coeff(res, res));
359  if (abs(m_matT.coeff(res, res - 1)) <= NumTraits<Scalar>::epsilon() * s)
360  break;
361  res--;
362  }
363  return res;
364  }

367  template<typename MatrixType>
368  inline void RealSchur<MatrixType>::splitOffTwoRows(Index iu, bool computeU, const Scalar& exshift) {
369  using std::sqrt;
370  using std::abs;
371  const Index size = m_matT.cols();
372 
373  // The eigenvalues of the 2x2 matrix [a b; c d] are
374  // trace +/- sqrt(discr/4) where discr = tr^2 - 4*det, tr = a + d, det = ad - bc
375  Scalar p = Scalar(0.5) * (m_matT.coeff(iu - 1, iu - 1) - m_matT.coeff(iu, iu));
376  Scalar q = p * p + m_matT.coeff(iu, iu - 1) * m_matT.coeff(iu - 1, iu); // q = tr^2 / 4 - det = discr/4
377  m_matT.coeffRef(iu, iu) += exshift;
378  m_matT.coeffRef(iu - 1, iu - 1) += exshift;
379 
380  if (q >= Scalar(0)) // Two real eigenvalues
381  {
382  Scalar z = sqrt(abs(q));
383  JacobiRotation<Scalar> rot;
384  if (p >= Scalar(0))
385  rot.makeGivens(p + z, m_matT.coeff(iu, iu - 1));
386  else
387  rot.makeGivens(p - z, m_matT.coeff(iu, iu - 1));
388 
389  m_matT.rightCols(size - iu + 1).applyOnTheLeft(iu - 1, iu, rot.adjoint());
390  m_matT.topRows(iu + 1).applyOnTheRight(iu - 1, iu, rot);
391  m_matT.coeffRef(iu, iu - 1) = Scalar(0);
392  if (computeU)
393  m_matU.applyOnTheRight(iu - 1, iu, rot);
394  }
395 
396  if (iu > 1)
397  m_matT.coeffRef(iu - 1, iu - 2) = Scalar(0);
398  }

401  template<typename MatrixType>
402  inline void RealSchur<MatrixType>::computeShift(Index iu, Index iter, Scalar& exshift, Vector3s& shiftInfo) {
403  using std::sqrt;
404  using std::abs;
405  shiftInfo.coeffRef(0) = m_matT.coeff(iu, iu);
406  shiftInfo.coeffRef(1) = m_matT.coeff(iu - 1, iu - 1);
407  shiftInfo.coeffRef(2) = m_matT.coeff(iu, iu - 1) * m_matT.coeff(iu - 1, iu);
408 
409  // Wilkinson's original ad hoc shift
410  if (iter == 10) {
411  exshift += shiftInfo.coeff(0);
412  for (Index i = 0; i <= iu; ++i)
413  m_matT.coeffRef(i, i) -= shiftInfo.coeff(0);
414  Scalar s = abs(m_matT.coeff(iu, iu - 1)) + abs(m_matT.coeff(iu - 1, iu - 2));
415  shiftInfo.coeffRef(0) = Scalar(0.75) * s;
416  shiftInfo.coeffRef(1) = Scalar(0.75) * s;
417  shiftInfo.coeffRef(2) = Scalar(-0.4375) * s * s;
418  }
419 
420  // MATLAB's new ad hoc shift
421  if (iter == 30) {
422  Scalar s = (shiftInfo.coeff(1) - shiftInfo.coeff(0)) / Scalar(2.0);
423  s = s * s + shiftInfo.coeff(2);
424  if (s > Scalar(0)) {
425  s = sqrt(s);
426  if (shiftInfo.coeff(1) < shiftInfo.coeff(0))
427  s = -s;
428  s = s + (shiftInfo.coeff(1) - shiftInfo.coeff(0)) / Scalar(2.0);
429  s = shiftInfo.coeff(0) - shiftInfo.coeff(2) / s;
430  exshift += s;
431  for (Index i = 0; i <= iu; ++i)
432  m_matT.coeffRef(i, i) -= s;
433  shiftInfo.setConstant(Scalar(0.964));
434  }
435  }
436  }

439  template<typename MatrixType>
440  inline void RealSchur<MatrixType>::initFrancisQRStep(Index il, Index iu, const Vector3s& shiftInfo, Index& im, Vector3s& firstHouseholderVector) {
441  using std::abs;
442  Vector3s& v = firstHouseholderVector; // alias to save typing
443 
444  for (im = iu - 2; im >= il; --im) {
445  const Scalar Tmm = m_matT.coeff(im, im);
446  const Scalar r = shiftInfo.coeff(0) - Tmm;
447  const Scalar s = shiftInfo.coeff(1) - Tmm;
448  v.coeffRef(0) = (r * s - shiftInfo.coeff(2)) / m_matT.coeff(im + 1, im) + m_matT.coeff(im, im + 1);
449  v.coeffRef(1) = m_matT.coeff(im + 1, im + 1) - Tmm - r - s;
450  v.coeffRef(2) = m_matT.coeff(im + 2, im + 1);
451  if (im == il) {
452  break;
453  }
454  const Scalar lhs = m_matT.coeff(im, im - 1) * (abs(v.coeff(1)) + abs(v.coeff(2)));
455  const Scalar rhs = v.coeff(0) * (abs(m_matT.coeff(im - 1, im - 1)) + abs(Tmm) + abs(m_matT.coeff(im + 1, im + 1)));
456  if (abs(lhs) < NumTraits<Scalar>::epsilon() * rhs)
457  break;
458  }
459  }

462  template<typename MatrixType>
463  inline void RealSchur<MatrixType>::performFrancisQRStep(Index il, Index im, Index iu, bool computeU, const Vector3s& firstHouseholderVector, Scalar* workspace) {
464  eigen_assert(im >= il);
465  eigen_assert(im <= iu - 2);
466 
467  const Index size = m_matT.cols();
468 
469  for (Index k = im; k <= iu - 2; ++k) {
470  bool firstIteration = (k == im);
471 
472  Vector3s v;
473  if (firstIteration)
474  v = firstHouseholderVector;
475  else
476  v = m_matT.template block<3, 1>(k, k - 1);
477 
478  Scalar tau, beta;
479  Matrix<Scalar, 2, 1> ess;
480  v.makeHouseholder(ess, tau, beta);
481 
482  if (beta != Scalar(0)) // if v is not zero
483  {
484  if (firstIteration && k > il)
485  m_matT.coeffRef(k, k - 1) = -m_matT.coeff(k, k - 1);
486  else if (!firstIteration)
487  m_matT.coeffRef(k, k - 1) = beta;
488 
489  // These Householder transformations form the O(n^3) part of the algorithm
490  m_matT.block(k, k, 3, size - k).applyHouseholderOnTheLeft(ess, tau, workspace);
491  m_matT.block(0, k, (std::min)(iu, k + 3) + 1, 3).applyHouseholderOnTheRight(ess, tau, workspace);
492  if (computeU)
493  m_matU.block(0, k, size, 3).applyHouseholderOnTheRight(ess, tau, workspace);
494  }
495  }
496 
497  Matrix<Scalar, 2, 1> v = m_matT.template block<2, 1>(iu - 1, iu - 2);
498  Scalar tau, beta;
499  Matrix<Scalar, 1, 1> ess;
500  v.makeHouseholder(ess, tau, beta);
501 
502  if (beta != Scalar(0)) // if v is not zero
503  {
504  m_matT.coeffRef(iu - 1, iu - 2) = beta;
505  m_matT.block(iu - 1, iu - 1, 2, size - iu + 1).applyHouseholderOnTheLeft(ess, tau, workspace);
506  m_matT.block(0, iu - 1, iu + 1, 2).applyHouseholderOnTheRight(ess, tau, workspace);
507  if (computeU)
508  m_matU.block(0, iu - 1, size, 2).applyHouseholderOnTheRight(ess, tau, workspace);
509  }
510 
511  // clean up pollution due to round-off errors
512  for (Index i = im + 2; i <= iu; ++i) {
513  m_matT.coeffRef(i, i - 2) = Scalar(0);
514  if (i > im + 2)
515  m_matT.coeffRef(i, i - 3) = Scalar(0);
516  }
517  }
