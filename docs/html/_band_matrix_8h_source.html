76  inline Block<CoefficientsType, Dynamic, 1> col(Index i) {
77  EIGEN_STATIC_ASSERT((Options & RowMajor) == 0, THIS_METHOD_IS_ONLY_FOR_COLUMN_MAJOR_MATRICES);
78  Index start = 0;
79  Index len = coeffs().rows();
80  if (i <= supers()) {
81  start = supers() - i;
82  len = (std::min)(rows(), std::max<Index>(0, coeffs().rows() - (supers() - i)));
83  } else if (i >= rows() - subs())
84  len = std::max<Index>(0, coeffs().rows() - (i + 1 - rows() + subs()));
85  return Block<CoefficientsType, Dynamic, 1>(coeffs(), start, i, len, 1);
86  }

98  template<int Index> struct DiagonalIntReturnType {
99 
100  enum {
101  ReturnOpposite = (Options&SelfAdjoint) && (((Index) > 0 && Supers == 0) || ((Index) < 0 && Subs == 0)),
102  Conjugate = ReturnOpposite && NumTraits<Scalar>::IsComplex,
103  ActualIndex = ReturnOpposite ? -Index : Index,
104  DiagonalSize = (RowsAtCompileTime == Dynamic || ColsAtCompileTime == Dynamic)
105  ? Dynamic
106  : (ActualIndex < 0
107  ? EIGEN_SIZE_MIN_PREFER_DYNAMIC(ColsAtCompileTime, RowsAtCompileTime + ActualIndex)
108  : EIGEN_SIZE_MIN_PREFER_DYNAMIC(RowsAtCompileTime, ColsAtCompileTime - ActualIndex))
109  };
110  typedef Block<CoefficientsType, 1, DiagonalSize> BuildType;
111  typedef typename internal::conditional<Conjugate,
112  CwiseUnaryOp<internal::scalar_conjugate_op<Scalar>, BuildType >,
113  BuildType>::type Type;
114  };

128  eigen_assert((i < 0 && -i <= subs()) || (i >= 0 && i <= supers()));

138  template<typename Dest> inline void evalTo(Dest& dst) const {
139  dst.resize(rows(), cols());
140  dst.setZero();
141  dst.diagonal() = diagonal();
142  for (Index i = 1; i <= supers(); ++i)
143  dst.diagonal(i) = diagonal(i);
144  for (Index i = 1; i <= subs(); ++i)
145  dst.diagonal(-i) = diagonal(-i);
146  }

148  DenseMatrixType toDenseMatrix() const {

150  evalTo(res);

152  }
