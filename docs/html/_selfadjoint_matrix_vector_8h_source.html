18  * This algorithm processes 2 columns at onces that allows to both reduce

37  template<typename Scalar, typename Index, int StorageOrder, int UpLo, bool ConjugateLhs, bool ConjugateRhs, int Version>
38  EIGEN_DONT_INLINE void selfadjoint_matrix_vector_product<Scalar, Index, StorageOrder, UpLo, ConjugateLhs, ConjugateRhs, Version>::run(
39  Index size,
40  const Scalar* lhs, Index lhsStride,
41  const Scalar* rhs,
42  Scalar* res,
43  Scalar alpha) {
44  typedef typename packet_traits<Scalar>::type Packet;
45  typedef typename NumTraits<Scalar>::Real RealScalar;
46  const Index PacketSize = sizeof (Packet) / sizeof (Scalar);
47 
48  enum {
49  IsRowMajor = StorageOrder == RowMajor ? 1 : 0,
50  IsLower = UpLo == Lower ? 1 : 0,
51  FirstTriangular = IsRowMajor == IsLower
52  };
53 
54  conj_helper<Scalar, Scalar, NumTraits<Scalar>::IsComplex && EIGEN_LOGICAL_XOR(ConjugateLhs, IsRowMajor), ConjugateRhs> cj0;
55  conj_helper<Scalar, Scalar, NumTraits<Scalar>::IsComplex && EIGEN_LOGICAL_XOR(ConjugateLhs, !IsRowMajor), ConjugateRhs> cj1;
56  conj_helper<RealScalar, Scalar, false, ConjugateRhs> cjd;
57 
58  conj_helper<Packet, Packet, NumTraits<Scalar>::IsComplex && EIGEN_LOGICAL_XOR(ConjugateLhs, IsRowMajor), ConjugateRhs> pcj0;
59  conj_helper<Packet, Packet, NumTraits<Scalar>::IsComplex && EIGEN_LOGICAL_XOR(ConjugateLhs, !IsRowMajor), ConjugateRhs> pcj1;
60 
61  Scalar cjAlpha = ConjugateRhs ? numext::conj(alpha) : alpha;
62 
63 
64  Index bound = (std::max)(Index(0), size - 8) & 0xfffffffe;
65  if (FirstTriangular)
66  bound = size - bound;
67 
68  for (Index j = FirstTriangular ? bound : 0;
69  j < (FirstTriangular ? size : bound); j += 2) {
70  const Scalar* EIGEN_RESTRICT A0 = lhs + j*lhsStride;
71  const Scalar* EIGEN_RESTRICT A1 = lhs + (j + 1) * lhsStride;
72 
73  Scalar t0 = cjAlpha * rhs[j];
74  Packet ptmp0 = pset1<Packet>(t0);
75  Scalar t1 = cjAlpha * rhs[j + 1];
76  Packet ptmp1 = pset1<Packet>(t1);
77 
78  Scalar t2(0);
79  Packet ptmp2 = pset1<Packet>(t2);
80  Scalar t3(0);
81  Packet ptmp3 = pset1<Packet>(t3);
82 
83  Index starti = FirstTriangular ? 0 : j + 2;
84  Index endi = FirstTriangular ? j : size;
85  Index alignedStart = (starti) + internal::first_default_aligned(&res[starti], endi - starti);
86  Index alignedEnd = alignedStart + ((endi - alignedStart) / (PacketSize))*(PacketSize);
87 
88  res[j] += cjd.pmul(numext::real(A0[j]), t0);
89  res[j + 1] += cjd.pmul(numext::real(A1[j + 1]), t1);
90  if (FirstTriangular) {
91  res[j] += cj0.pmul(A1[j], t1);
92  t3 += cj1.pmul(A1[j], rhs[j]);
93  } else {
94  res[j + 1] += cj0.pmul(A0[j + 1], t0);
95  t2 += cj1.pmul(A0[j + 1], rhs[j + 1]);
96  }
97 
98  for (Index i = starti; i < alignedStart; ++i) {
99  res[i] += cj0.pmul(A0[i], t0) + cj0.pmul(A1[i], t1);
100  t2 += cj1.pmul(A0[i], rhs[i]);
101  t3 += cj1.pmul(A1[i], rhs[i]);
102  }
103  // Yes this an optimization for gcc 4.3 and 4.4 (=> huge speed up)
104  // gcc 4.2 does this optimization automatically.
105  const Scalar* EIGEN_RESTRICT a0It = A0 + alignedStart;
106  const Scalar* EIGEN_RESTRICT a1It = A1 + alignedStart;
107  const Scalar* EIGEN_RESTRICT rhsIt = rhs + alignedStart;
108  Scalar* EIGEN_RESTRICT resIt = res + alignedStart;
109  for (Index i = alignedStart; i < alignedEnd; i += PacketSize) {
110  Packet A0i = ploadu<Packet>(a0It);
111  a0It += PacketSize;
112  Packet A1i = ploadu<Packet>(a1It);
113  a1It += PacketSize;
114  Packet Bi = ploadu<Packet>(rhsIt);
115  rhsIt += PacketSize; // FIXME should be aligned in most cases
116  Packet Xi = pload <Packet>(resIt);
117 
118  Xi = pcj0.pmadd(A0i, ptmp0, pcj0.pmadd(A1i, ptmp1, Xi));
119  ptmp2 = pcj1.pmadd(A0i, Bi, ptmp2);
120  ptmp3 = pcj1.pmadd(A1i, Bi, ptmp3);
121  pstore(resIt, Xi);
122  resIt += PacketSize;
123  }
124  for (Index i = alignedEnd; i < endi; i++) {
125  res[i] += cj0.pmul(A0[i], t0) + cj0.pmul(A1[i], t1);
126  t2 += cj1.pmul(A0[i], rhs[i]);
127  t3 += cj1.pmul(A1[i], rhs[i]);
128  }
129 
130  res[j] += alpha * (t2 + predux(ptmp2));
131  res[j + 1] += alpha * (t3 + predux(ptmp3));
132  }
133  for (Index j = FirstTriangular ? 0 : bound; j < (FirstTriangular ? bound : size); j++) {
134  const Scalar* EIGEN_RESTRICT A0 = lhs + j*lhsStride;
135 
136  Scalar t1 = cjAlpha * rhs[j];
137  Scalar t2(0);
138  res[j] += cjd.pmul(numext::real(A0[j]), t1);
139  for (Index i = FirstTriangular ? 0 : j + 1; i < (FirstTriangular ? j : size); i++) {
140  res[i] += cj0.pmul(A0[i], t1);
141  t2 += cj1.pmul(A0[i], rhs[i]);
142  }
143  res[j] += alpha * t2;
144  }
145  }

155  template<typename Lhs, int LhsMode, typename Rhs>
156  struct selfadjoint_product_impl<Lhs, LhsMode, false, Rhs, 0, true> {
157  typedef typename Product<Lhs, Rhs>::Scalar Scalar;
158 
159  typedef internal::blas_traits<Lhs> LhsBlasTraits;
160  typedef typename LhsBlasTraits::DirectLinearAccessType ActualLhsType;
161  typedef typename internal::remove_all<ActualLhsType>::type ActualLhsTypeCleaned;
162 
163  typedef internal::blas_traits<Rhs> RhsBlasTraits;
164  typedef typename RhsBlasTraits::DirectLinearAccessType ActualRhsType;
165  typedef typename internal::remove_all<ActualRhsType>::type ActualRhsTypeCleaned;
166 
167  enum {
168  LhsUpLo = LhsMode & (Upper | Lower)
169  };
170 
171  template<typename Dest>
172  static void run(Dest& dest, const Lhs &a_lhs, const Rhs &a_rhs, const Scalar& alpha) {
173  typedef typename Dest::Scalar ResScalar;
174  typedef typename Rhs::Scalar RhsScalar;
175  typedef Map < Matrix<ResScalar, Dynamic, 1>, EIGEN_PLAIN_ENUM_MIN(AlignedMax, internal::packet_traits<ResScalar>::size) > MappedDest;
176 
177  eigen_assert(dest.rows() == a_lhs.rows() && dest.cols() == a_rhs.cols());
178 
179  typename internal::add_const_on_value_type<ActualLhsType>::type lhs = LhsBlasTraits::extract(a_lhs);
180  typename internal::add_const_on_value_type<ActualRhsType>::type rhs = RhsBlasTraits::extract(a_rhs);
181 
182  Scalar actualAlpha = alpha * LhsBlasTraits::extractScalarFactor(a_lhs)
183  * RhsBlasTraits::extractScalarFactor(a_rhs);
184 
185  enum {
186  EvalToDest = (Dest::InnerStrideAtCompileTime == 1),
187  UseRhs = (ActualRhsTypeCleaned::InnerStrideAtCompileTime == 1)
188  };
189 
190  internal::gemv_static_vector_if<ResScalar, Dest::SizeAtCompileTime, Dest::MaxSizeAtCompileTime, !EvalToDest> static_dest;
191  internal::gemv_static_vector_if<RhsScalar, ActualRhsTypeCleaned::SizeAtCompileTime, ActualRhsTypeCleaned::MaxSizeAtCompileTime, !UseRhs> static_rhs;
192 
193  ei_declare_aligned_stack_constructed_variable(ResScalar, actualDestPtr, dest.size(),
194  EvalToDest ? dest.data() : static_dest.data());
195 
196  ei_declare_aligned_stack_constructed_variable(RhsScalar, actualRhsPtr, rhs.size(),
197  UseRhs ? const_cast<RhsScalar*> (rhs.data()) : static_rhs.data());
198 
199  if (!EvalToDest) {
200 #ifdef EIGEN_DENSE_STORAGE_CTOR_PLUGIN
201  Index size = dest.size();
202  EIGEN_DENSE_STORAGE_CTOR_PLUGIN
203 #endif
204  MappedDest(actualDestPtr, dest.size()) = dest;
205  }
206 
207  if (!UseRhs) {
208 #ifdef EIGEN_DENSE_STORAGE_CTOR_PLUGIN
209  Index size = rhs.size();
210  EIGEN_DENSE_STORAGE_CTOR_PLUGIN
211 #endif
212  Map<typename ActualRhsTypeCleaned::PlainObject > (actualRhsPtr, rhs.size()) = rhs;
213  }
214 
215 
216  internal::selfadjoint_matrix_vector_product<Scalar, Index, (internal::traits<ActualLhsTypeCleaned>::Flags & RowMajorBit) ? RowMajor : ColMajor,
217  int(LhsUpLo), bool(LhsBlasTraits::NeedToConjugate), bool(RhsBlasTraits::NeedToConjugate)>::run
218  (
219  lhs.rows(), // size
220  &lhs.coeffRef(0, 0), lhs.outerStride(), // lhs info
221  actualRhsPtr, // rhs info
222  actualDestPtr, // result info
223  actualAlpha // scale factor
224  );
225 
226  if (!EvalToDest)
227  dest = MappedDest(actualDestPtr, dest.size());
228  }
229  };

231  template<typename Lhs, typename Rhs, int RhsMode>
232  struct selfadjoint_product_impl<Lhs, 0, true, Rhs, RhsMode, false> {
233  typedef typename Product<Lhs, Rhs>::Scalar Scalar;
234 
235  enum {
236  RhsUpLo = RhsMode & (Upper | Lower) };

238  template<typename Dest>
239  static void run(Dest& dest, const Lhs &a_lhs, const Rhs &a_rhs, const Scalar& alpha) {
240  // let's simply transpose the product
241  Transpose<Dest> destT(dest);
242  selfadjoint_product_impl < Transpose<const Rhs>, int(RhsUpLo) == Upper ? Lower : Upper, false,
243  Transpose<const Lhs>, 0, true > ::run(destT, a_rhs.transpose(), a_lhs.transpose(), alpha);
244  }
