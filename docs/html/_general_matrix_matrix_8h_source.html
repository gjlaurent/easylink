46  /* Specialization for a col-major destination matrix
47  * => Blocking algorithm following Goto's paper */
48  template<
49  typename Index,
50  typename LhsScalar, int LhsStorageOrder, bool ConjugateLhs,
51  typename RhsScalar, int RhsStorageOrder, bool ConjugateRhs>
52  struct general_matrix_matrix_product<Index, LhsScalar, LhsStorageOrder, ConjugateLhs, RhsScalar, RhsStorageOrder, ConjugateRhs, ColMajor> {
53  typedef gebp_traits<LhsScalar, RhsScalar> Traits;
54 
55  typedef typename ScalarBinaryOpTraits<LhsScalar, RhsScalar>::ReturnType ResScalar;
56 
57  static void run(Index rows, Index cols, Index depth,
58  const LhsScalar* _lhs, Index lhsStride,
59  const RhsScalar* _rhs, Index rhsStride,
60  ResScalar* _res, Index resStride,
61  ResScalar alpha,
62  level3_blocking<LhsScalar, RhsScalar>& blocking,
63  GemmParallelInfo<Index>* info = 0) {
64  typedef const_blas_data_mapper<LhsScalar, Index, LhsStorageOrder> LhsMapper;
65  typedef const_blas_data_mapper<RhsScalar, Index, RhsStorageOrder> RhsMapper;
66  typedef blas_data_mapper<typename Traits::ResScalar, Index, ColMajor> ResMapper;
67  LhsMapper lhs(_lhs, lhsStride);
68  RhsMapper rhs(_rhs, rhsStride);
69  ResMapper res(_res, resStride);
70 
71  Index kc = blocking.kc(); // cache block size along the K direction
72  Index mc = (std::min)(rows, blocking.mc()); // cache block size along the M direction
73  Index nc = (std::min)(cols, blocking.nc()); // cache block size along the N direction
74 
75  gemm_pack_lhs<LhsScalar, Index, LhsMapper, Traits::mr, Traits::LhsProgress, LhsStorageOrder> pack_lhs;
76  gemm_pack_rhs<RhsScalar, Index, RhsMapper, Traits::nr, RhsStorageOrder> pack_rhs;
77  gebp_kernel<LhsScalar, RhsScalar, Index, ResMapper, Traits::mr, Traits::nr, ConjugateLhs, ConjugateRhs> gebp;
78 
79 #ifdef EIGEN_HAS_OPENMP
80  if (info) {
81  // this is the parallel version!
82  int tid = omp_get_thread_num();
83  int threads = omp_get_num_threads();
84 
85  LhsScalar* blockA = blocking.blockA();
86  eigen_internal_assert(blockA != 0);
87 
88  std::size_t sizeB = kc*nc;
89  ei_declare_aligned_stack_constructed_variable(RhsScalar, blockB, sizeB, 0);
90 
91  // For each horizontal panel of the rhs, and corresponding vertical panel of the lhs...
92  for (Index k = 0; k < depth; k += kc) {
93  const Index actual_kc = (std::min)(k + kc, depth) - k; // => rows of B', and cols of the A'
94 
95  // In order to reduce the chance that a thread has to wait for the other,
96  // let's start by packing B'.
97  pack_rhs(blockB, rhs.getSubMapper(k, 0), actual_kc, nc);
98 
99  // Pack A_k to A' in a parallel fashion:
100  // each thread packs the sub block A_k,i to A'_i where i is the thread id.
101 
102  // However, before copying to A'_i, we have to make sure that no other thread is still using it,
103  // i.e., we test that info[tid].users equals 0.
104  // Then, we set info[tid].users to the number of threads to mark that all other threads are going to use it.
105  while (info[tid].users != 0) {
106  }
107  info[tid].users += threads;
108 
109  pack_lhs(blockA + info[tid].lhs_start*actual_kc, lhs.getSubMapper(info[tid].lhs_start, k), actual_kc, info[tid].lhs_length);
110 
111  // Notify the other threads that the part A'_i is ready to go.
112  info[tid].sync = k;
113 
114  // Computes C_i += A' * B' per A'_i
115  for (int shift = 0; shift < threads; ++shift) {
116  int i = (tid + shift) % threads;
117 
118  // At this point we have to make sure that A'_i has been updated by the thread i,
119  // we use testAndSetOrdered to mimic a volatile access.
120  // However, no need to wait for the B' part which has been updated by the current thread!
121  if (shift > 0) {
122  while (info[i].sync != k) {
123  }
124  }
125 
126  gebp(res.getSubMapper(info[i].lhs_start, 0), blockA + info[i].lhs_start*actual_kc, blockB, info[i].lhs_length, actual_kc, nc, alpha);
127  }
128 
129  // Then keep going as usual with the remaining B'
130  for (Index j = nc; j < cols; j += nc) {
131  const Index actual_nc = (std::min)(j + nc, cols) - j;
132 
133  // pack B_k,j to B'
134  pack_rhs(blockB, rhs.getSubMapper(k, j), actual_kc, actual_nc);
135 
136  // C_j += A' * B'
137  gebp(res.getSubMapper(0, j), blockA, blockB, rows, actual_kc, actual_nc, alpha);
138  }
139 
140  // Release all the sub blocks A'_i of A' for the current thread,
141  // i.e., we simply decrement the number of users by 1
142  for (Index i = 0; i < threads; ++i)
143 #pragma omp atomic
144  info[i].users -= 1;
145  }
146  } else
147 #endif // EIGEN_HAS_OPENMP
148  {
149  EIGEN_UNUSED_VARIABLE(info);
150 
151  // this is the sequential version!
152  std::size_t sizeA = kc*mc;
153  std::size_t sizeB = kc*nc;
154 
155  ei_declare_aligned_stack_constructed_variable(LhsScalar, blockA, sizeA, blocking.blockA());
156  ei_declare_aligned_stack_constructed_variable(RhsScalar, blockB, sizeB, blocking.blockB());
157 
158  const bool pack_rhs_once = mc != rows && kc == depth && nc == cols;
159 
160  // For each horizontal panel of the rhs, and corresponding panel of the lhs...
161  for (Index i2 = 0; i2 < rows; i2 += mc) {
162  const Index actual_mc = (std::min)(i2 + mc, rows) - i2;
163 
164  for (Index k2 = 0; k2 < depth; k2 += kc) {
165  const Index actual_kc = (std::min)(k2 + kc, depth) - k2;
166 
167  // OK, here we have selected one horizontal panel of rhs and one vertical panel of lhs.
168  // => Pack lhs's panel into a sequential chunk of memory (L2/L3 caching)
169  // Note that this panel will be read as many times as the number of blocks in the rhs's
170  // horizontal panel which is, in practice, a very low number.
171  pack_lhs(blockA, lhs.getSubMapper(i2, k2), actual_kc, actual_mc);
172 
173  // For each kc x nc block of the rhs's horizontal panel...
174  for (Index j2 = 0; j2 < cols; j2 += nc) {
175  const Index actual_nc = (std::min)(j2 + nc, cols) - j2;
176 
177  // We pack the rhs's block into a sequential chunk of memory (L2 caching)
178  // Note that this block will be read a very high number of times, which is equal to the number of
179  // micro horizontal panel of the large rhs's panel (e.g., rows/12 times).
180  if ((!pack_rhs_once) || i2 == 0)
181  pack_rhs(blockB, rhs.getSubMapper(k2, j2), actual_kc, actual_nc);
182 
183  // Everything is packed, we can now call the panel * block kernel:
184  gebp(res.getSubMapper(i2, j2), blockA, blockB, actual_mc, actual_kc, actual_nc, alpha);
185  }
186  }
187  }
188  }
189  }
