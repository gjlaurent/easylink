159  template<typename Derived>

161  MatrixBase<Derived>::stableNormalized() const {
162  typedef typename internal::nested_eval<Derived, 3>::type _Nested;
163  _Nested n(derived());
164  RealScalar w = n.cwiseAbs().maxCoeff();
165  RealScalar z = (n / w).squaredNorm();

167  return n / (numext::sqrt(z) * w);

268  template<typename OtherDerived>
269  bool MatrixBase<Derived>::isOrthogonal
270  (const MatrixBase<OtherDerived>& other, const RealScalar& prec) const {
271  typename internal::nested_eval<Derived, 2>::type nested(derived());
272  typename internal::nested_eval<OtherDerived, 2>::type otherNested(other.derived());
273  return numext::abs2(nested.dot(otherNested)) <= prec * prec * nested.squaredNorm() * otherNested.squaredNorm();
274  }

287  template<typename Derived>
288  bool MatrixBase<Derived>::isUnitary(const RealScalar& prec) const {
289  typename internal::nested_eval<Derived, 1>::type self(derived());
290  for (Index i = 0; i < cols(); ++i) {
291  if (!internal::isApprox(self.col(i).squaredNorm(), static_cast<RealScalar> (1), prec))
292  return false;
293  for (Index j = 0; j < i; ++j)
294  if (!internal::isMuchSmallerThan(self.col(i).dot(self.col(j)), static_cast<Scalar> (1), prec))
295  return false;
296  }
297  return true;
298  }
