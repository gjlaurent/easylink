33  template<typename Index, int Mode, typename LhsScalar, bool ConjLhs, typename RhsScalar, bool ConjRhs, int Version>
34  EIGEN_DONT_INLINE void triangular_matrix_vector_product<Index, Mode, LhsScalar, ConjLhs, RhsScalar, ConjRhs, ColMajor, Version>
35  ::run(Index _rows, Index _cols, const LhsScalar* _lhs, Index lhsStride,
36  const RhsScalar* _rhs, Index rhsIncr, ResScalar* _res, Index resIncr, const RhsScalar& alpha) {
37  static const Index PanelWidth = EIGEN_TUNE_TRIANGULAR_PANEL_WIDTH;
38  Index size = (std::min)(_rows, _cols);
39  Index rows = IsLower ? _rows : (std::min)(_rows, _cols);
40  Index cols = IsLower ? (std::min)(_rows, _cols) : _cols;
41 
42  typedef Map<const Matrix<LhsScalar, Dynamic, Dynamic, ColMajor>, 0, OuterStride<> > LhsMap;
43  const LhsMap lhs(_lhs, rows, cols, OuterStride<>(lhsStride));
44  typename conj_expr_if<ConjLhs, LhsMap>::type cjLhs(lhs);
45 
46  typedef Map<const Matrix<RhsScalar, Dynamic, 1>, 0, InnerStride<> > RhsMap;
47  const RhsMap rhs(_rhs, cols, InnerStride<>(rhsIncr));
48  typename conj_expr_if<ConjRhs, RhsMap>::type cjRhs(rhs);
49 
50  typedef Map<Matrix<ResScalar, Dynamic, 1> > ResMap;
51  ResMap res(_res, rows);
52 
53  typedef const_blas_data_mapper<LhsScalar, Index, ColMajor> LhsMapper;
54  typedef const_blas_data_mapper<RhsScalar, Index, RowMajor> RhsMapper;
55 
56  for (Index pi = 0; pi < size; pi += PanelWidth) {
57  Index actualPanelWidth = (std::min)(PanelWidth, size - pi);
58  for (Index k = 0; k < actualPanelWidth; ++k) {
59  Index i = pi + k;
60  Index s = IsLower ? ((HasUnitDiag || HasZeroDiag) ? i + 1 : i) : pi;
61  Index r = IsLower ? actualPanelWidth - k : k + 1;
62  if ((!(HasUnitDiag || HasZeroDiag)) || (--r) > 0)
63  res.segment(s, r) += (alpha * cjRhs.coeff(i)) * cjLhs.col(i).segment(s, r);
64  if (HasUnitDiag)
65  res.coeffRef(i) += alpha * cjRhs.coeff(i);
66  }
67  Index r = IsLower ? rows - pi - actualPanelWidth : pi;
68  if (r > 0) {
69  Index s = IsLower ? pi + actualPanelWidth : 0;
70  general_matrix_vector_product<Index, LhsScalar, LhsMapper, ColMajor, ConjLhs, RhsScalar, RhsMapper, ConjRhs, BuiltIn>::run(
71  r, actualPanelWidth,
72  LhsMapper(&lhs.coeffRef(s, pi), lhsStride),
73  RhsMapper(&rhs.coeffRef(pi), rhsIncr),
74  &res.coeffRef(s), resIncr, alpha);
75  }
76  }
77  if ((!IsLower) && cols > size) {
78  general_matrix_vector_product<Index, LhsScalar, LhsMapper, ColMajor, ConjLhs, RhsScalar, RhsMapper, ConjRhs>::run(
79  rows, cols - size,
80  LhsMapper(&lhs.coeffRef(0, size), lhsStride),
81  RhsMapper(&rhs.coeffRef(size), rhsIncr),
82  _res, resIncr, alpha);
83  }
84  }

99  template<typename Index, int Mode, typename LhsScalar, bool ConjLhs, typename RhsScalar, bool ConjRhs, int Version>
100  EIGEN_DONT_INLINE void triangular_matrix_vector_product<Index, Mode, LhsScalar, ConjLhs, RhsScalar, ConjRhs, RowMajor, Version>
101  ::run(Index _rows, Index _cols, const LhsScalar* _lhs, Index lhsStride,
102  const RhsScalar* _rhs, Index rhsIncr, ResScalar* _res, Index resIncr, const ResScalar& alpha) {
103  static const Index PanelWidth = EIGEN_TUNE_TRIANGULAR_PANEL_WIDTH;
104  Index diagSize = (std::min)(_rows, _cols);
105  Index rows = IsLower ? _rows : diagSize;
106  Index cols = IsLower ? diagSize : _cols;
107 
108  typedef Map<const Matrix<LhsScalar, Dynamic, Dynamic, RowMajor>, 0, OuterStride<> > LhsMap;
109  const LhsMap lhs(_lhs, rows, cols, OuterStride<>(lhsStride));
110  typename conj_expr_if<ConjLhs, LhsMap>::type cjLhs(lhs);
111 
112  typedef Map<const Matrix<RhsScalar, Dynamic, 1> > RhsMap;
113  const RhsMap rhs(_rhs, cols);
114  typename conj_expr_if<ConjRhs, RhsMap>::type cjRhs(rhs);
115 
116  typedef Map<Matrix<ResScalar, Dynamic, 1>, 0, InnerStride<> > ResMap;
117  ResMap res(_res, rows, InnerStride<>(resIncr));
118 
119  typedef const_blas_data_mapper<LhsScalar, Index, RowMajor> LhsMapper;
120  typedef const_blas_data_mapper<RhsScalar, Index, RowMajor> RhsMapper;
121 
122  for (Index pi = 0; pi < diagSize; pi += PanelWidth) {
123  Index actualPanelWidth = (std::min)(PanelWidth, diagSize - pi);
124  for (Index k = 0; k < actualPanelWidth; ++k) {
125  Index i = pi + k;
126  Index s = IsLower ? pi : ((HasUnitDiag || HasZeroDiag) ? i + 1 : i);
127  Index r = IsLower ? k + 1 : actualPanelWidth - k;
128  if ((!(HasUnitDiag || HasZeroDiag)) || (--r) > 0)
129  res.coeffRef(i) += alpha * (cjLhs.row(i).segment(s, r).cwiseProduct(cjRhs.segment(s, r).transpose())).sum();
130  if (HasUnitDiag)
131  res.coeffRef(i) += alpha * cjRhs.coeff(i);
132  }
133  Index r = IsLower ? pi : cols - pi - actualPanelWidth;
134  if (r > 0) {
135  Index s = IsLower ? 0 : pi + actualPanelWidth;
136  general_matrix_vector_product<Index, LhsScalar, LhsMapper, RowMajor, ConjLhs, RhsScalar, RhsMapper, ConjRhs, BuiltIn>::run(
137  actualPanelWidth, r,
138  LhsMapper(&lhs.coeffRef(pi, s), lhsStride),
139  RhsMapper(&rhs.coeffRef(s), rhsIncr),
140  &res.coeffRef(pi), resIncr, alpha);
141  }
142  }
143  if (IsLower && rows > diagSize) {
144  general_matrix_vector_product<Index, LhsScalar, LhsMapper, RowMajor, ConjLhs, RhsScalar, RhsMapper, ConjRhs>::run(
145  rows - diagSize, cols,
146  LhsMapper(&lhs.coeffRef(diagSize, 0), lhsStride),
147  RhsMapper(&rhs.coeffRef(0), rhsIncr),
148  &res.coeffRef(diagSize), resIncr, alpha);
149  }
150  }

194  template<typename Lhs, typename Rhs, typename Dest>
195  static void run(const Lhs &lhs, const Rhs &rhs, Dest& dest, const typename Dest::Scalar& alpha) {
196  typedef typename Lhs::Scalar LhsScalar;
197  typedef typename Rhs::Scalar RhsScalar;
198  typedef typename Dest::Scalar ResScalar;
199  typedef typename Dest::RealScalar RealScalar;
200 
201  typedef internal::blas_traits<Lhs> LhsBlasTraits;
202  typedef typename LhsBlasTraits::DirectLinearAccessType ActualLhsType;
203  typedef internal::blas_traits<Rhs> RhsBlasTraits;
204  typedef typename RhsBlasTraits::DirectLinearAccessType ActualRhsType;
205 
206  typedef Map < Matrix<ResScalar, Dynamic, 1>, EIGEN_PLAIN_ENUM_MIN(AlignedMax, internal::packet_traits<ResScalar>::size) > MappedDest;
207 
208  typename internal::add_const_on_value_type<ActualLhsType>::type actualLhs = LhsBlasTraits::extract(lhs);
209  typename internal::add_const_on_value_type<ActualRhsType>::type actualRhs = RhsBlasTraits::extract(rhs);
210 
211  LhsScalar lhs_alpha = LhsBlasTraits::extractScalarFactor(lhs);
212  RhsScalar rhs_alpha = RhsBlasTraits::extractScalarFactor(rhs);
213  ResScalar actualAlpha = alpha * lhs_alpha * rhs_alpha;
214 
215  enum {
216  // FIXME find a way to allow an inner stride on the result if packet_traits<Scalar>::size==1
217  // on, the other hand it is good for the cache to pack the vector anyways...
218  EvalToDestAtCompileTime = Dest::InnerStrideAtCompileTime == 1,
219  ComplexByReal = (NumTraits<LhsScalar>::IsComplex) && (!NumTraits<RhsScalar>::IsComplex),
220  MightCannotUseDest = (Dest::InnerStrideAtCompileTime != 1) || ComplexByReal
221  };
222 
223  gemv_static_vector_if<ResScalar, Dest::SizeAtCompileTime, Dest::MaxSizeAtCompileTime, MightCannotUseDest> static_dest;
224 
225  bool alphaIsCompatible = (!ComplexByReal) || (numext::imag(actualAlpha) == RealScalar(0));
226  bool evalToDest = EvalToDestAtCompileTime && alphaIsCompatible;
227 
228  RhsScalar compatibleAlpha = get_factor<ResScalar, RhsScalar>::run(actualAlpha);
229 
230  ei_declare_aligned_stack_constructed_variable(ResScalar, actualDestPtr, dest.size(),
231  evalToDest ? dest.data() : static_dest.data());
232 
233  if (!evalToDest) {
234 #ifdef EIGEN_DENSE_STORAGE_CTOR_PLUGIN
235  Index size = dest.size();
236  EIGEN_DENSE_STORAGE_CTOR_PLUGIN
237 #endif
238  if (!alphaIsCompatible) {
239  MappedDest(actualDestPtr, dest.size()).setZero();
240  compatibleAlpha = RhsScalar(1);
241  } else
242  MappedDest(actualDestPtr, dest.size()) = dest;
243  }
244 
245  internal::triangular_matrix_vector_product
246  <Index, Mode,
247  LhsScalar, LhsBlasTraits::NeedToConjugate,
248  RhsScalar, RhsBlasTraits::NeedToConjugate,
249  ColMajor>
250  ::run(actualLhs.rows(), actualLhs.cols(),
251  actualLhs.data(), actualLhs.outerStride(),
252  actualRhs.data(), actualRhs.innerStride(),
253  actualDestPtr, 1, compatibleAlpha);
254 
255  if (!evalToDest) {
256  if (!alphaIsCompatible)
257  dest += actualAlpha * MappedDest(actualDestPtr, dest.size());
258  else
259  dest = MappedDest(actualDestPtr, dest.size());
260  }
261 
262  if (((Mode & UnitDiag) == UnitDiag) && (lhs_alpha != LhsScalar(1))) {
263  Index diagSize = (std::min)(lhs.rows(), lhs.cols());
264  dest.head(diagSize) -= (lhs_alpha - LhsScalar(1)) * rhs.head(diagSize);
265  }
266  }
