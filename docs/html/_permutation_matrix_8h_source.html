173  Derived& applyTranspositionOnTheLeft(Index i, Index j) {
174  eigen_assert(i >= 0 && j >= 0 && i < size() && j < size());
175  for (Index k = 0; k < size(); ++k) {
176  if (indices().coeff(k) == i) indices().coeffRef(k) = StorageIndex(j);
177  else if (indices().coeff(k) == j) indices().coeffRef(k) = StorageIndex(i);
178  }
179  return derived();
180  }

190  Derived& applyTranspositionOnTheRight(Index i, Index j) {
191  eigen_assert(i >= 0 && j >= 0 && i < size() && j < size());
192  std::swap(indices().coeffRef(i), indices().coeffRef(j));
193  return derived();

263  Index determinant() const {
264  Index res = 1;
265  Index n = size();
266  Matrix<bool, RowsAtCompileTime, 1, 0, MaxRowsAtCompileTime > mask(n);
267  mask.fill(false);
268  Index r = 0;
269  while (r < n) {
270  // search for the next seed
271  while (r < n && mask[r]) r++;
272  if (r >= n)
273  break;
274  // we got one, let's follow it until we are back to the seed
275  Index k0 = r++;
276  mask.coeffRef(k0) = true;
277  for (Index k = indices().coeff(k0); k != k0; k = indices().coeff(k)) {
278  mask.coeffRef(k) = true;
279  res = -res;
280  }
281  }
282  return res;
283  }
