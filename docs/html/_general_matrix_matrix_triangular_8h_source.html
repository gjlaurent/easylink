56  template <typename Index, typename LhsScalar, int LhsStorageOrder, bool ConjugateLhs,
57  typename RhsScalar, int RhsStorageOrder, bool ConjugateRhs, int UpLo, int Version>
58  struct general_matrix_matrix_triangular_product<Index, LhsScalar, LhsStorageOrder, ConjugateLhs, RhsScalar, RhsStorageOrder, ConjugateRhs, ColMajor, UpLo, Version> {
59  typedef typename ScalarBinaryOpTraits<LhsScalar, RhsScalar>::ReturnType ResScalar;
60 
61  static EIGEN_STRONG_INLINE void run(Index size, Index depth, const LhsScalar* _lhs, Index lhsStride,
62  const RhsScalar* _rhs, Index rhsStride, ResScalar* _res, Index resStride,
63  const ResScalar& alpha, level3_blocking<LhsScalar, RhsScalar>& blocking) {
64  typedef gebp_traits<LhsScalar, RhsScalar> Traits;
65 
66  typedef const_blas_data_mapper<LhsScalar, Index, LhsStorageOrder> LhsMapper;
67  typedef const_blas_data_mapper<RhsScalar, Index, RhsStorageOrder> RhsMapper;
68  typedef blas_data_mapper<typename Traits::ResScalar, Index, ColMajor> ResMapper;
69  LhsMapper lhs(_lhs, lhsStride);
70  RhsMapper rhs(_rhs, rhsStride);
71  ResMapper res(_res, resStride);
72 
73  Index kc = blocking.kc();
74  Index mc = (std::min)(size, blocking.mc());
75 
76  // !!! mc must be a multiple of nr:
77  if (mc > Traits::nr)
78  mc = (mc / Traits::nr) * Traits::nr;
79 
80  std::size_t sizeA = kc*mc;
81  std::size_t sizeB = kc*size;
82 
83  ei_declare_aligned_stack_constructed_variable(LhsScalar, blockA, sizeA, blocking.blockA());
84  ei_declare_aligned_stack_constructed_variable(RhsScalar, blockB, sizeB, blocking.blockB());
85 
86  gemm_pack_lhs<LhsScalar, Index, LhsMapper, Traits::mr, Traits::LhsProgress, LhsStorageOrder> pack_lhs;
87  gemm_pack_rhs<RhsScalar, Index, RhsMapper, Traits::nr, RhsStorageOrder> pack_rhs;
88  gebp_kernel<LhsScalar, RhsScalar, Index, ResMapper, Traits::mr, Traits::nr, ConjugateLhs, ConjugateRhs> gebp;
89  tribb_kernel<LhsScalar, RhsScalar, Index, Traits::mr, Traits::nr, ConjugateLhs, ConjugateRhs, UpLo> sybb;
90 
91  for (Index k2 = 0; k2 < depth; k2 += kc) {
92  const Index actual_kc = (std::min)(k2 + kc, depth) - k2;
93 
94  // note that the actual rhs is the transpose/adjoint of mat
95  pack_rhs(blockB, rhs.getSubMapper(k2, 0), actual_kc, size);
96 
97  for (Index i2 = 0; i2 < size; i2 += mc) {
98  const Index actual_mc = (std::min)(i2 + mc, size) - i2;
99 
100  pack_lhs(blockA, lhs.getSubMapper(i2, k2), actual_kc, actual_mc);
101 
102  // the selected actual_mc * size panel of res is split into three different part:
103  // 1 - before the diagonal => processed with gebp or skipped
104  // 2 - the actual_mc x actual_mc symmetric block => processed with a special kernel
105  // 3 - after the diagonal => processed with gebp or skipped
106  if (UpLo == Lower)
107  gebp(res.getSubMapper(i2, 0), blockA, blockB, actual_mc, actual_kc,
108  (std::min)(size, i2), alpha, -1, -1, 0, 0);
109 
110 
111  sybb(_res + resStride * i2 + i2, resStride, blockA, blockB + actual_kc*i2, actual_mc, actual_kc, alpha);
112 
113  if (UpLo == Upper) {
114  Index j2 = i2 + actual_mc;
115  gebp(res.getSubMapper(i2, j2), blockA, blockB + actual_kc*j2, actual_mc,
116  actual_kc, (std::max)(Index(0), size - j2), alpha, -1, -1, 0, 0);
117  }
118  }
119  }
120  }
121  };

123  // Optimized packed Block * packed Block product kernel evaluating only one given triangular part
124  // This kernel is built on top of the gebp kernel:
125  // - the current destination block is processed per panel of actual_mc x BlockSize
126  // where BlockSize is set to the minimal value allowing gebp to be as fast as possible
127  // - then, as usual, each panel is split into three parts along the diagonal,
128  // the sub blocks above and below the diagonal are processed as usual,
129  // while the triangular block overlapping the diagonal is evaluated into a
130  // small temporary buffer which is then accumulated into the result using a
131  // triangular traversal.
132 
133  template<typename LhsScalar, typename RhsScalar, typename Index, int mr, int nr, bool ConjLhs, bool ConjRhs, int UpLo>
134  struct tribb_kernel {
135  typedef gebp_traits<LhsScalar, RhsScalar, ConjLhs, ConjRhs> Traits;
136  typedef typename Traits::ResScalar ResScalar;
137 
138  enum {
139  BlockSize = meta_least_common_multiple < EIGEN_PLAIN_ENUM_MAX(mr, nr), EIGEN_PLAIN_ENUM_MIN(mr, nr)>::ret
140  };
141 
142  void operator()(ResScalar* _res, Index resStride, const LhsScalar* blockA, const RhsScalar* blockB, Index size, Index depth, const ResScalar& alpha) {
143  typedef blas_data_mapper<ResScalar, Index, ColMajor> ResMapper;
144  ResMapper res(_res, resStride);
145  gebp_kernel<LhsScalar, RhsScalar, Index, ResMapper, mr, nr, ConjLhs, ConjRhs> gebp_kernel;
146 
147  Matrix<ResScalar, BlockSize, BlockSize, ColMajor> buffer((internal::constructor_without_unaligned_array_assert()));
148 
149  // let's process the block per panel of actual_mc x BlockSize,
150  // again, each is split into three parts, etc.
151  for (Index j = 0; j < size; j += BlockSize) {
152  Index actualBlockSize = std::min<Index>(BlockSize, size - j);
153  const RhsScalar* actual_b = blockB + j*depth;
154 
155  if (UpLo == Upper)
156  gebp_kernel(res.getSubMapper(0, j), blockA, actual_b, j, depth, actualBlockSize, alpha,
157  -1, -1, 0, 0);
158 
159  // selfadjoint micro block
160  {
161  Index i = j;
162  buffer.setZero();
163  // 1 - apply the kernel on the temporary buffer
164  gebp_kernel(ResMapper(buffer.data(), BlockSize), blockA + depth*i, actual_b, actualBlockSize, depth, actualBlockSize, alpha,
165  -1, -1, 0, 0);
166  // 2 - triangular accumulation
167  for (Index j1 = 0; j1 < actualBlockSize; ++j1) {
168  ResScalar* r = &res(i, j + j1);
169  for (Index i1 = UpLo == Lower ? j1 : 0;
170  UpLo == Lower ? i1 < actualBlockSize : i1 <= j1; ++i1)
171  r[i1] += buffer(i1, j1);
172  }
173  }
174 
175  if (UpLo == Lower) {
176  Index i = j + actualBlockSize;
177  gebp_kernel(res.getSubMapper(i, j), blockA + depth*i, actual_b, size - i,
178  depth, actualBlockSize, alpha, -1, -1, 0, 0);
179  }
180  }
181  }
182  };

241  static void run(MatrixType& mat, const ProductType& prod, const typename MatrixType::Scalar& alpha, bool beta) {
242  typedef typename internal::remove_all<typename ProductType::LhsNested>::type Lhs;
243  typedef internal::blas_traits<Lhs> LhsBlasTraits;
244  typedef typename LhsBlasTraits::DirectLinearAccessType ActualLhs;
245  typedef typename internal::remove_all<ActualLhs>::type _ActualLhs;
246  typename internal::add_const_on_value_type<ActualLhs>::type actualLhs = LhsBlasTraits::extract(prod.lhs());
247 
248  typedef typename internal::remove_all<typename ProductType::RhsNested>::type Rhs;
249  typedef internal::blas_traits<Rhs> RhsBlasTraits;
250  typedef typename RhsBlasTraits::DirectLinearAccessType ActualRhs;
251  typedef typename internal::remove_all<ActualRhs>::type _ActualRhs;
252  typename internal::add_const_on_value_type<ActualRhs>::type actualRhs = RhsBlasTraits::extract(prod.rhs());
253 
254  typename ProductType::Scalar actualAlpha = alpha * LhsBlasTraits::extractScalarFactor(prod.lhs().derived()) * RhsBlasTraits::extractScalarFactor(prod.rhs().derived());
255 
256  if (!beta)
257  mat.template triangularView<UpLo>().setZero();
258 
259  enum {
260  IsRowMajor = (internal::traits<MatrixType>::Flags&RowMajorBit) ? 1 : 0,
261  LhsIsRowMajor = _ActualLhs::Flags & RowMajorBit ? 1 : 0,
262  RhsIsRowMajor = _ActualRhs::Flags & RowMajorBit ? 1 : 0,
263  SkipDiag = (UpLo & (UnitDiag | ZeroDiag)) != 0
264  };
265 
266  Index size = mat.cols();
267  if (SkipDiag)
268  size--;
269  Index depth = actualLhs.cols();
270 
271  typedef internal::gemm_blocking_space < IsRowMajor ? RowMajor : ColMajor, typename Lhs::Scalar, typename Rhs::Scalar,
272  MatrixType::MaxColsAtCompileTime, MatrixType::MaxColsAtCompileTime, _ActualRhs::MaxColsAtCompileTime> BlockingType;
273 
274  BlockingType blocking(size, size, depth, 1, false);
275 
276  internal::general_matrix_matrix_triangular_product<Index,
277  typename Lhs::Scalar, LhsIsRowMajor ? RowMajor : ColMajor, LhsBlasTraits::NeedToConjugate,
278  typename Rhs::Scalar, RhsIsRowMajor ? RowMajor : ColMajor, RhsBlasTraits::NeedToConjugate,
279  IsRowMajor ? RowMajor : ColMajor, UpLo & (Lower | Upper)>
280  ::run(size, depth,
281  &actualLhs.coeffRef(SkipDiag && (UpLo & Lower) == Lower ? 1 : 0, 0), actualLhs.outerStride(),
282  &actualRhs.coeffRef(0, SkipDiag && (UpLo & Upper) == Upper ? 1 : 0), actualRhs.outerStride(),
283  mat.data() + (SkipDiag ? (bool(IsRowMajor) != ((UpLo & Lower) == Lower) ? 1 : mat.outerStride()) : 0), mat.outerStride(), actualAlpha, blocking);
284  }
285  };
