40  // Explanation for this CoeffReturnType typedef.
41  // - This is the return type of the coeff() method.
42  // - The LvalueBit means exactly that we can offer a coeffRef() method, which means exactly that we can get references
43  // to coeffs, which means exactly that we can have coeff() return a const reference (as opposed to returning a value).
44  // - The is_artihmetic check is required since "const int", "const double", etc. will cause warnings on some systems
45  // while the declaration of "const T", where T is a non arithmetic type does not. Always returning "const Scalar&" is
46  // not possible, since the underlying expressions might not offer a valid address the reference could be referring to.
47  typedef typename internal::conditional<bool(internal::traits<Derived>::Flags&LvalueBit),
48  const Scalar&,
49  typename internal::conditional<internal::is_arithmetic<Scalar>::value, Scalar, const Scalar>::type
50  >::type CoeffReturnType;
51 
52  typedef typename internal::add_const_on_value_type_if_arithmetic<
53  typename internal::packet_traits<Scalar>::type
54  >::type PacketReturnType;
