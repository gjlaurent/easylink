129  template<typename Derived>
130  std::ostream & print_matrix(std::ostream & s, const Derived& _m, const IOFormat& fmt) {
131  if (_m.size() == 0) {
132  s << fmt.matPrefix << fmt.matSuffix;
133  return s;
134  }
135 
136  typename Derived::Nested m = _m;
137  typedef typename Derived::Scalar Scalar;
138 
139  Index width = 0;
140 
141  std::streamsize explicit_precision;
142  if (fmt.precision == StreamPrecision) {
143  explicit_precision = 0;
144  } else if (fmt.precision == FullPrecision) {
145  if (NumTraits<Scalar>::IsInteger) {
146  explicit_precision = 0;
147  } else {
148  explicit_precision = significant_decimals_impl<Scalar>::run();
149  }
150  } else {
151  explicit_precision = fmt.precision;
152  }
153 
154  std::streamsize old_precision = 0;
155  if (explicit_precision) old_precision = s.precision(explicit_precision);
156 
157  bool align_cols = !(fmt.flags & DontAlignCols);
158  if (align_cols) {
159  // compute the largest width
160  for (Index j = 0; j < m.cols(); ++j)
161  for (Index i = 0; i < m.rows(); ++i) {
162  std::stringstream sstr;
163  sstr.copyfmt(s);
164  sstr << m.coeff(i, j);
165  width = std::max<Index>(width, Index(sstr.str().length()));
166  }
167  }
168  s << fmt.matPrefix;
169  for (Index i = 0; i < m.rows(); ++i) {
170  if (i)
171  s << fmt.rowSpacer;
172  s << fmt.rowPrefix;
173  if (width) s.width(width);
174  s << m.coeff(i, 0);
175  for (Index j = 1; j < m.cols(); ++j) {
176  s << fmt.coeffSeparator;
177  if (width) s.width(width);
178  s << m.coeff(i, j);
179  }
180  s << fmt.rowSuffix;
181  if (i < m.rows() - 1)
182  s << fmt.rowSeparator;
183  }
184  s << fmt.matSuffix;
185  if (explicit_precision) s.precision(old_precision);
186  return s;
187  }
