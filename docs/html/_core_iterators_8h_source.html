94  class inner_iterator_selector<XprType, IndexBased> {
95  protected:
96  typedef evaluator<XprType> EvaluatorType;
97  typedef typename traits<XprType>::Scalar Scalar;
98 
99  enum {
100  IsRowMajor = (XprType::Flags&RowMajorBit) == RowMajorBit
101  };
102 
103  public:
104 
105  EIGEN_STRONG_INLINE inner_iterator_selector(const EvaluatorType &eval, const Index &outerId, const Index &innerSize)
106  : m_eval(eval), m_inner(0), m_outer(outerId), m_end(innerSize) {
107  }
108 
109  EIGEN_STRONG_INLINE Scalar value() const {
110  return (IsRowMajor) ? m_eval.coeff(m_outer, m_inner)
111  : m_eval.coeff(m_inner, m_outer);
112  }
113 
114  EIGEN_STRONG_INLINE inner_iterator_selector& operator++() {
115  m_inner++;
116  return *this;
117  }
118 
119  EIGEN_STRONG_INLINE Index index() const {
120  return m_inner;
121  }
122 
123  inline Index row() const {
124  return IsRowMajor ? m_outer : index();
125  }
126 
127  inline Index col() const {
128  return IsRowMajor ? index() : m_outer;
129  }
130 
131  EIGEN_STRONG_INLINE operator bool() const {
132  return m_inner < m_end && m_inner >= 0;
133  }
134 
135  protected:
136  const EvaluatorType& m_eval;
137  Index m_inner;
138  const Index m_outer;
139  const Index m_end;
140  };
