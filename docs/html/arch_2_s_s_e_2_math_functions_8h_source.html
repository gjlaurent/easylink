23  Packet4f plog<Packet4f>(const Packet4f& _x) {
24  Packet4f x = _x;
25  _EIGEN_DECLARE_CONST_Packet4f(1, 1.0f);
26  _EIGEN_DECLARE_CONST_Packet4f(half, 0.5f);
27  _EIGEN_DECLARE_CONST_Packet4i(0x7f, 0x7f);
28 
29  _EIGEN_DECLARE_CONST_Packet4f_FROM_INT(inv_mant_mask, ~0x7f800000);
30 
31  /* the smallest non denormalized float number */
32  _EIGEN_DECLARE_CONST_Packet4f_FROM_INT(min_norm_pos, 0x00800000);
33  _EIGEN_DECLARE_CONST_Packet4f_FROM_INT(minus_inf, 0xff800000); //-1.f/0.f);
34 
35  /* natural logarithm computed for 4 simultaneous float
36  return NaN for x <= 0
37  */
38  _EIGEN_DECLARE_CONST_Packet4f(cephes_SQRTHF, 0.707106781186547524f);
39  _EIGEN_DECLARE_CONST_Packet4f(cephes_log_p0, 7.0376836292E-2f);
40  _EIGEN_DECLARE_CONST_Packet4f(cephes_log_p1, -1.1514610310E-1f);
41  _EIGEN_DECLARE_CONST_Packet4f(cephes_log_p2, 1.1676998740E-1f);
42  _EIGEN_DECLARE_CONST_Packet4f(cephes_log_p3, -1.2420140846E-1f);
43  _EIGEN_DECLARE_CONST_Packet4f(cephes_log_p4, +1.4249322787E-1f);
44  _EIGEN_DECLARE_CONST_Packet4f(cephes_log_p5, -1.6668057665E-1f);
45  _EIGEN_DECLARE_CONST_Packet4f(cephes_log_p6, +2.0000714765E-1f);
46  _EIGEN_DECLARE_CONST_Packet4f(cephes_log_p7, -2.4999993993E-1f);
47  _EIGEN_DECLARE_CONST_Packet4f(cephes_log_p8, +3.3333331174E-1f);
48  _EIGEN_DECLARE_CONST_Packet4f(cephes_log_q1, -2.12194440e-4f);
49  _EIGEN_DECLARE_CONST_Packet4f(cephes_log_q2, 0.693359375f);
50 
51 
52  Packet4i emm0;
53 
54  Packet4f invalid_mask = _mm_cmpnge_ps(x, _mm_setzero_ps()); // not greater equal is true if x is NaN
55  Packet4f iszero_mask = _mm_cmpeq_ps(x, _mm_setzero_ps());
56 
57  x = pmax(x, p4f_min_norm_pos); /* cut off denormalized stuff */
58  emm0 = _mm_srli_epi32(_mm_castps_si128(x), 23);
59 
60  /* keep only the fractional part */
61  x = _mm_and_ps(x, p4f_inv_mant_mask);
62  x = _mm_or_ps(x, p4f_half);
63 
64  emm0 = _mm_sub_epi32(emm0, p4i_0x7f);
65  Packet4f e = padd(Packet4f(_mm_cvtepi32_ps(emm0)), p4f_1);
66 
67  /* part2:
68  if( x < SQRTHF ) {
69  e -= 1;
70  x = x + x - 1.0;
71  } else { x = x - 1.0; }
72  */
73  Packet4f mask = _mm_cmplt_ps(x, p4f_cephes_SQRTHF);
74  Packet4f tmp = pand(x, mask);
75  x = psub(x, p4f_1);
76  e = psub(e, pand(p4f_1, mask));
77  x = padd(x, tmp);
78 
79  Packet4f x2 = pmul(x, x);
80  Packet4f x3 = pmul(x2, x);
81 
82  Packet4f y, y1, y2;
83  y = pmadd(p4f_cephes_log_p0, x, p4f_cephes_log_p1);
84  y1 = pmadd(p4f_cephes_log_p3, x, p4f_cephes_log_p4);
85  y2 = pmadd(p4f_cephes_log_p6, x, p4f_cephes_log_p7);
86  y = pmadd(y, x, p4f_cephes_log_p2);
87  y1 = pmadd(y1, x, p4f_cephes_log_p5);
88  y2 = pmadd(y2, x, p4f_cephes_log_p8);
89  y = pmadd(y, x3, y1);
90  y = pmadd(y, x3, y2);
91  y = pmul(y, x3);
92 
93  y1 = pmul(e, p4f_cephes_log_q1);
94  tmp = pmul(x2, p4f_half);
95  y = padd(y, y1);
96  x = psub(x, tmp);
97  y2 = pmul(e, p4f_cephes_log_q2);
98  x = padd(x, y);
99  x = padd(x, y2);
100  // negative arg will be NAN, 0 will be -INF
101  return _mm_or_ps(_mm_andnot_ps(iszero_mask, _mm_or_ps(x, invalid_mask)),
102  _mm_and_ps(iszero_mask, p4f_minus_inf));
103  }

106  Packet4f pexp<Packet4f>(const Packet4f& _x) {
107  Packet4f x = _x;
108  _EIGEN_DECLARE_CONST_Packet4f(1, 1.0f);
109  _EIGEN_DECLARE_CONST_Packet4f(half, 0.5f);
110  _EIGEN_DECLARE_CONST_Packet4i(0x7f, 0x7f);
111 
112 
113  _EIGEN_DECLARE_CONST_Packet4f(exp_hi, 88.3762626647950f);
114  _EIGEN_DECLARE_CONST_Packet4f(exp_lo, -88.3762626647949f);
115 
116  _EIGEN_DECLARE_CONST_Packet4f(cephes_LOG2EF, 1.44269504088896341f);
117  _EIGEN_DECLARE_CONST_Packet4f(cephes_exp_C1, 0.693359375f);
118  _EIGEN_DECLARE_CONST_Packet4f(cephes_exp_C2, -2.12194440e-4f);
119 
120  _EIGEN_DECLARE_CONST_Packet4f(cephes_exp_p0, 1.9875691500E-4f);
121  _EIGEN_DECLARE_CONST_Packet4f(cephes_exp_p1, 1.3981999507E-3f);
122  _EIGEN_DECLARE_CONST_Packet4f(cephes_exp_p2, 8.3334519073E-3f);
123  _EIGEN_DECLARE_CONST_Packet4f(cephes_exp_p3, 4.1665795894E-2f);
124  _EIGEN_DECLARE_CONST_Packet4f(cephes_exp_p4, 1.6666665459E-1f);
125  _EIGEN_DECLARE_CONST_Packet4f(cephes_exp_p5, 5.0000001201E-1f);
126 
127  Packet4f tmp, fx;
128  Packet4i emm0;
129 
130  // clamp x
131  x = pmax(pmin(x, p4f_exp_hi), p4f_exp_lo);
132 
133  /* express exp(x) as exp(g + n*log(2)) */
134  fx = pmadd(x, p4f_cephes_LOG2EF, p4f_half);
135 
136 #ifdef EIGEN_VECTORIZE_SSE4_1
137  fx = _mm_floor_ps(fx);
138 #else
139  emm0 = _mm_cvttps_epi32(fx);
140  tmp = _mm_cvtepi32_ps(emm0);
141  /* if greater, substract 1 */
142  Packet4f mask = _mm_cmpgt_ps(tmp, fx);
143  mask = _mm_and_ps(mask, p4f_1);
144  fx = psub(tmp, mask);
145 #endif
146 
147  tmp = pmul(fx, p4f_cephes_exp_C1);
148  Packet4f z = pmul(fx, p4f_cephes_exp_C2);
149  x = psub(x, tmp);
150  x = psub(x, z);
151 
152  z = pmul(x, x);
153 
154  Packet4f y = p4f_cephes_exp_p0;
155  y = pmadd(y, x, p4f_cephes_exp_p1);
156  y = pmadd(y, x, p4f_cephes_exp_p2);
157  y = pmadd(y, x, p4f_cephes_exp_p3);
158  y = pmadd(y, x, p4f_cephes_exp_p4);
159  y = pmadd(y, x, p4f_cephes_exp_p5);
160  y = pmadd(y, z, x);
161  y = padd(y, p4f_1);
162 
163  // build 2^n
164  emm0 = _mm_cvttps_epi32(fx);
165  emm0 = _mm_add_epi32(emm0, p4i_0x7f);
166  emm0 = _mm_slli_epi32(emm0, 23);
167  return pmax(pmul(y, Packet4f(_mm_castsi128_ps(emm0))), _x);
168  }

174  _EIGEN_DECLARE_CONST_Packet2d(1, 1.0);
175  _EIGEN_DECLARE_CONST_Packet2d(2, 2.0);
176  _EIGEN_DECLARE_CONST_Packet2d(half, 0.5);
177 
178  _EIGEN_DECLARE_CONST_Packet2d(exp_hi, 709.437);
179  _EIGEN_DECLARE_CONST_Packet2d(exp_lo, -709.436139303);
180 
181  _EIGEN_DECLARE_CONST_Packet2d(cephes_LOG2EF, 1.4426950408889634073599);
182 
183  _EIGEN_DECLARE_CONST_Packet2d(cephes_exp_p0, 1.26177193074810590878e-4);
184  _EIGEN_DECLARE_CONST_Packet2d(cephes_exp_p1, 3.02994407707441961300e-2);
185  _EIGEN_DECLARE_CONST_Packet2d(cephes_exp_p2, 9.99999999999999999910e-1);
186 
187  _EIGEN_DECLARE_CONST_Packet2d(cephes_exp_q0, 3.00198505138664455042e-6);
188  _EIGEN_DECLARE_CONST_Packet2d(cephes_exp_q1, 2.52448340349684104192e-3);
189  _EIGEN_DECLARE_CONST_Packet2d(cephes_exp_q2, 2.27265548208155028766e-1);
190  _EIGEN_DECLARE_CONST_Packet2d(cephes_exp_q3, 2.00000000000000000009e0);
191 
192  _EIGEN_DECLARE_CONST_Packet2d(cephes_exp_C1, 0.693145751953125);
193  _EIGEN_DECLARE_CONST_Packet2d(cephes_exp_C2, 1.42860682030941723212e-6);
194  static const __m128i p4i_1023_0 = _mm_setr_epi32(1023, 1023, 0, 0);
195 
196  Packet2d tmp, fx;
197  Packet4i emm0;
198 
199  // clamp x
200  x = pmax(pmin(x, p2d_exp_hi), p2d_exp_lo);
201  /* express exp(x) as exp(g + n*log(2)) */
202  fx = pmadd(p2d_cephes_LOG2EF, x, p2d_half);
203 
204 #ifdef EIGEN_VECTORIZE_SSE4_1
205  fx = _mm_floor_pd(fx);
206 #else
207  emm0 = _mm_cvttpd_epi32(fx);
208  tmp = _mm_cvtepi32_pd(emm0);
209  /* if greater, substract 1 */
210  Packet2d mask = _mm_cmpgt_pd(tmp, fx);
211  mask = _mm_and_pd(mask, p2d_1);
212  fx = psub(tmp, mask);
213 #endif
214 
215  tmp = pmul(fx, p2d_cephes_exp_C1);
216  Packet2d z = pmul(fx, p2d_cephes_exp_C2);
217  x = psub(x, tmp);
218  x = psub(x, z);
219 
220  Packet2d x2 = pmul(x, x);
221 
222  Packet2d px = p2d_cephes_exp_p0;
223  px = pmadd(px, x2, p2d_cephes_exp_p1);
224  px = pmadd(px, x2, p2d_cephes_exp_p2);
225  px = pmul(px, x);
226 
227  Packet2d qx = p2d_cephes_exp_q0;
228  qx = pmadd(qx, x2, p2d_cephes_exp_q1);
229  qx = pmadd(qx, x2, p2d_cephes_exp_q2);
230  qx = pmadd(qx, x2, p2d_cephes_exp_q3);
231 
232  x = pdiv(px, psub(qx, px));
233  x = pmadd(p2d_2, x, p2d_1);
234 
235  // build 2^n
236  emm0 = _mm_cvttpd_epi32(fx);
237  emm0 = _mm_add_epi32(emm0, p4i_1023_0);
238  emm0 = _mm_slli_epi32(emm0, 20);
239  emm0 = _mm_shuffle_epi32(emm0, _MM_SHUFFLE(1, 2, 0, 3));
240  return pmax(pmul(x, Packet2d(_mm_castsi128_pd(emm0))), _x);
241  }

243  /* evaluation of 4 sines at onces, using SSE2 intrinsics.
244 
245  The code is the exact rewriting of the cephes sinf function.
246  Precision is excellent as long as x < 8192 (I did not bother to
247  take into account the special handling they have for greater values
248  -- it does not return garbage for arguments over 8192, though, but

251  Note that it is such that sinf((float)M_PI) = 8.74e-8, which is the

255  template<> EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS EIGEN_UNUSED
256  Packet4f psin<Packet4f>(const Packet4f& _x) {
257  Packet4f x = _x;
258  _EIGEN_DECLARE_CONST_Packet4f(1, 1.0f);
259  _EIGEN_DECLARE_CONST_Packet4f(half, 0.5f);
260 
261  _EIGEN_DECLARE_CONST_Packet4i(1, 1);
262  _EIGEN_DECLARE_CONST_Packet4i(not1, ~1);
263  _EIGEN_DECLARE_CONST_Packet4i(2, 2);
264  _EIGEN_DECLARE_CONST_Packet4i(4, 4);
265 
266  _EIGEN_DECLARE_CONST_Packet4f_FROM_INT(sign_mask, 0x80000000);
267 
268  _EIGEN_DECLARE_CONST_Packet4f(minus_cephes_DP1, -0.78515625f);
269  _EIGEN_DECLARE_CONST_Packet4f(minus_cephes_DP2, -2.4187564849853515625e-4f);
270  _EIGEN_DECLARE_CONST_Packet4f(minus_cephes_DP3, -3.77489497744594108e-8f);
271  _EIGEN_DECLARE_CONST_Packet4f(sincof_p0, -1.9515295891E-4f);
272  _EIGEN_DECLARE_CONST_Packet4f(sincof_p1, 8.3321608736E-3f);
273  _EIGEN_DECLARE_CONST_Packet4f(sincof_p2, -1.6666654611E-1f);
274  _EIGEN_DECLARE_CONST_Packet4f(coscof_p0, 2.443315711809948E-005f);
275  _EIGEN_DECLARE_CONST_Packet4f(coscof_p1, -1.388731625493765E-003f);
276  _EIGEN_DECLARE_CONST_Packet4f(coscof_p2, 4.166664568298827E-002f);
277  _EIGEN_DECLARE_CONST_Packet4f(cephes_FOPI, 1.27323954473516f); // 4 / M_PI
278 
279  Packet4f xmm1, xmm2, xmm3, sign_bit, y;
280 
281  Packet4i emm0, emm2;
282  sign_bit = x;
283  /* take the absolute value */
284  x = pabs(x);
285 
286  /* take the modulo */
287 
288  /* extract the sign bit (upper one) */
289  sign_bit = _mm_and_ps(sign_bit, p4f_sign_mask);
290 
291  /* scale by 4/Pi */
292  y = pmul(x, p4f_cephes_FOPI);
293 
294  /* store the integer part of y in mm0 */
295  emm2 = _mm_cvttps_epi32(y);
296  /* j=(j+1) & (~1) (see the cephes sources) */
297  emm2 = _mm_add_epi32(emm2, p4i_1);
298  emm2 = _mm_and_si128(emm2, p4i_not1);
299  y = _mm_cvtepi32_ps(emm2);
300  /* get the swap sign flag */
301  emm0 = _mm_and_si128(emm2, p4i_4);
302  emm0 = _mm_slli_epi32(emm0, 29);
303  /* get the polynom selection mask
304  there is one polynom for 0 <= x <= Pi/4
305  and another one for Pi/4<x<=Pi/2
306 
307  Both branches will be computed.
308  */
309  emm2 = _mm_and_si128(emm2, p4i_2);
310  emm2 = _mm_cmpeq_epi32(emm2, _mm_setzero_si128());
311 
312  Packet4f swap_sign_bit = _mm_castsi128_ps(emm0);
313  Packet4f poly_mask = _mm_castsi128_ps(emm2);
314  sign_bit = _mm_xor_ps(sign_bit, swap_sign_bit);
315 
316  /* The magic pass: "Extended precision modular arithmetic"
317  x = ((x - y * DP1) - y * DP2) - y * DP3; */
318  xmm1 = pmul(y, p4f_minus_cephes_DP1);
319  xmm2 = pmul(y, p4f_minus_cephes_DP2);
320  xmm3 = pmul(y, p4f_minus_cephes_DP3);
321  x = padd(x, xmm1);
322  x = padd(x, xmm2);
323  x = padd(x, xmm3);
324 
325  /* Evaluate the first polynom (0 <= x <= Pi/4) */
326  y = p4f_coscof_p0;
327  Packet4f z = _mm_mul_ps(x, x);
328 
329  y = pmadd(y, z, p4f_coscof_p1);
330  y = pmadd(y, z, p4f_coscof_p2);
331  y = pmul(y, z);
332  y = pmul(y, z);
333  Packet4f tmp = pmul(z, p4f_half);
334  y = psub(y, tmp);
335  y = padd(y, p4f_1);
336 
337  /* Evaluate the second polynom (Pi/4 <= x <= 0) */
338 
339  Packet4f y2 = p4f_sincof_p0;
340  y2 = pmadd(y2, z, p4f_sincof_p1);
341  y2 = pmadd(y2, z, p4f_sincof_p2);
342  y2 = pmul(y2, z);
343  y2 = pmul(y2, x);
344  y2 = padd(y2, x);
345 
346  /* select the correct result from the two polynoms */
347  y2 = _mm_and_ps(poly_mask, y2);
348  y = _mm_andnot_ps(poly_mask, y);
349  y = _mm_or_ps(y, y2);
350  /* update the sign */
351  return _mm_xor_ps(y, sign_bit);
352  }

355  template<> EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS EIGEN_UNUSED
356  Packet4f pcos<Packet4f>(const Packet4f& _x) {
357  Packet4f x = _x;
358  _EIGEN_DECLARE_CONST_Packet4f(1, 1.0f);
359  _EIGEN_DECLARE_CONST_Packet4f(half, 0.5f);
360 
361  _EIGEN_DECLARE_CONST_Packet4i(1, 1);
362  _EIGEN_DECLARE_CONST_Packet4i(not1, ~1);
363  _EIGEN_DECLARE_CONST_Packet4i(2, 2);
364  _EIGEN_DECLARE_CONST_Packet4i(4, 4);
365 
366  _EIGEN_DECLARE_CONST_Packet4f(minus_cephes_DP1, -0.78515625f);
367  _EIGEN_DECLARE_CONST_Packet4f(minus_cephes_DP2, -2.4187564849853515625e-4f);
368  _EIGEN_DECLARE_CONST_Packet4f(minus_cephes_DP3, -3.77489497744594108e-8f);
369  _EIGEN_DECLARE_CONST_Packet4f(sincof_p0, -1.9515295891E-4f);
370  _EIGEN_DECLARE_CONST_Packet4f(sincof_p1, 8.3321608736E-3f);
371  _EIGEN_DECLARE_CONST_Packet4f(sincof_p2, -1.6666654611E-1f);
372  _EIGEN_DECLARE_CONST_Packet4f(coscof_p0, 2.443315711809948E-005f);
373  _EIGEN_DECLARE_CONST_Packet4f(coscof_p1, -1.388731625493765E-003f);
374  _EIGEN_DECLARE_CONST_Packet4f(coscof_p2, 4.166664568298827E-002f);
375  _EIGEN_DECLARE_CONST_Packet4f(cephes_FOPI, 1.27323954473516f); // 4 / M_PI
376 
377  Packet4f xmm1, xmm2, xmm3, y;
378  Packet4i emm0, emm2;
379 
380  x = pabs(x);
381 
382  /* scale by 4/Pi */
383  y = pmul(x, p4f_cephes_FOPI);
384 
385  /* get the integer part of y */
386  emm2 = _mm_cvttps_epi32(y);
387  /* j=(j+1) & (~1) (see the cephes sources) */
388  emm2 = _mm_add_epi32(emm2, p4i_1);
389  emm2 = _mm_and_si128(emm2, p4i_not1);
390  y = _mm_cvtepi32_ps(emm2);
391 
392  emm2 = _mm_sub_epi32(emm2, p4i_2);
393 
394  /* get the swap sign flag */
395  emm0 = _mm_andnot_si128(emm2, p4i_4);
396  emm0 = _mm_slli_epi32(emm0, 29);
397  /* get the polynom selection mask */
398  emm2 = _mm_and_si128(emm2, p4i_2);
399  emm2 = _mm_cmpeq_epi32(emm2, _mm_setzero_si128());
400 
401  Packet4f sign_bit = _mm_castsi128_ps(emm0);
402  Packet4f poly_mask = _mm_castsi128_ps(emm2);
403 
404  /* The magic pass: "Extended precision modular arithmetic"
405  x = ((x - y * DP1) - y * DP2) - y * DP3; */
406  xmm1 = pmul(y, p4f_minus_cephes_DP1);
407  xmm2 = pmul(y, p4f_minus_cephes_DP2);
408  xmm3 = pmul(y, p4f_minus_cephes_DP3);
409  x = padd(x, xmm1);
410  x = padd(x, xmm2);
411  x = padd(x, xmm3);
412 
413  /* Evaluate the first polynom (0 <= x <= Pi/4) */
414  y = p4f_coscof_p0;
415  Packet4f z = pmul(x, x);
416 
417  y = pmadd(y, z, p4f_coscof_p1);
418  y = pmadd(y, z, p4f_coscof_p2);
419  y = pmul(y, z);
420  y = pmul(y, z);
421  Packet4f tmp = _mm_mul_ps(z, p4f_half);
422  y = psub(y, tmp);
423  y = padd(y, p4f_1);
424 
425  /* Evaluate the second polynom (Pi/4 <= x <= 0) */
426  Packet4f y2 = p4f_sincof_p0;
427  y2 = pmadd(y2, z, p4f_sincof_p1);
428  y2 = pmadd(y2, z, p4f_sincof_p2);
429  y2 = pmul(y2, z);
430  y2 = pmadd(y2, x, x);
431 
432  /* select the correct result from the two polynoms */
433  y2 = _mm_and_ps(poly_mask, y2);
434  y = _mm_andnot_ps(poly_mask, y);
435  y = _mm_or_ps(y, y2);
436 
437  /* update the sign */
438  return _mm_xor_ps(y, sign_bit);
439  }

443  // Functions for sqrt.
444  // The EIGEN_FAST_MATH version uses the _mm_rsqrt_ps approximation and one step
445  // of Newton's method, at a cost of 1-2 bits of precision as opposed to the
446  // exact solution. It does not handle +inf, or denormalized numbers correctly.
447  // The main advantage of this approach is not just speed, but also the fact that
448  // it can be inlined and pipelined with other computations, further reducing its
449  // effective latency. This is similar to Quake3's fast inverse square root.
450  // For detail see here: http://www.beyond3d.com/content/articles/8/
451 
452  template<> EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS EIGEN_UNUSED
453  Packet4f psqrt<Packet4f>(const Packet4f& _x) {
454  Packet4f half = pmul(_x, pset1<Packet4f>(.5f));
455  Packet4f denormal_mask = _mm_and_ps(
456  _mm_cmpge_ps(_x, _mm_setzero_ps()),
457  _mm_cmplt_ps(_x, pset1<Packet4f>((std::numeric_limits<float>::min)())));
458 
459  // Compute approximate reciprocal sqrt.
460  Packet4f x = _mm_rsqrt_ps(_x);
461  // Do a single step of Newton's iteration.
462  x = pmul(x, psub(pset1<Packet4f>(1.5f), pmul(half, pmul(x, x))));
463  // Flush results for denormals to zero.
464  return _mm_andnot_ps(denormal_mask, pmul(_x, x));
465  }

471  return _mm_sqrt_ps(x);
472  }

483  template<> EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS EIGEN_UNUSED
484  Packet4f prsqrt<Packet4f>(const Packet4f& _x) {
485  _EIGEN_DECLARE_CONST_Packet4f_FROM_INT(inf, 0x7f800000);
486  _EIGEN_DECLARE_CONST_Packet4f_FROM_INT(nan, 0x7fc00000);
487  _EIGEN_DECLARE_CONST_Packet4f(one_point_five, 1.5f);
488  _EIGEN_DECLARE_CONST_Packet4f(minus_half, -0.5f);
489  _EIGEN_DECLARE_CONST_Packet4f_FROM_INT(flt_min, 0x00800000);
490 
491  Packet4f neg_half = pmul(_x, p4f_minus_half);
492 
493  // select only the inverse sqrt of positive normal inputs (denormals are
494  // flushed to zero and cause infs as well).
495  Packet4f le_zero_mask = _mm_cmple_ps(_x, p4f_flt_min);
496  Packet4f x = _mm_andnot_ps(le_zero_mask, _mm_rsqrt_ps(_x));
497 
498  // Fill in NaNs and Infs for the negative/zero entries.
499  Packet4f neg_mask = _mm_cmplt_ps(_x, _mm_setzero_ps());
500  Packet4f zero_mask = _mm_andnot_ps(neg_mask, le_zero_mask);
501  Packet4f infs_and_nans = _mm_or_ps(_mm_and_ps(neg_mask, p4f_nan),
502  _mm_and_ps(zero_mask, p4f_inf));
503 
504  // Do a single step of Newton's iteration.
505  x = pmul(x, pmadd(neg_half, pmul(x, x), p4f_one_point_five));
506 
507  // Insert NaNs and Infs in all the right places.
508  return _mm_or_ps(x, infs_and_nans);
509  }
