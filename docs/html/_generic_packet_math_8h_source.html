666  template<typename Packet> EIGEN_DEVICE_FUNC inline Packet
667  pinsertfirst(const Packet& a, typename unpacket_traits<Packet>::type b) {
668  // Default implementation based on pblend.
669  // It must be specialized for higher performance.
670  Selector<unpacket_traits<Packet>::size> mask;
671  mask.select[0] = true;
672  // This for loop should be optimized away by the compiler.
673  for (Index i = 1; i < unpacket_traits<Packet>::size; ++i)
674  mask.select[i] = false;
675  return pblend(mask, pset1<Packet>(b), a);
676  }
677 
679  template<typename Packet> EIGEN_DEVICE_FUNC inline Packet
680  pinsertlast(const Packet& a, typename unpacket_traits<Packet>::type b) {
681  // Default implementation based on pblend.
682  // It must be specialized for higher performance.
683  Selector<unpacket_traits<Packet>::size> mask;
684  // This for loop should be optimized away by the compiler.
685  for (Index i = 0; i < unpacket_traits<Packet>::size - 1; ++i)
686  mask.select[i] = false;
687  mask.select[unpacket_traits<Packet>::size - 1] = true;
688  return pblend(mask, pset1<Packet>(b), a);
689  }
