59  template<typename _Scalar, int _Options, typename _StorageIndex>
60  cholmod_sparse viewAsCholmod(Ref<SparseMatrix<_Scalar, _Options, _StorageIndex> > mat) {
61  cholmod_sparse res;
62  res.nzmax = mat.nonZeros();
63  res.nrow = mat.rows();
64  res.ncol = mat.cols();
65  res.p = mat.outerIndexPtr();
66  res.i = mat.innerIndexPtr();
67  res.x = mat.valuePtr();
68  res.z = 0;
69  res.sorted = 1;
70  if (mat.isCompressed()) {
71  res.packed = 1;
72  res.nz = 0;
73  } else {
74  res.packed = 0;
75  res.nz = mat.innerNonZeroPtr();
76  }
77 
78  res.dtype = 0;
79  res.stype = -1;
80 
81  if (internal::is_same<_StorageIndex, int>::value) {
82  res.itype = CHOLMOD_INT;
83  } else if (internal::is_same<_StorageIndex, long>::value) {
84  res.itype = CHOLMOD_LONG;
85  } else {
86  eigen_assert(false && "Index type not supported yet");
87  }
88 
89  // setup res.xtype
90  internal::cholmod_configure_matrix<_Scalar>::run(res);
91 
92  res.stype = 0;
93 
94  return res;
95  }

235  void analyzePattern(const MatrixType& matrix) {
236  if (m_cholmodFactor) {
237  cholmod_free_factor(&m_cholmodFactor, &m_cholmod);
238  m_cholmodFactor = 0;
239  }
240  cholmod_sparse A = viewAsCholmod(matrix.template selfadjointView<UpLo>());
241  m_cholmodFactor = cholmod_analyze(&A, &m_cholmod);
242 
243  this->m_isInitialized = true;
244  this->m_info = Success;
245  m_analysisIsOk = true;
246  m_factorizationIsOk = false;
247  }

255  void factorize(const MatrixType& matrix) {
256  eigen_assert(m_analysisIsOk && "You must first call analyzePattern()");
257  cholmod_sparse A = viewAsCholmod(matrix.template selfadjointView<UpLo>());
258  cholmod_factorize_p(&A, m_shiftOffset, 0, 0, m_cholmodFactor, &m_cholmod);
259 
260  // If the factorization failed, minor is the column at which it did. On success minor == n.
261  this->m_info = (m_cholmodFactor->minor == m_cholmodFactor->n ? Success : NumericalIssue);
262  m_factorizationIsOk = true;
263  }

274  template<typename Rhs, typename Dest>
275  void _solve_impl(const MatrixBase<Rhs> &b, MatrixBase<Dest> &dest) const {
276  eigen_assert(m_factorizationIsOk && "The decomposition is not in a valid state for solving, you must first call either compute() or symbolic()/numeric()");
277  const Index size = m_cholmodFactor->n;
278  EIGEN_UNUSED_VARIABLE(size);
279  eigen_assert(size == b.rows());
280 
281  // Cholmod needs column-major stoarge without inner-stride, which corresponds to the default behavior of Ref.
282  Ref<const Matrix<typename Rhs::Scalar, Dynamic, Dynamic, ColMajor> > b_ref(b.derived());
283 
284  cholmod_dense b_cd = viewAsCholmod(b_ref);
285  cholmod_dense* x_cd = cholmod_solve(CHOLMOD_A, m_cholmodFactor, &b_cd, &m_cholmod);
286  if (!x_cd) {
287  this->m_info = NumericalIssue;
288  return;
289  }
290  // TODO optimize this copy by swapping when possible (be careful with alignment, etc.)
291  dest = Matrix<Scalar, Dest::RowsAtCompileTime, Dest::ColsAtCompileTime>::Map(reinterpret_cast<Scalar*> (x_cd->x), b.rows(), b.cols());
292  cholmod_free_dense(&x_cd, &m_cholmod);
293  }

296  template<typename RhsDerived, typename DestDerived>
297  void _solve_impl(const SparseMatrixBase<RhsDerived> &b, SparseMatrixBase<DestDerived> &dest) const {
298  eigen_assert(m_factorizationIsOk && "The decomposition is not in a valid state for solving, you must first call either compute() or symbolic()/numeric()");
299  const Index size = m_cholmodFactor->n;
300  EIGEN_UNUSED_VARIABLE(size);
301  eigen_assert(size == b.rows());
302 
303  // note: cs stands for Cholmod Sparse
304  Ref < SparseMatrix<typename RhsDerived::Scalar, ColMajor, typename RhsDerived::StorageIndex> > b_ref(b.const_cast_derived());
305  cholmod_sparse b_cs = viewAsCholmod(b_ref);
306  cholmod_sparse* x_cs = cholmod_spsolve(CHOLMOD_A, m_cholmodFactor, &b_cs, &m_cholmod);
307  if (!x_cs) {
308  this->m_info = NumericalIssue;
309  return;
310  }
311  // TODO optimize this copy by swapping when possible (be careful with alignment, etc.)
312  dest.derived() = viewAsEigen<typename DestDerived::Scalar, ColMajor, typename DestDerived::StorageIndex > (*x_cs);
313  cholmod_free_sparse(&x_cs, &m_cholmod);
314  }

338  Scalar logDeterminant() const {
339  using std::log;
340  using numext::real;
341  eigen_assert(m_factorizationIsOk && "The decomposition is not in a valid state for solving, you must first call either compute() or symbolic()/numeric()");
342 
343  RealScalar logDet = 0;
344  Scalar *x = static_cast<Scalar*> (m_cholmodFactor->x);
345  if (m_cholmodFactor->is_super) {
346  // Supernodal factorization stored as a packed list of dense column-major blocs,
347  // as described by the following structure:
348 
349  // super[k] == index of the first column of the j-th super node
350  StorageIndex *super = static_cast<StorageIndex*> (m_cholmodFactor->super);
351  // pi[k] == offset to the description of row indices
352  StorageIndex *pi = static_cast<StorageIndex*> (m_cholmodFactor->pi);
353  // px[k] == offset to the respective dense block
354  StorageIndex *px = static_cast<StorageIndex*> (m_cholmodFactor->px);
355 
356  Index nb_super_nodes = m_cholmodFactor->nsuper;
357  for (Index k = 0; k < nb_super_nodes; ++k) {
358  StorageIndex ncols = super[k + 1] - super[k];
359  StorageIndex nrows = pi[k + 1] - pi[k];
360 
361  Map<const Array<Scalar, 1, Dynamic>, 0, InnerStride<> > sk(x + px[k], ncols, InnerStride<>(nrows + 1));
362  logDet += sk.real().log().sum();
363  }
364  } else {
365  // Simplicial factorization stored as standard CSC matrix.
366  StorageIndex *p = static_cast<StorageIndex*> (m_cholmodFactor->p);
367  Index size = m_cholmodFactor->n;
368  for (Index k = 0; k < size; ++k)
369  logDet += log(real(x[p[k]]));
370  }
371  if (m_cholmodFactor->is_ll)
372  logDet *= 2.0;
373  return logDet;
374  };
