44  static void setInputPort(int port, int nRows, int nCols, mxClassID type = mxDOUBLE_CLASS, mxComplexity complexFlag = mxREAL) {
45  const mxArray* mxarray = prhs[port];
46  if (mxGetClassID(mxarray) != type) {
47  throw std::runtime_error("Input " + toString(port) + " has a wrong type.");
48  }
49  if (mxIsSparse(mxarray)) {
50  throw std::runtime_error("Input " + toString(port) + " must not be sparse.");
51  }
52  if (mxIsComplex(mxarray)) {
53  throw std::runtime_error("Input " + toString(port) + " must not be complex.");
54  }
55  if (nRows > 0 && mxGetM(mxarray) != nRows) {
56  throw std::runtime_error("Input " + toString(port) + " must have " + toString(nRows) + " rows.");
57  }
58  if (nCols > 0 && mxGetN(mxarray) != nCols) {
59  throw std::runtime_error("Input " + toString(port) + " must have " + toString(nCols) + " cols.");
60  }
61  }
