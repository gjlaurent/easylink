283  template<typename MatrixType>
284  void HessenbergDecomposition<MatrixType>::_compute(MatrixType& matA, CoeffVectorType& hCoeffs, VectorType& temp) {
285  eigen_assert(matA.rows() == matA.cols());
286  Index n = matA.rows();
287  temp.resize(n);
288  for (Index i = 0; i < n - 1; ++i) {
289  // let's consider the vector v = i-th column starting at position i+1
290  Index remainingSize = n - i - 1;
291  RealScalar beta;
292  Scalar h;
293  matA.col(i).tail(remainingSize).makeHouseholderInPlace(h, beta);
294  matA.col(i).coeffRef(i + 1) = beta;
295  hCoeffs.coeffRef(i) = h;
296 
297  // Apply similarity transformation to remaining columns,
298  // i.e., compute A = H A H'
299 
300  // A = H A
301  matA.bottomRightCorner(remainingSize, remainingSize)
302  .applyHouseholderOnTheLeft(matA.col(i).tail(remainingSize - 1), h, &temp.coeffRef(0));
303 
304  // A = A H'
305  matA.rightCols(remainingSize)
306  .applyHouseholderOnTheRight(matA.col(i).tail(remainingSize - 1).conjugate(), numext::conj(h), &temp.coeffRef(0));
307  }
308  }
