172  EIGEN_DEVICE_FUNC inline VectorType corner(CornerType corner) const {
173  EIGEN_STATIC_ASSERT(_AmbientDim <= 3, THIS_METHOD_IS_ONLY_FOR_VECTORS_OF_A_SPECIFIC_SIZE);
174 
175  VectorType res;
176 
177  Index mult = 1;
178  for (Index d = 0; d < dim(); ++d) {
179  if (mult & corner) res[d] = m_max[d];
180  else res[d] = m_min[d];
181  mult *= 2;
182  }
183  return res;
184  }

188  EIGEN_DEVICE_FUNC inline VectorType sample() const {
189  VectorType r(dim());
190  for (Index d = 0; d < dim(); ++d) {
191  if (!ScalarTraits::IsInteger) {
192  r[d] = m_min[d] + (m_max[d] - m_min[d])
193  * internal::random<Scalar>(Scalar(0), Scalar(1));
194  } else
195  r[d] = internal::random(m_min[d], m_max[d]);
196  }
197  return r;
198  }

330  template<typename Scalar, int AmbientDim>
331  template<typename Derived>
332  EIGEN_DEVICE_FUNC inline Scalar AlignedBox<Scalar, AmbientDim>::squaredExteriorDistance(const MatrixBase<Derived>& a_p) const {
333  typename internal::nested_eval<Derived, 2 * AmbientDim>::type p(a_p.derived());
334  Scalar dist2(0);
335  Scalar aux;
336  for (Index k = 0; k < dim(); ++k) {
337  if (m_min[k] > p[k]) {
338  aux = m_min[k] - p[k];
339  dist2 += aux*aux;
340  } else if (p[k] > m_max[k]) {
341  aux = p[k] - m_max[k];
342  dist2 += aux*aux;
343  }
344  }
345  return dist2;
346  }

349  EIGEN_DEVICE_FUNC inline Scalar AlignedBox<Scalar, AmbientDim>::squaredExteriorDistance(const AlignedBox& b) const {
350  Scalar dist2(0);
351  Scalar aux;
352  for (Index k = 0; k < dim(); ++k) {
353  if (m_min[k] > b.m_max[k]) {
354  aux = m_min[k] - b.m_max[k];
355  dist2 += aux*aux;
356  } else if (b.m_min[k] > m_max[k]) {
357  aux = b.m_min[k] - m_max[k];
358  dist2 += aux*aux;
359  }
360  }
361  return dist2;
362  }
