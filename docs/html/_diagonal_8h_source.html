44  enum {
45  RowsAtCompileTime = (int(DiagIndex) == DynamicIndex || int(MatrixType::SizeAtCompileTime) == Dynamic) ? Dynamic
46  : (EIGEN_PLAIN_ENUM_MIN(MatrixType::RowsAtCompileTime - EIGEN_PLAIN_ENUM_MAX(-DiagIndex, 0),
47  MatrixType::ColsAtCompileTime - EIGEN_PLAIN_ENUM_MAX(DiagIndex, 0))),
48  ColsAtCompileTime = 1,

53  MatrixType::MaxColsAtCompileTime - EIGEN_PLAIN_ENUM_MAX(DiagIndex, 0))),
54  MaxColsAtCompileTime = 1,
55  MaskLvalueBit = is_lvalue<MatrixType>::value ? LvalueBit : 0,
56  Flags = (unsigned int) _MatrixTypeNested::Flags & (RowMajorBit | MaskLvalueBit | DirectAccessBit) & ~RowMajorBit, // FIXME DirectAccessBit should not be handled by expressions
57  MatrixTypeOuterStride = outer_stride_at_compile_time<MatrixType>::ret,
58  InnerStrideAtCompileTime = MatrixTypeOuterStride == Dynamic ? Dynamic : MatrixTypeOuterStride + 1,
59  OuterStrideAtCompileTime = 0
