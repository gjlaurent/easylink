71  static void initializeInputPortSizes()
72  {
73  /* check for proper number of inputs */
74  if (nrhs!=INPUT_PORT_NUMBER)
75  throw runtime_error(toString(INPUT_PORT_NUMBER) + " inputs required.");
76 
77  /* check for proper type and size of inputs */
78  for (int i=0;i<INPUT_PORT_NUMBER;i++)
79  {
80  if ((inputPortType[i]==2)||(inputPortType[i]==3))
81  {
82  // check if it is really a string
83  if (!mxIsChar(prhs[i]))
84  throw runtime_error("Argument " + toString(i+1) + " must be a string.");
85 
86  if (inputPortType[i]==3)
87  {
88  // check if it is really a identifier
89  string str=getInputString(i);
90  if (!isIdentifier(str))
91  throw runtime_error("Argument " + toString(i+1) + " must be a valid identifier.");
92  }
93  }
94  else
95  {
96  // check if it is a scalar or a vector
97  if (mxIsSparse(prhs[i])||!mxIsDouble(prhs[i]))
98  throw runtime_error("Argument " + toString(i+1) + " must be a scalar or an array of real values.");
99 
100  // check the number of rows
101  if ( (inputPortRows[i]>0) && (mxGetM(prhs[i])!=inputPortRows[i]) )
102  throw runtime_error("Argument " + toString(i+1) + " must have " + toString(parameterRows[i]) + " rows.");
103 
104  // check the number of cols
105  if ( (inputPortCols[i]>0) && (mxGetN(prhs[i])!=inputPortCols[i]) )
106  throw runtime_error("Argument " + toString(i+1) + " must have " + toString(parameterCols[i]) + " cols.");
107  }
108  }
109 
110  }
