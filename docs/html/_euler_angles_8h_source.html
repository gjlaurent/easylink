35  template<typename Derived>
36  EIGEN_DEVICE_FUNC inline Matrix<typename MatrixBase<Derived>::Scalar, 3, 1 >
37  MatrixBase<Derived>::eulerAngles(Index a0, Index a1, Index a2) const {
38  EIGEN_USING_STD_MATH(atan2)
39  EIGEN_USING_STD_MATH(sin)
40  EIGEN_USING_STD_MATH(cos)
41  /* Implemented from Graphics Gems IV */
42  EIGEN_STATIC_ASSERT_MATRIX_SPECIFIC_SIZE(Derived, 3, 3)
43 
44  Matrix<Scalar, 3, 1> res;
45  typedef Matrix<typename Derived::Scalar, 2, 1 > Vector2;
46 
47  const Index odd = ((a0 + 1) % 3 == a1) ? 0 : 1;
48  const Index i = a0;
49  const Index j = (a0 + 1 + odd) % 3;
50  const Index k = (a0 + 2 - odd) % 3;
51 
52  if (a0 == a2) {
53  res[0] = atan2(coeff(j, i), coeff(k, i));
54  if ((odd && res[0] < Scalar(0)) || ((!odd) && res[0] > Scalar(0))) {
55  if (res[0] > Scalar(0)) {
56  res[0] -= Scalar(EIGEN_PI);
57  } else {
58  res[0] += Scalar(EIGEN_PI);
59  }
60  Scalar s2 = Vector2(coeff(j, i), coeff(k, i)).norm();
61  res[1] = -atan2(s2, coeff(i, i));
62  } else {
63  Scalar s2 = Vector2(coeff(j, i), coeff(k, i)).norm();
64  res[1] = atan2(s2, coeff(i, i));
65  }
66 
67  // With a=(0,1,0), we have i=0; j=1; k=2, and after computing the first two angles,
68  // we can compute their respective rotation, and apply its inverse to M. Since the result must
69  // be a rotation around x, we have:
70  //
71  // c2 s1.s2 c1.s2 1 0 0
72  // 0 c1 -s1 * M = 0 c3 s3
73  // -s2 s1.c2 c1.c2 0 -s3 c3
74  //
75  // Thus: m11.c1 - m21.s1 = c3 & m12.c1 - m22.s1 = s3
76 
77  Scalar s1 = sin(res[0]);
78  Scalar c1 = cos(res[0]);
79  res[2] = atan2(c1 * coeff(j, k) - s1 * coeff(k, k), c1 * coeff(j, j) - s1 * coeff(k, j));
80  }
81  else {
82  res[0] = atan2(coeff(j, k), coeff(k, k));
83  Scalar c2 = Vector2(coeff(i, i), coeff(i, j)).norm();
84  if ((odd && res[0] < Scalar(0)) || ((!odd) && res[0] > Scalar(0))) {
85  if (res[0] > Scalar(0)) {
86  res[0] -= Scalar(EIGEN_PI);
87  } else {
88  res[0] += Scalar(EIGEN_PI);
89  }
90  res[1] = atan2(-coeff(i, k), -c2);
91  } else
92  res[1] = atan2(-coeff(i, k), c2);
93  Scalar s1 = sin(res[0]);
94  Scalar c1 = cos(res[0]);
95  res[2] = atan2(s1 * coeff(k, i) - c1 * coeff(j, i), c1 * coeff(j, j) - s1 * coeff(k, j));
96  }
97  if (!odd)
98  res = -res;
99 
100  return res;
101  }
