195  template<typename Scalar>
196  template<typename QuatDerived>
197  EIGEN_DEVICE_FUNC AngleAxis<Scalar>& AngleAxis<Scalar>::operator=(const QuaternionBase<QuatDerived>& q) {
198  EIGEN_USING_STD_MATH(atan2)
199  EIGEN_USING_STD_MATH(abs)
200  Scalar n = q.vec().norm();
201  if (n < NumTraits<Scalar>::epsilon())
202  n = q.vec().stableNorm();
203 
204  if (n != Scalar(0)) {
205  m_angle = Scalar(2) * atan2(n, abs(q.w()));
206  if (q.w() < Scalar(0))
207  n = -n;
208  m_axis = q.vec() / n;
209  } else {
210  m_angle = Scalar(0);
211  m_axis << Scalar(1), Scalar(0), Scalar(0);
212  }
213  return *this;
214  }

237  template<typename Scalar>
238  typename AngleAxis<Scalar>::Matrix3
239  EIGEN_DEVICE_FUNC AngleAxis<Scalar>::toRotationMatrix(void) const {
240  EIGEN_USING_STD_MATH(sin)
241  EIGEN_USING_STD_MATH(cos)
242  Matrix3 res;
243  Vector3 sin_axis = sin(m_angle) * m_axis;
244  Scalar c = cos(m_angle);
245  Vector3 cos1_axis = (Scalar(1) - c) * m_axis;
246 
247  Scalar tmp;
248  tmp = cos1_axis.x() * m_axis.y();
249  res.coeffRef(0, 1) = tmp - sin_axis.z();
250  res.coeffRef(1, 0) = tmp + sin_axis.z();
251 
252  tmp = cos1_axis.x() * m_axis.z();
253  res.coeffRef(0, 2) = tmp + sin_axis.y();
254  res.coeffRef(2, 0) = tmp - sin_axis.y();
255 
256  tmp = cos1_axis.y() * m_axis.z();
257  res.coeffRef(1, 2) = tmp - sin_axis.x();
258  res.coeffRef(2, 1) = tmp + sin_axis.x();
259 
260  res.diagonal() = (cos1_axis.cwiseProduct(m_axis)).array() + c;
261 
262  return res;
263  }
