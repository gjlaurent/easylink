277  template<typename MatrixType>
278  GeneralizedEigenSolver<MatrixType>&
279  GeneralizedEigenSolver<MatrixType>::compute(const MatrixType& A, const MatrixType& B, bool computeEigenvectors) {
280  check_template_parameters();
281 
282  using std::sqrt;
283  using std::abs;
284  eigen_assert(A.cols() == A.rows() && B.cols() == A.rows() && B.cols() == B.rows());
285  Index size = A.cols();
286  m_valuesOkay = false;
287  m_vectorsOkay = false;
288  // Reduce to generalized real Schur form:
289  // A = Q S Z and B = Q T Z
290  m_realQZ.compute(A, B, computeEigenvectors);
291  if (m_realQZ.info() == Success) {
292  // Resize storage
293  m_alphas.resize(size);
294  m_betas.resize(size);
295  if (computeEigenvectors) {
296  m_eivec.resize(size, size);
297  m_tmp.resize(size);
298  }
299 
300  // Aliases:
301  Map<VectorType> v(reinterpret_cast<Scalar*> (m_tmp.data()), size);
302  ComplexVectorType &cv = m_tmp;
303  const MatrixType &mS = m_realQZ.matrixS();
304  const MatrixType &mT = m_realQZ.matrixT();
305 
306  Index i = 0;
307  while (i < size) {
308  if (i == size - 1 || mS.coeff(i + 1, i) == Scalar(0)) {
309  // Real eigenvalue
310  m_alphas.coeffRef(i) = mS.diagonal().coeff(i);
311  m_betas.coeffRef(i) = mT.diagonal().coeff(i);
312  if (computeEigenvectors) {
313  v.setConstant(Scalar(0.0));
314  v.coeffRef(i) = Scalar(1.0);
315  // For singular eigenvalues do nothing more
316  if (abs(m_betas.coeffRef(i)) >= (std::numeric_limits<RealScalar>::min)()) {
317  // Non-singular eigenvalue
318  const Scalar alpha = real(m_alphas.coeffRef(i));
319  const Scalar beta = m_betas.coeffRef(i);
320  for (Index j = i - 1; j >= 0; j--) {
321  const Index st = j + 1;
322  const Index sz = i - j;
323  if (j > 0 && mS.coeff(j, j - 1) != Scalar(0)) {
324  // 2x2 block
325  Matrix<Scalar, 2, 1> rhs = (alpha * mT.template block<2, Dynamic>(j - 1, st, 2, sz) - beta * mS.template block<2, Dynamic>(j - 1, st, 2, sz)) .lazyProduct(v.segment(st, sz));
326  Matrix<Scalar, 2, 2> lhs = beta * mS.template block<2, 2>(j - 1, j - 1) - alpha * mT.template block<2, 2>(j - 1, j - 1);
327  v.template segment<2>(j - 1) = lhs.partialPivLu().solve(rhs);
328  j--;
329  } else {
330  v.coeffRef(j) = -v.segment(st, sz).transpose().cwiseProduct(beta * mS.block(j, st, 1, sz) - alpha * mT.block(j, st, 1, sz)).sum() / (beta * mS.coeffRef(j, j) - alpha * mT.coeffRef(j, j));
331  }
332  }
333  }
334  m_eivec.col(i).real().noalias() = m_realQZ.matrixZ().transpose() * v;
335  m_eivec.col(i).real().normalize();
336  m_eivec.col(i).imag().setConstant(0);
337  }
338  ++i;
339  } else {
340  // We need to extract the generalized eigenvalues of the pair of a general 2x2 block S and a positive diagonal 2x2 block T
341  // Then taking beta=T_00*T_11, we can avoid any division, and alpha is the eigenvalues of A = (U^-1 * S * U) * diag(T_11,T_00):
342 
343  // T = [a 0]
344  // [0 b]
345  RealScalar a = mT.diagonal().coeff(i),
346  b = mT.diagonal().coeff(i + 1);
347  const RealScalar beta = m_betas.coeffRef(i) = m_betas.coeffRef(i + 1) = a*b;
348 
349  // ^^ NOTE: using diagonal()(i) instead of coeff(i,i) workarounds a MSVC bug.
350  Matrix<RealScalar, 2, 2> S2 = mS.template block<2, 2>(i, i) * Matrix<Scalar, 2, 1>(b, a).asDiagonal();
351 
352  Scalar p = Scalar(0.5) * (S2.coeff(0, 0) - S2.coeff(1, 1));
353  Scalar z = sqrt(abs(p * p + S2.coeff(1, 0) * S2.coeff(0, 1)));
354  const ComplexScalar alpha = ComplexScalar(S2.coeff(1, 1) + p, (beta > 0) ? z : -z);
355  m_alphas.coeffRef(i) = conj(alpha);
356  m_alphas.coeffRef(i + 1) = alpha;
357 
358  if (computeEigenvectors) {
359  // Compute eigenvector in position (i+1) and then position (i) is just the conjugate
360  cv.setZero();
361  cv.coeffRef(i + 1) = Scalar(1.0);
362  // here, the "static_cast" workaound expression template issues.
363  cv.coeffRef(i) = -(static_cast<Scalar> (beta * mS.coeffRef(i, i + 1)) - alpha * mT.coeffRef(i, i + 1))
364  / (static_cast<Scalar> (beta * mS.coeffRef(i, i)) - alpha * mT.coeffRef(i, i));
365  for (Index j = i - 1; j >= 0; j--) {
366  const Index st = j + 1;
367  const Index sz = i + 1 - j;
368  if (j > 0 && mS.coeff(j, j - 1) != Scalar(0)) {
369  // 2x2 block
370  Matrix<ComplexScalar, 2, 1> rhs = (alpha * mT.template block<2, Dynamic>(j - 1, st, 2, sz) - beta * mS.template block<2, Dynamic>(j - 1, st, 2, sz)) .lazyProduct(cv.segment(st, sz));
371  Matrix<ComplexScalar, 2, 2> lhs = beta * mS.template block<2, 2>(j - 1, j - 1) - alpha * mT.template block<2, 2>(j - 1, j - 1);
372  cv.template segment<2>(j - 1) = lhs.partialPivLu().solve(rhs);
373  j--;
374  } else {
375  cv.coeffRef(j) = cv.segment(st, sz).transpose().cwiseProduct(beta * mS.block(j, st, 1, sz) - alpha * mT.block(j, st, 1, sz)).sum()
376  / (alpha * mT.coeffRef(j, j) - static_cast<Scalar> (beta * mS.coeffRef(j, j)));
377  }
378  }
379  m_eivec.col(i + 1).noalias() = (m_realQZ.matrixZ().transpose() * cv);
380  m_eivec.col(i + 1).normalize();
381  m_eivec.col(i) = m_eivec.col(i + 1).conjugate();
382  }
383  i += 2;
384  }
385  }
386 
387  m_valuesOkay = true;
388  m_vectorsOkay = computeEigenvectors;
389  }
390  return *this;
391  }
