50 inline Array<double> newMatlabArray(int nrows, int ncols, std::string name, std::string workspace = "base") {
51  mxArray *mxarray;
52 
53  mxarray = mxCreateDoubleMatrix(nrows, ncols, mxREAL);
54  if (!mxarray)
55  throw std::runtime_error("EasyLink error: Unable to create a double array named " + name + ".");
56 
57  if (mexPutVariable(workspace.c_str(), name.c_str(), mxarray))
58  throw std::runtime_error("EasyLink error: Unable to put variable " + name + " in workspace " + workspace + ".");
59 
60  mxDestroyArray((mxArray*) mxarray);
61 
62  return getMatlabArray(name, workspace);
63 }
