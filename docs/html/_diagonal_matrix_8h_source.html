333  template<typename Derived>
334  bool MatrixBase<Derived>::isDiagonal(const RealScalar& prec) const {
335  if (cols() != rows()) return false;
336  RealScalar maxAbsOnDiagonal = static_cast<RealScalar> (-1);
337  for (Index j = 0; j < cols(); ++j) {
338  RealScalar absOnDiagonal = numext::abs(coeff(j, j));
339  if (absOnDiagonal > maxAbsOnDiagonal) maxAbsOnDiagonal = absOnDiagonal;
340  }
341  for (Index j = 0; j < cols(); ++j)
342  for (Index i = 0; i < j; ++i) {
343  if (!internal::isMuchSmallerThan(coeff(i, j), maxAbsOnDiagonal, prec)) return false;
344  if (!internal::isMuchSmallerThan(coeff(j, i), maxAbsOnDiagonal, prec)) return false;
345  }
346  return true;
347  }
