57  static void assertParameterPort(int port, bool tunable, int nRows, int nCols, mxClassID type = mxDOUBLE_CLASS, mxComplexity complexFlag = mxREAL) {
58  if (!tunable) {
59  ssSetSFcnParamNotTunable(simStruct, port);
60  }
61 
62  const mxArray* mxarray = ssGetSFcnParam(simStruct, port);
63  if (mxIsEmpty(mxarray)) {
64  throw std::runtime_error("Parameter port " + toString(port) + " is empty.");
65  }
66  if (mxGetClassID(mxarray) != type) {
67  throw std::runtime_error("Parameter port " + toString(port) + " has a wrong type.");
68  }
69  if (mxIsSparse(mxarray)) {
70  throw std::runtime_error("Parameter port " + toString(port) + " must not be sparse.");
71  }
72  if (mxIsComplex(mxarray)) {
73  throw std::runtime_error("Parameter port " + toString(port) + " must not be complex.");
74  }
75  if (nRows > 0 && mxGetM(mxarray) != nRows) {
76  throw std::runtime_error("Parameter port " + toString(port) + " must have " + toString(nRows) + " rows.");
77  }
78  if (nCols > 0 && mxGetN(mxarray) != nCols) {
79  throw std::runtime_error("Parameter port " + toString(port) + " must have " + toString(nCols) + " cols.");
80  }
81  }
