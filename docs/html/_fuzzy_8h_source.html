18  template<typename Derived, typename OtherDerived, bool is_integer = NumTraits<typename Derived::Scalar>::IsInteger>
19  struct isApprox_selector {
20 
21  EIGEN_DEVICE_FUNC
22  static bool run(const Derived& x, const OtherDerived& y, const typename Derived::RealScalar& prec) {
23  typename internal::nested_eval<Derived, 2>::type nested(x);
24  typename internal::nested_eval<OtherDerived, 2>::type otherNested(y);
25  return (nested - otherNested).cwiseAbs2().sum() <= prec * prec * numext::mini(nested.cwiseAbs2().sum(), otherNested.cwiseAbs2().sum());
26  }
27  };
