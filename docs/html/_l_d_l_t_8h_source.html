289  template<typename MatrixType, typename TranspositionType, typename Workspace>
290  static bool unblocked(MatrixType& mat, TranspositionType& transpositions, Workspace& temp, SignMatrix& sign) {
291  using std::abs;
292  typedef typename MatrixType::Scalar Scalar;
293  typedef typename MatrixType::RealScalar RealScalar;
294  typedef typename TranspositionType::StorageIndex IndexType;
295  eigen_assert(mat.rows() == mat.cols());
296  const Index size = mat.rows();
297  bool found_zero_pivot = false;
298  bool ret = true;
299 
300  if (size <= 1) {
301  transpositions.setIdentity();
302  if (size == 0) sign = ZeroSign;
303  else if (numext::real(mat.coeff(0, 0)) > static_cast<RealScalar> (0)) sign = PositiveSemiDef;
304  else if (numext::real(mat.coeff(0, 0)) < static_cast<RealScalar> (0)) sign = NegativeSemiDef;
305  else sign = ZeroSign;
306  return true;
307  }
308 
309  for (Index k = 0; k < size; ++k) {
310  // Find largest diagonal element
311  Index index_of_biggest_in_corner;
312  mat.diagonal().tail(size - k).cwiseAbs().maxCoeff(&index_of_biggest_in_corner);
313  index_of_biggest_in_corner += k;
314 
315  transpositions.coeffRef(k) = IndexType(index_of_biggest_in_corner);
316  if (k != index_of_biggest_in_corner) {
317  // apply the transposition while taking care to consider only
318  // the lower triangular part
319  Index s = size - index_of_biggest_in_corner - 1; // trailing size after the biggest element
320  mat.row(k).head(k).swap(mat.row(index_of_biggest_in_corner).head(k));
321  mat.col(k).tail(s).swap(mat.col(index_of_biggest_in_corner).tail(s));
322  std::swap(mat.coeffRef(k, k), mat.coeffRef(index_of_biggest_in_corner, index_of_biggest_in_corner));
323  for (Index i = k + 1; i < index_of_biggest_in_corner; ++i) {
324  Scalar tmp = mat.coeffRef(i, k);
325  mat.coeffRef(i, k) = numext::conj(mat.coeffRef(index_of_biggest_in_corner, i));
326  mat.coeffRef(index_of_biggest_in_corner, i) = numext::conj(tmp);
327  }
328  if (NumTraits<Scalar>::IsComplex)
329  mat.coeffRef(index_of_biggest_in_corner, k) = numext::conj(mat.coeff(index_of_biggest_in_corner, k));
330  }
331 
332  // partition the matrix:
333  // A00 | - | -
334  // lu = A10 | A11 | -
335  // A20 | A21 | A22
336  Index rs = size - k - 1;
337  Block<MatrixType, Dynamic, 1> A21(mat, k + 1, k, rs, 1);
338  Block<MatrixType, 1, Dynamic> A10(mat, k, 0, 1, k);
339  Block<MatrixType, Dynamic, Dynamic> A20(mat, k + 1, 0, rs, k);
340 
341  if (k > 0) {
342  temp.head(k) = mat.diagonal().real().head(k).asDiagonal() * A10.adjoint();
343  mat.coeffRef(k, k) -= (A10 * temp.head(k)).value();
344  if (rs > 0)
345  A21.noalias() -= A20 * temp.head(k);
346  }
347 
348  // In some previous versions of Eigen (e.g., 3.2.1), the scaling was omitted if the pivot
349  // was smaller than the cutoff value. However, since LDLT is not rank-revealing
350  // we should only make sure that we do not introduce INF or NaN values.
351  // Remark that LAPACK also uses 0 as the cutoff value.
352  RealScalar realAkk = numext::real(mat.coeffRef(k, k));
353  bool pivot_is_valid = (abs(realAkk) > RealScalar(0));
354 
355  if (k == 0 && !pivot_is_valid) {
356  // The entire diagonal is zero, there is nothing more to do
357  // except filling the transpositions, and checking whether the matrix is zero.
358  sign = ZeroSign;
359  for (Index j = 0; j < size; ++j) {
360  transpositions.coeffRef(j) = IndexType(j);
361  ret = ret && (mat.col(j).tail(size - j - 1).array() == Scalar(0)).all();
362  }
363  return ret;
364  }
365 
366  if ((rs > 0) && pivot_is_valid)
367  A21 /= realAkk;
368  else if (rs > 0)
369  ret = ret && (A21.array() == Scalar(0)).all();
370 
371  if (found_zero_pivot && pivot_is_valid) ret = false; // factorization failed
372  else if (!pivot_is_valid) found_zero_pivot = true;
373 
374  if (sign == PositiveSemiDef) {
375  if (realAkk < static_cast<RealScalar> (0)) sign = Indefinite;
376  } else if (sign == NegativeSemiDef) {
377  if (realAkk > static_cast<RealScalar> (0)) sign = Indefinite;
378  } else if (sign == ZeroSign) {
379  if (realAkk > static_cast<RealScalar> (0)) sign = PositiveSemiDef;
380  else if (realAkk < static_cast<RealScalar> (0)) sign = NegativeSemiDef;
381  }
382  }
383 
384  return ret;
385  }

387  // Reference for the algorithm: Davis and Hager, "Multiple Rank
388  // Modifications of a Sparse Cholesky Factorization" (Algorithm 1)
389  // Trivial rearrangements of their computations (Timothy E. Holy)
390  // allow their algorithm to work for rank-1 updates even if the
391  // original matrix is not of full rank.
392  // Here only rank-1 updates are implemented, to reduce the
393  // requirement for intermediate storage and improve accuracy
394 
395  template<typename MatrixType, typename WDerived>
396  static bool updateInPlace(MatrixType& mat, MatrixBase<WDerived>& w, const typename MatrixType::RealScalar& sigma = 1) {
397  using numext::isfinite;
398  typedef typename MatrixType::Scalar Scalar;
399  typedef typename MatrixType::RealScalar RealScalar;
400 
401  const Index size = mat.rows();
402  eigen_assert(mat.cols() == size && w.size() == size);
403 
404  RealScalar alpha = 1;
405 
406  // Apply the update
407  for (Index j = 0; j < size; j++) {
408  // Check for termination due to an original decomposition of low-rank
409  if (!(isfinite) (alpha))
410  break;
411 
412  // Update the diagonal terms
413  RealScalar dj = numext::real(mat.coeff(j, j));
414  Scalar wj = w.coeff(j);
415  RealScalar swj2 = sigma * numext::abs2(wj);
416  RealScalar gamma = dj * alpha + swj2;
417 
418  mat.coeffRef(j, j) += swj2 / alpha;
419  alpha += swj2 / dj;
420 
421 
422  // Update the terms of L
423  Index rs = size - j - 1;
424  w.tail(rs) -= wj * mat.col(j).tail(rs);
425  if (gamma != 0)
426  mat.col(j).tail(rs) += (sigma * numext::conj(wj) / gamma) * w.tail(rs);
427  }
428  return true;
429  }

485  template<typename MatrixType, int _UpLo>
486  template<typename InputType>
487  LDLT<MatrixType, _UpLo>& LDLT<MatrixType, _UpLo>::compute(const EigenBase<InputType>& a) {
488  check_template_parameters();
489 
490  eigen_assert(a.rows() == a.cols());
491  const Index size = a.rows();
492 
493  m_matrix = a.derived();
494 
495  // Compute matrix L1 norm = max abs column sum.
496  m_l1_norm = RealScalar(0);
497  // TODO move this code to SelfAdjointView
498  for (Index col = 0; col < size; ++col) {
499  RealScalar abs_col_sum;
500  if (_UpLo == Lower)
501  abs_col_sum = m_matrix.col(col).tail(size - col).template lpNorm<1>() + m_matrix.row(col).head(col).template lpNorm<1>();
502  else
503  abs_col_sum = m_matrix.col(col).head(col).template lpNorm<1>() + m_matrix.row(col).tail(size - col).template lpNorm<1>();
504  if (abs_col_sum > m_l1_norm)
505  m_l1_norm = abs_col_sum;
506  }
507 
508  m_transpositions.resize(size);
509  m_isInitialized = false;
510  m_temporary.resize(size);
511  m_sign = internal::ZeroSign;
512 
513  m_info = internal::ldlt_inplace<UpLo>::unblocked(m_matrix, m_transpositions, m_temporary, m_sign) ? Success : NumericalIssue;
514 
515  m_isInitialized = true;
516  return *this;
517  }

549 template<typename _MatrixType, int _UpLo>
550  template<typename RhsType, typename DstType>
551  void LDLT<_MatrixType, _UpLo>::_solve_impl(const RhsType &rhs, DstType &dst) const {
552  eigen_assert(rhs.rows() == rows());
553  // dst = P b
554  dst = m_transpositions * rhs;
555 
556  // dst = L^-1 (P b)
557  matrixL().solveInPlace(dst);
558 
559  // dst = D^-1 (L^-1 P b)
560  // more precisely, use pseudo-inverse of D (see bug 241)
561  using std::abs;
562  const typename Diagonal<const MatrixType>::RealReturnType vecD(vectorD());
563  // In some previous versions, tolerance was set to the max of 1/highest (or rather numeric_limits::min())
564  // and the maximal diagonal entry * epsilon as motivated by LAPACK's xGELSS:
565  // RealScalar tolerance = numext::maxi(vecD.array().abs().maxCoeff() * NumTraits<RealScalar>::epsilon(),RealScalar(1) / NumTraits<RealScalar>::highest());
566  // However, LDLT is not rank revealing, and so adjusting the tolerance wrt to the highest
567  // diagonal element is not well justified and leads to numerical issues in some cases.
568  // Moreover, Lapack's xSYTRS routines use 0 for the tolerance.
569  // Using numeric_limits::min() gives us more robustness to denormals.
570  RealScalar tolerance = (std::numeric_limits<RealScalar>::min)();
571 
572  for (Index i = 0; i < vecD.size(); ++i) {
573  if (abs(vecD(i)) > tolerance)
574  dst.row(i) /= vecD(i);
575  else
576  dst.row(i).setZero();
577  }
578 
579  // dst = L^-T (D^-1 L^-1 P b)
580  matrixU().solveInPlace(dst);
581 
582  // dst = P^-1 (L^-T D^-1 L^-1 P b) = A^-1 b
583  dst = m_transpositions.transpose() * dst;
584  }
585 #endif

614  template<typename MatrixType, int _UpLo>
615  MatrixType LDLT<MatrixType, _UpLo>::reconstructedMatrix() const {
616  eigen_assert(m_isInitialized && "LDLT is not initialized.");
617  const Index size = m_matrix.rows();
618  MatrixType res(size, size);
619 
620  // P
621  res.setIdentity();
622  res = transpositionsP() * res;
623  // L^* P
624  res = matrixU() * res;
625  // D(L^*P)
626  res = vectorD().real().asDiagonal() * res;
627  // L(DL^*P)
628  res = matrixL() * res;
629  // P^T (LDL^*P)
630  res = transpositionsP().transpose() * res;
631 
632  return res;
