19  template<> EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS EIGEN_UNUSED
20  Packet4f pexp<Packet4f>(const Packet4f& _x) {
21  Packet4f x = _x;
22  Packet4f tmp, fx;
23 
24  _EIGEN_DECLARE_CONST_Packet4f(1, 1.0f);
25  _EIGEN_DECLARE_CONST_Packet4f(half, 0.5f);
26  _EIGEN_DECLARE_CONST_Packet4i(0x7f, 0x7f);
27  _EIGEN_DECLARE_CONST_Packet4f(exp_hi, 88.3762626647950f);
28  _EIGEN_DECLARE_CONST_Packet4f(exp_lo, -88.3762626647949f);
29  _EIGEN_DECLARE_CONST_Packet4f(cephes_LOG2EF, 1.44269504088896341f);
30  _EIGEN_DECLARE_CONST_Packet4f(cephes_exp_C1, 0.693359375f);
31  _EIGEN_DECLARE_CONST_Packet4f(cephes_exp_C2, -2.12194440e-4f);
32  _EIGEN_DECLARE_CONST_Packet4f(cephes_exp_p0, 1.9875691500E-4f);
33  _EIGEN_DECLARE_CONST_Packet4f(cephes_exp_p1, 1.3981999507E-3f);
34  _EIGEN_DECLARE_CONST_Packet4f(cephes_exp_p2, 8.3334519073E-3f);
35  _EIGEN_DECLARE_CONST_Packet4f(cephes_exp_p3, 4.1665795894E-2f);
36  _EIGEN_DECLARE_CONST_Packet4f(cephes_exp_p4, 1.6666665459E-1f);
37  _EIGEN_DECLARE_CONST_Packet4f(cephes_exp_p5, 5.0000001201E-1f);
38 
39  x = vminq_f32(x, p4f_exp_hi);
40  x = vmaxq_f32(x, p4f_exp_lo);
41 
42  /* express exp(x) as exp(g + n*log(2)) */
43  fx = vmlaq_f32(p4f_half, x, p4f_cephes_LOG2EF);
44 
45  /* perform a floorf */
46  tmp = vcvtq_f32_s32(vcvtq_s32_f32(fx));
47 
48  /* if greater, substract 1 */
49  Packet4ui mask = vcgtq_f32(tmp, fx);
50  mask = vandq_u32(mask, vreinterpretq_u32_f32(p4f_1));
51 
52  fx = vsubq_f32(tmp, vreinterpretq_f32_u32(mask));
53 
54  tmp = vmulq_f32(fx, p4f_cephes_exp_C1);
55  Packet4f z = vmulq_f32(fx, p4f_cephes_exp_C2);
56  x = vsubq_f32(x, tmp);
57  x = vsubq_f32(x, z);
58 
59  Packet4f y = vmulq_f32(p4f_cephes_exp_p0, x);
60  z = vmulq_f32(x, x);
61  y = vaddq_f32(y, p4f_cephes_exp_p1);
62  y = vmulq_f32(y, x);
63  y = vaddq_f32(y, p4f_cephes_exp_p2);
64  y = vmulq_f32(y, x);
65  y = vaddq_f32(y, p4f_cephes_exp_p3);
66  y = vmulq_f32(y, x);
67  y = vaddq_f32(y, p4f_cephes_exp_p4);
68  y = vmulq_f32(y, x);
69  y = vaddq_f32(y, p4f_cephes_exp_p5);
70 
71  y = vmulq_f32(y, z);
72  y = vaddq_f32(y, x);
73  y = vaddq_f32(y, p4f_1);
74 
75  /* build 2^n */
76  int32x4_t mm;
77  mm = vcvtq_s32_f32(fx);
78  mm = vaddq_s32(mm, p4i_0x7f);
79  mm = vshlq_n_s32(mm, 23);
80  Packet4f pow2n = vreinterpretq_f32_s32(mm);
81 
82  y = vmulq_f32(y, pow2n);
83  return y;
84  }
