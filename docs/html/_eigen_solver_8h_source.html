317  MatrixType EigenSolver<MatrixType>::pseudoEigenvalueMatrix() const {
318  eigen_assert(m_isInitialized && "EigenSolver is not initialized.");
319  const RealScalar precision = RealScalar(2) * NumTraits<RealScalar>::epsilon();
320  Index n = m_eivalues.rows();
321  MatrixType matD = MatrixType::Zero(n, n);
322  for (Index i = 0; i < n; ++i) {
323  if (internal::isMuchSmallerThan(numext::imag(m_eivalues.coeff(i)), numext::real(m_eivalues.coeff(i)), precision))
324  matD.coeffRef(i, i) = numext::real(m_eivalues.coeff(i));
325  else {
326  matD.template block<2, 2>(i, i) << numext::real(m_eivalues.coeff(i)), numext::imag(m_eivalues.coeff(i)),
327  -numext::imag(m_eivalues.coeff(i)), numext::real(m_eivalues.coeff(i));
328  ++i;
329  }
330  }
331  return matD;
332  }

334  template<typename MatrixType>
335  typename EigenSolver<MatrixType>::EigenvectorsType EigenSolver<MatrixType>::eigenvectors() const {
336  eigen_assert(m_isInitialized && "EigenSolver is not initialized.");
337  eigen_assert(m_eigenvectorsOk && "The eigenvectors have not been computed together with the eigenvalues.");
338  const RealScalar precision = RealScalar(2) * NumTraits<RealScalar>::epsilon();
339  Index n = m_eivec.cols();
340  EigenvectorsType matV(n, n);
341  for (Index j = 0; j < n; ++j) {
342  if (internal::isMuchSmallerThan(numext::imag(m_eivalues.coeff(j)), numext::real(m_eivalues.coeff(j)), precision) || j + 1 == n) {
343  // we have a real eigen value
344  matV.col(j) = m_eivec.col(j).template cast<ComplexScalar>();
345  matV.col(j).normalize();
346  } else {
347  // we have a pair of complex eigen values
348  for (Index i = 0; i < n; ++i) {
349  matV.coeffRef(i, j) = ComplexScalar(m_eivec.coeff(i, j), m_eivec.coeff(i, j + 1));
350  matV.coeffRef(i, j + 1) = ComplexScalar(m_eivec.coeff(i, j), -m_eivec.coeff(i, j + 1));
351  }
352  matV.col(j).normalize();
353  matV.col(j + 1).normalize();
354  ++j;
355  }
356  }
357  return matV;
358  }

362  EigenSolver<MatrixType>&
363  EigenSolver<MatrixType>::compute(const EigenBase<InputType>& matrix, bool computeEigenvectors) {
364  check_template_parameters();
365 
366  using std::sqrt;
367  using std::abs;
368  using numext::isfinite;
369  eigen_assert(matrix.cols() == matrix.rows());
370 
371  // Reduce to real Schur form.
372  m_realSchur.compute(matrix.derived(), computeEigenvectors);
373 
374  m_info = m_realSchur.info();
375 
376  if (m_info == Success) {
377  m_matT = m_realSchur.matrixT();
378  if (computeEigenvectors)
379  m_eivec = m_realSchur.matrixU();
380 
381  // Compute eigenvalues from matT
382  m_eivalues.resize(matrix.cols());
383  Index i = 0;
384  while (i < matrix.cols()) {
385  if (i == matrix.cols() - 1 || m_matT.coeff(i + 1, i) == Scalar(0)) {
386  m_eivalues.coeffRef(i) = m_matT.coeff(i, i);
387  if (!(isfinite) (m_eivalues.coeffRef(i))) {
388  m_isInitialized = true;
389  m_eigenvectorsOk = false;
390  m_info = NumericalIssue;
391  return *this;
392  }
393  ++i;
394  } else {
395  Scalar p = Scalar(0.5) * (m_matT.coeff(i, i) - m_matT.coeff(i + 1, i + 1));
396  Scalar z;
397  // Compute z = sqrt(abs(p * p + m_matT.coeff(i+1, i) * m_matT.coeff(i, i+1)));
398  // without overflow
399  {
400  Scalar t0 = m_matT.coeff(i + 1, i);
401  Scalar t1 = m_matT.coeff(i, i + 1);
402  Scalar maxval = numext::maxi<Scalar>(abs(p), numext::maxi<Scalar>(abs(t0), abs(t1)));
403  t0 /= maxval;
404  t1 /= maxval;
405  Scalar p0 = p / maxval;
406  z = maxval * sqrt(abs(p0 * p0 + t0 * t1));
407  }
408 
409  m_eivalues.coeffRef(i) = ComplexScalar(m_matT.coeff(i + 1, i + 1) + p, z);
410  m_eivalues.coeffRef(i + 1) = ComplexScalar(m_matT.coeff(i + 1, i + 1) + p, -z);
411  if (!((isfinite) (m_eivalues.coeffRef(i)) && (isfinite) (m_eivalues.coeffRef(i + 1)))) {
412  m_isInitialized = true;
413  m_eigenvectorsOk = false;
414  m_info = NumericalIssue;
415  return *this;
416  }
417  i += 2;
418  }
419  }
420 
421  // Compute eigenvectors.
422  if (computeEigenvectors)
423  doComputeEigenvectors();
424  }
425 
426  m_isInitialized = true;
427  m_eigenvectorsOk = computeEigenvectors;
428 
429  return *this;
430  }

432  template<typename MatrixType>
433  void EigenSolver<MatrixType>::doComputeEigenvectors() {
434  using std::abs;
435  const Index size = m_eivec.cols();
436  const Scalar eps = NumTraits<Scalar>::epsilon();
437 
438  // inefficient! this is already computed in RealSchur
439  Scalar norm(0);
440  for (Index j = 0; j < size; ++j) {
441  norm += m_matT.row(j).segment((std::max)(j - 1, Index(0)), size - (std::max)(j - 1, Index(0))).cwiseAbs().sum();
442  }
443 
444  // Backsubstitute to find vectors of upper triangular form
445  if (norm == Scalar(0)) {
446  return;
447  }
448 
449  for (Index n = size - 1; n >= 0; n--) {
450  Scalar p = m_eivalues.coeff(n).real();
451  Scalar q = m_eivalues.coeff(n).imag();
452 
453  // Scalar vector
454  if (q == Scalar(0)) {
455  Scalar lastr(0), lastw(0);
456  Index l = n;
457 
458  m_matT.coeffRef(n, n) = Scalar(1);
459  for (Index i = n - 1; i >= 0; i--) {
460  Scalar w = m_matT.coeff(i, i) - p;
461  Scalar r = m_matT.row(i).segment(l, n - l + 1).dot(m_matT.col(n).segment(l, n - l + 1));
462 
463  if (m_eivalues.coeff(i).imag() < Scalar(0)) {
464  lastw = w;
465  lastr = r;
466  } else {
467  l = i;
468  if (m_eivalues.coeff(i).imag() == Scalar(0)) {
469  if (w != Scalar(0))
470  m_matT.coeffRef(i, n) = -r / w;
471  else
472  m_matT.coeffRef(i, n) = -r / (eps * norm);
473  } else // Solve real equations
474  {
475  Scalar x = m_matT.coeff(i, i + 1);
476  Scalar y = m_matT.coeff(i + 1, i);
477  Scalar denom = (m_eivalues.coeff(i).real() - p) * (m_eivalues.coeff(i).real() - p) + m_eivalues.coeff(i).imag() * m_eivalues.coeff(i).imag();
478  Scalar t = (x * lastr - lastw * r) / denom;
479  m_matT.coeffRef(i, n) = t;
480  if (abs(x) > abs(lastw))
481  m_matT.coeffRef(i + 1, n) = (-r - w * t) / x;
482  else
483  m_matT.coeffRef(i + 1, n) = (-lastr - y * t) / lastw;
484  }
485 
486  // Overflow control
487  Scalar t = abs(m_matT.coeff(i, n));
488  if ((eps * t) * t > Scalar(1))
489  m_matT.col(n).tail(size - i) /= t;
490  }
491  }
492  } else if (q < Scalar(0) && n > 0) // Complex vector
493  {
494  Scalar lastra(0), lastsa(0), lastw(0);
495  Index l = n - 1;
496 
497  // Last vector component imaginary so matrix is triangular
498  if (abs(m_matT.coeff(n, n - 1)) > abs(m_matT.coeff(n - 1, n))) {
499  m_matT.coeffRef(n - 1, n - 1) = q / m_matT.coeff(n, n - 1);
500  m_matT.coeffRef(n - 1, n) = -(m_matT.coeff(n, n) - p) / m_matT.coeff(n, n - 1);
501  } else {
502  ComplexScalar cc = ComplexScalar(Scalar(0), -m_matT.coeff(n - 1, n)) / ComplexScalar(m_matT.coeff(n - 1, n - 1) - p, q);
503  m_matT.coeffRef(n - 1, n - 1) = numext::real(cc);
504  m_matT.coeffRef(n - 1, n) = numext::imag(cc);
505  }
506  m_matT.coeffRef(n, n - 1) = Scalar(0);
507  m_matT.coeffRef(n, n) = Scalar(1);
508  for (Index i = n - 2; i >= 0; i--) {
509  Scalar ra = m_matT.row(i).segment(l, n - l + 1).dot(m_matT.col(n - 1).segment(l, n - l + 1));
510  Scalar sa = m_matT.row(i).segment(l, n - l + 1).dot(m_matT.col(n).segment(l, n - l + 1));
511  Scalar w = m_matT.coeff(i, i) - p;
512 
513  if (m_eivalues.coeff(i).imag() < Scalar(0)) {
514  lastw = w;
515  lastra = ra;
516  lastsa = sa;
517  } else {
518  l = i;
519  if (m_eivalues.coeff(i).imag() == RealScalar(0)) {
520  ComplexScalar cc = ComplexScalar(-ra, -sa) / ComplexScalar(w, q);
521  m_matT.coeffRef(i, n - 1) = numext::real(cc);
522  m_matT.coeffRef(i, n) = numext::imag(cc);
523  } else {
524  // Solve complex equations
525  Scalar x = m_matT.coeff(i, i + 1);
526  Scalar y = m_matT.coeff(i + 1, i);
527  Scalar vr = (m_eivalues.coeff(i).real() - p) * (m_eivalues.coeff(i).real() - p) + m_eivalues.coeff(i).imag() * m_eivalues.coeff(i).imag() - q * q;
528  Scalar vi = (m_eivalues.coeff(i).real() - p) * Scalar(2) * q;
529  if ((vr == Scalar(0)) && (vi == Scalar(0)))
530  vr = eps * norm * (abs(w) + abs(q) + abs(x) + abs(y) + abs(lastw));
531 
532  ComplexScalar cc = ComplexScalar(x * lastra - lastw * ra + q*sa, x * lastsa - lastw * sa - q * ra) / ComplexScalar(vr, vi);
533  m_matT.coeffRef(i, n - 1) = numext::real(cc);
534  m_matT.coeffRef(i, n) = numext::imag(cc);
535  if (abs(x) > (abs(lastw) + abs(q))) {
536  m_matT.coeffRef(i + 1, n - 1) = (-ra - w * m_matT.coeff(i, n - 1) + q * m_matT.coeff(i, n)) / x;
537  m_matT.coeffRef(i + 1, n) = (-sa - w * m_matT.coeff(i, n) - q * m_matT.coeff(i, n - 1)) / x;
538  } else {
539  cc = ComplexScalar(-lastra - y * m_matT.coeff(i, n - 1), -lastsa - y * m_matT.coeff(i, n)) / ComplexScalar(lastw, q);
540  m_matT.coeffRef(i + 1, n - 1) = numext::real(cc);
541  m_matT.coeffRef(i + 1, n) = numext::imag(cc);
542  }
543  }
544 
545  // Overflow control
546  Scalar t = numext::maxi<Scalar>(abs(m_matT.coeff(i, n - 1)), abs(m_matT.coeff(i, n)));
547  if ((eps * t) * t > Scalar(1))
548  m_matT.block(i, n - 1, size - i, 2) /= t;
549 
550  }
551  }
552 
553  // We handled a pair of complex conjugate eigenvalues, so need to skip them both
554  n--;
555  } else {
556  eigen_assert(0 && "Internal bug in EigenSolver (INF or NaN has not been detected)"); // this should not happen
557  }
558  }
559 
560  // Back transformation to get eigenvectors of original matrix
561  for (Index j = size - 1; j >= 0; j--) {
562  m_tmp.noalias() = m_eivec.leftCols(j + 1) * m_matT.col(j).segment(0, j + 1);
563  m_eivec.col(j) = m_tmp;
564  }
565  }
