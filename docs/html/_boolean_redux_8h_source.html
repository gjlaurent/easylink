86  template<typename Derived>
87  inline bool DenseBase<Derived>::all() const {
88  typedef internal::evaluator<Derived> Evaluator;
89 
90  enum {
91  unroll = SizeAtCompileTime != Dynamic
92  && SizeAtCompileTime * (Evaluator::CoeffReadCost + NumTraits<Scalar>::AddCost) <= EIGEN_UNROLLING_LIMIT
93  };
94  Evaluator evaluator(derived());
95  if (unroll)
96  return internal::all_unroller < Evaluator, unroll ? int(SizeAtCompileTime) : Dynamic>::run(evaluator);
97  else {
98  for (Index j = 0; j < cols(); ++j)
99  for (Index i = 0; i < rows(); ++i)
100  if (!evaluator.coeff(i, j)) return false;
101  return true;
102  }
103  }
