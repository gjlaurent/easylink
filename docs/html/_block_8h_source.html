26  enum {
27  MatrixRows = traits<XprType>::RowsAtCompileTime,
28  MatrixCols = traits<XprType>::ColsAtCompileTime,
29  RowsAtCompileTime = MatrixRows == 0 ? 0 : BlockRows,
30  ColsAtCompileTime = MatrixCols == 0 ? 0 : BlockCols,
31  MaxRowsAtCompileTime = BlockRows == 0 ? 0
32  : RowsAtCompileTime != Dynamic ? int(RowsAtCompileTime)
33  : int(traits<XprType>::MaxRowsAtCompileTime),
34  MaxColsAtCompileTime = BlockCols == 0 ? 0
35  : ColsAtCompileTime != Dynamic ? int(ColsAtCompileTime)
36  : int(traits<XprType>::MaxColsAtCompileTime),
37 
38  XprTypeIsRowMajor = (int(traits<XprType>::Flags) & RowMajorBit) != 0,
39  IsRowMajor = (MaxRowsAtCompileTime == 1 && MaxColsAtCompileTime != 1) ? 1
40  : (MaxColsAtCompileTime == 1 && MaxRowsAtCompileTime != 1) ? 0
41  : XprTypeIsRowMajor,
42  HasSameStorageOrderAsXprType = (IsRowMajor == XprTypeIsRowMajor),
43  InnerSize = IsRowMajor ? int(ColsAtCompileTime) : int(RowsAtCompileTime),
44  InnerStrideAtCompileTime = HasSameStorageOrderAsXprType
45  ? int(inner_stride_at_compile_time<XprType>::ret)
46  : int(outer_stride_at_compile_time<XprType>::ret),
47  OuterStrideAtCompileTime = HasSameStorageOrderAsXprType
48  ? int(outer_stride_at_compile_time<XprType>::ret)
49  : int(inner_stride_at_compile_time<XprType>::ret),
50 
51  // FIXME, this traits is rather specialized for dense object and it needs to be cleaned further
52  FlagsLvalueBit = is_lvalue<XprType>::value ? LvalueBit : 0,
53  FlagsRowMajorBit = IsRowMajor ? RowMajorBit : 0,
54  Flags = (traits<XprType>::Flags & (DirectAccessBit | (InnerPanel ? CompressedAccessBit : 0))) | FlagsLvalueBit | FlagsRowMajorBit,
55  // FIXME DirectAccessBit should not be handled by expressions
56  //
57  // Alignment is needed by MapBase's assertions
58  // We can sefely set it to false here. Internal alignment errors will be detected by an eigen_internal_assert in the respective evaluator
59  Alignment = 0
60  };

104  template<typename XprType, int BlockRows, int BlockCols, bool InnerPanel> class Block
105  : public BlockImpl<XprType, BlockRows, BlockCols, InnerPanel, typename internal::traits<XprType>::StorageKind> {

114 
117  EIGEN_DEVICE_FUNC
118  inline Block(XprType& xpr, Index i) : Impl(xpr, i) {
119  eigen_assert((i >= 0) && (
120  ((BlockRows == 1) && (BlockCols == XprType::ColsAtCompileTime) && i < xpr.rows())
121  || ((BlockRows == XprType::RowsAtCompileTime) && (BlockCols == 1) && i < xpr.cols())));
122  }
123 
126  EIGEN_DEVICE_FUNC
127  inline Block(XprType& xpr, Index startRow, Index startCol)
128  : Impl(xpr, startRow, startCol) {
129  EIGEN_STATIC_ASSERT(RowsAtCompileTime != Dynamic && ColsAtCompileTime != Dynamic, THIS_METHOD_IS_ONLY_FOR_FIXED_SIZE)
130  eigen_assert(startRow >= 0 && BlockRows >= 0 && startRow + BlockRows <= xpr.rows()
131  && startCol >= 0 && BlockCols >= 0 && startCol + BlockCols <= xpr.cols());
132  }
133 
136  EIGEN_DEVICE_FUNC
137  inline Block(XprType& xpr,
138  Index startRow, Index startCol,
139  Index blockRows, Index blockCols)
140  : Impl(xpr, startRow, startCol, blockRows, blockCols) {
141  eigen_assert((RowsAtCompileTime == Dynamic || RowsAtCompileTime == blockRows)
142  && (ColsAtCompileTime == Dynamic || ColsAtCompileTime == blockCols));
143  eigen_assert(startRow >= 0 && blockRows >= 0 && startRow <= xpr.rows() - blockRows
144  && startCol >= 0 && blockCols >= 0 && startCol <= xpr.cols() - blockCols);
145  }
