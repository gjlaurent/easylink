41  template<typename Dst, typename Src>
42  class vml_assign_traits {
43  private:
44 
45  enum {
46  DstHasDirectAccess = Dst::Flags & DirectAccessBit,
47  SrcHasDirectAccess = Src::Flags & DirectAccessBit,
48  StorageOrdersAgree = (int(Dst::IsRowMajor) == int(Src::IsRowMajor)),
49  InnerSize = int(Dst::IsVectorAtCompileTime) ? int(Dst::SizeAtCompileTime)
50  : int(Dst::Flags) & RowMajorBit ? int(Dst::ColsAtCompileTime)
51  : int(Dst::RowsAtCompileTime),
52  InnerMaxSize = int(Dst::IsVectorAtCompileTime) ? int(Dst::MaxSizeAtCompileTime)
53  : int(Dst::Flags) & RowMajorBit ? int(Dst::MaxColsAtCompileTime)
54  : int(Dst::MaxRowsAtCompileTime),
55  MaxSizeAtCompileTime = Dst::SizeAtCompileTime,
56 
57  MightEnableVml = StorageOrdersAgree && DstHasDirectAccess && SrcHasDirectAccess && Src::InnerStrideAtCompileTime == 1 && Dst::InnerStrideAtCompileTime == 1,
58  MightLinearize = MightEnableVml && (int(Dst::Flags) & int(Src::Flags) & LinearAccessBit),
59  VmlSize = MightLinearize ? MaxSizeAtCompileTime : InnerMaxSize,
60  LargeEnough = VmlSize == Dynamic || VmlSize >= EIGEN_MKL_VML_THRESHOLD
61  };
62  public:
63 
64  enum {
65  EnableVml = MightEnableVml && LargeEnough,
66  Traversal = MightLinearize ? LinearTraversal : DefaultTraversal
67  };
68  };

83 #define EIGEN_MKL_VML_DECLARE_UNARY_CALL(EIGENOP, VMLOP, EIGENTYPE, VMLTYPE, VMLMODE) \
84  template< typename DstXprType, typename SrcXprNested> \
85  struct Assignment<DstXprType, CwiseUnaryOp<scalar_##EIGENOP##_op<EIGENTYPE>, SrcXprNested>, assign_op<EIGENTYPE,EIGENTYPE>, \
86  Dense2Dense, typename enable_if<vml_assign_traits<DstXprType,SrcXprNested>::EnableVml>::type> { \
87  typedef CwiseUnaryOp<scalar_##EIGENOP##_op<EIGENTYPE>, SrcXprNested> SrcXprType; \
88  static void run(DstXprType &dst, const SrcXprType &src, const assign_op<EIGENTYPE,EIGENTYPE> &func) { \
89  resize_if_allowed(dst, src, func); \
90  eigen_assert(dst.rows() == src.rows() && dst.cols() == src.cols()); \
91  if(vml_assign_traits<DstXprType,SrcXprNested>::Traversal==LinearTraversal) { \
92  VMLOP(dst.size(), (const VMLTYPE*)src.nestedExpression().data(), \
93  (VMLTYPE*)dst.data() EIGEN_PP_EXPAND(EIGEN_VMLMODE_EXPAND_##VMLMODE) ); \
94  } else { \
95  const Index outerSize = dst.outerSize(); \
96  for(Index outer = 0; outer < outerSize; ++outer) { \
97  const EIGENTYPE *src_ptr = src.IsRowMajor ? &(src.nestedExpression().coeffRef(outer,0)) : \
98  &(src.nestedExpression().coeffRef(0, outer)); \
99  EIGENTYPE *dst_ptr = dst.IsRowMajor ? &(dst.coeffRef(outer,0)) : &(dst.coeffRef(0, outer)); \
100  VMLOP( dst.innerSize(), (const VMLTYPE*)src_ptr, \
101  (VMLTYPE*)dst_ptr EIGEN_PP_EXPAND(EIGEN_VMLMODE_EXPAND_##VMLMODE)); \
102  } \
103  } \
104  } \
105  }; \

142 #define EIGEN_MKL_VML_DECLARE_POW_CALL(EIGENOP, VMLOP, EIGENTYPE, VMLTYPE, VMLMODE) \
143  template< typename DstXprType, typename SrcXprNested, typename Plain> \
144  struct Assignment<DstXprType, CwiseBinaryOp<scalar_##EIGENOP##_op<EIGENTYPE,EIGENTYPE>, SrcXprNested, \
145  const CwiseNullaryOp<internal::scalar_constant_op<EIGENTYPE>,Plain> >, assign_op<EIGENTYPE,EIGENTYPE>, \
146  Dense2Dense, typename enable_if<vml_assign_traits<DstXprType,SrcXprNested>::EnableVml>::type> { \
147  typedef CwiseBinaryOp<scalar_##EIGENOP##_op<EIGENTYPE,EIGENTYPE>, SrcXprNested, \
148  const CwiseNullaryOp<internal::scalar_constant_op<EIGENTYPE>,Plain> > SrcXprType; \
149  static void run(DstXprType &dst, const SrcXprType &src, const assign_op<EIGENTYPE,EIGENTYPE> &func) { \
150  resize_if_allowed(dst, src, func); \
151  eigen_assert(dst.rows() == src.rows() && dst.cols() == src.cols()); \
152  VMLTYPE exponent = reinterpret_cast<const VMLTYPE&>(src.rhs().functor().m_other); \
153  if(vml_assign_traits<DstXprType,SrcXprNested>::Traversal==LinearTraversal) \
154  { \
155  VMLOP( dst.size(), (const VMLTYPE*)src.lhs().data(), exponent, \
156  (VMLTYPE*)dst.data() EIGEN_PP_EXPAND(EIGEN_VMLMODE_EXPAND_##VMLMODE) ); \
157  } else { \
158  const Index outerSize = dst.outerSize(); \
159  for(Index outer = 0; outer < outerSize; ++outer) { \
160  const EIGENTYPE *src_ptr = src.IsRowMajor ? &(src.lhs().coeffRef(outer,0)) : \
161  &(src.lhs().coeffRef(0, outer)); \
162  EIGENTYPE *dst_ptr = dst.IsRowMajor ? &(dst.coeffRef(outer,0)) : &(dst.coeffRef(0, outer)); \
163  VMLOP( dst.innerSize(), (const VMLTYPE*)src_ptr, exponent, \
164  (VMLTYPE*)dst_ptr EIGEN_PP_EXPAND(EIGEN_VMLMODE_EXPAND_##VMLMODE)); \
165  } \
166  } \
167  } \
168  };
