27  template<typename XprType>
28  struct CommaInitializer {

31  EIGEN_DEVICE_FUNC
32  inline CommaInitializer(XprType& xpr, const Scalar& s)

37  template<typename OtherDerived>
38  EIGEN_DEVICE_FUNC
39  inline CommaInitializer(XprType& xpr, const DenseBase<OtherDerived>& other)
40  : m_xpr(xpr), m_row(0), m_col(other.cols()), m_currentBlockRows(other.rows()) {
41  m_xpr.block(0, 0, other.rows(), other.cols()) = other;

48  EIGEN_DEVICE_FUNC
49  inline CommaInitializer(const CommaInitializer& o)
50  : m_xpr(o.m_xpr), m_row(o.m_row), m_col(o.m_col), m_currentBlockRows(o.m_currentBlockRows) {
51  // Mark original object as finished. In absence of R-value references we need to const_cast:
52  const_cast<CommaInitializer&> (o).m_row = m_xpr.rows();
53  const_cast<CommaInitializer&> (o).m_col = m_xpr.cols();
54  const_cast<CommaInitializer&> (o).m_currentBlockRows = 0;
55  }

57  /* inserts a scalar value in the target matrix */
58  EIGEN_DEVICE_FUNC
59  CommaInitializer& operator,(const Scalar& s) {
60  if (m_col == m_xpr.cols()) {
61  m_row += m_currentBlockRows;
62  m_col = 0;
63  m_currentBlockRows = 1;
64  eigen_assert(m_row < m_xpr.rows()
65  && "Too many rows passed to comma initializer (operator<<)");
66  }
67  eigen_assert(m_col < m_xpr.cols()
68  && "Too many coefficients passed to comma initializer (operator<<)");
69  eigen_assert(m_currentBlockRows == 1);
70  m_xpr.coeffRef(m_row, m_col++) = s;
71  return *this;
72  }
73 
74  /* inserts a matrix expression in the target matrix */
75  template<typename OtherDerived>
76  EIGEN_DEVICE_FUNC
77  CommaInitializer& operator,(const DenseBase<OtherDerived>& other) {
78  if (m_col == m_xpr.cols() && (other.cols() != 0 || other.rows() != m_currentBlockRows)) {
79  m_row += m_currentBlockRows;
80  m_col = 0;
81  m_currentBlockRows = other.rows();
82  eigen_assert(m_row + m_currentBlockRows <= m_xpr.rows()
83  && "Too many rows passed to comma initializer (operator<<)");
84  }
85  eigen_assert((m_col + other.cols() <= m_xpr.cols())
86  && "Too many coefficients passed to comma initializer (operator<<)");
87  eigen_assert(m_currentBlockRows == other.rows());
88  m_xpr.template block<OtherDerived::RowsAtCompileTime, OtherDerived::ColsAtCompileTime>
89  (m_row, m_col, other.rows(), other.cols()) = other;
90  m_col += other.cols();
91  return *this;
92  }

110  EIGEN_DEVICE_FUNC
111  inline XprType& finished() {
112  eigen_assert(((m_row + m_currentBlockRows) == m_xpr.rows() || m_xpr.cols() == 0)
113  && m_col == m_xpr.cols()
114  && "Too few coefficients passed to comma initializer (operator<<)");
115  return m_xpr;
116  }
117 
118  XprType& m_xpr; // target expression
119  Index m_row; // current row id
120  Index m_col; // current col id
121  Index m_currentBlockRows; // current block height
122  };
