31  struct triangular_solve_vector<LhsScalar, RhsScalar, Index, OnTheLeft, Mode, Conjugate, RowMajor> {
32 
33  enum {
34  IsLower = ((Mode & Lower) == Lower)
35  };
36 
37  static void run(Index size, const LhsScalar* _lhs, Index lhsStride, RhsScalar* rhs) {
38  typedef Map<const Matrix<LhsScalar, Dynamic, Dynamic, RowMajor>, 0, OuterStride<> > LhsMap;
39  const LhsMap lhs(_lhs, size, size, OuterStride<>(lhsStride));
40 
41  typedef const_blas_data_mapper<LhsScalar, Index, RowMajor> LhsMapper;
42  typedef const_blas_data_mapper<RhsScalar, Index, ColMajor> RhsMapper;
43 
44  typename internal::conditional<
45  Conjugate,
46  const CwiseUnaryOp<typename internal::scalar_conjugate_op<LhsScalar>, LhsMap>,
47  const LhsMap&>
48  ::type cjLhs(lhs);
49  static const Index PanelWidth = EIGEN_TUNE_TRIANGULAR_PANEL_WIDTH;
50  for (Index pi = IsLower ? 0 : size;
51  IsLower ? pi < size : pi > 0;
52  IsLower ? pi += PanelWidth : pi -= PanelWidth) {
53  Index actualPanelWidth = (std::min)(IsLower ? size - pi : pi, PanelWidth);
54 
55  Index r = IsLower ? pi : size - pi; // remaining size
56  if (r > 0) {
57  // let's directly call the low level product function because:
58  // 1 - it is faster to compile
59  // 2 - it is slighlty faster at runtime
60  Index startRow = IsLower ? pi : pi - actualPanelWidth;
61  Index startCol = IsLower ? 0 : pi;
62 
63  general_matrix_vector_product<Index, LhsScalar, LhsMapper, RowMajor, Conjugate, RhsScalar, RhsMapper, false>::run(
64  actualPanelWidth, r,
65  LhsMapper(&lhs.coeffRef(startRow, startCol), lhsStride),
66  RhsMapper(rhs + startCol, 1),
67  rhs + startRow, 1,
68  RhsScalar(-1));
69  }
70 
71  for (Index k = 0; k < actualPanelWidth; ++k) {
72  Index i = IsLower ? pi + k : pi - k - 1;
73  Index s = IsLower ? pi : i + 1;
74  if (k > 0)
75  rhs[i] -= (cjLhs.row(i).segment(s, k).transpose().cwiseProduct(Map<const Matrix<RhsScalar, Dynamic, 1> >(rhs + s, k))).sum();
76 
77  if (!(Mode & UnitDiag))
78  rhs[i] /= cjLhs(i, i);
79  }
80  }
81  }
82  };

87  struct triangular_solve_vector<LhsScalar, RhsScalar, Index, OnTheLeft, Mode, Conjugate, ColMajor> {

89  enum {
90  IsLower = ((Mode & Lower) == Lower)
91  };

93  static void run(Index size, const LhsScalar* _lhs, Index lhsStride, RhsScalar* rhs) {
94  typedef Map<const Matrix<LhsScalar, Dynamic, Dynamic, ColMajor>, 0, OuterStride<> > LhsMap;
95  const LhsMap lhs(_lhs, size, size, OuterStride<>(lhsStride));
96  typedef const_blas_data_mapper<LhsScalar, Index, ColMajor> LhsMapper;
97  typedef const_blas_data_mapper<RhsScalar, Index, ColMajor> RhsMapper;
98  typename internal::conditional<Conjugate,
99  const CwiseUnaryOp<typename internal::scalar_conjugate_op<LhsScalar>, LhsMap>,
100  const LhsMap&
101  >::type cjLhs(lhs);
102  static const Index PanelWidth = EIGEN_TUNE_TRIANGULAR_PANEL_WIDTH;
103 
104  for (Index pi = IsLower ? 0 : size;
105  IsLower ? pi < size : pi > 0;
106  IsLower ? pi += PanelWidth : pi -= PanelWidth) {
107  Index actualPanelWidth = (std::min)(IsLower ? size - pi : pi, PanelWidth);
108  Index startBlock = IsLower ? pi : pi - actualPanelWidth;
109  Index endBlock = IsLower ? pi + actualPanelWidth : 0;
110 
111  for (Index k = 0; k < actualPanelWidth; ++k) {
112  Index i = IsLower ? pi + k : pi - k - 1;
113  if (!(Mode & UnitDiag))
114  rhs[i] /= cjLhs.coeff(i, i);
115 
116  Index r = actualPanelWidth - k - 1; // remaining size
117  Index s = IsLower ? i + 1 : i - r;
118  if (r > 0)
119  Map<Matrix<RhsScalar, Dynamic, 1> >(rhs + s, r) -= rhs[i] * cjLhs.col(i).segment(s, r);
120  }
121  Index r = IsLower ? size - endBlock : startBlock; // remaining size
122  if (r > 0) {
123  // let's directly call the low level product function because:
124  // 1 - it is faster to compile
125  // 2 - it is slighlty faster at runtime
126  general_matrix_vector_product<Index, LhsScalar, LhsMapper, ColMajor, Conjugate, RhsScalar, RhsMapper, false>::run(
127  r, actualPanelWidth,
128  LhsMapper(&lhs.coeffRef(endBlock, startBlock), lhsStride),
129  RhsMapper(rhs + startBlock, 1),
130  rhs + endBlock, 1, RhsScalar(-1));
131  }
132  }
133  }
134  };
