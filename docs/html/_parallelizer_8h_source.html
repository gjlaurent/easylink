18  inline void manage_multi_threading(Action action, int* v) {
19  static EIGEN_UNUSED int m_maxThreads = -1;
20 
21  if (action == SetAction) {
22  eigen_internal_assert(v != 0);
23  m_maxThreads = *v;
24  } else if (action == GetAction) {
25  eigen_internal_assert(v != 0);
26 #ifdef EIGEN_HAS_OPENMP
27  if (m_maxThreads > 0)
28  *v = m_maxThreads;
29  else
30  *v = omp_get_max_threads();
31 #else
32  *v = 1;
33 #endif
34  } else {
35  eigen_internal_assert(false);
36  }
37  }
38 
39  }

77  template<bool Condition, typename Functor, typename Index>
78  void parallelize_gemm(const Functor& func, Index rows, Index cols, Index depth, bool transpose) {
79  // TODO when EIGEN_USE_BLAS is defined,
80  // we should still enable OMP for other scalar types
81 #if !(defined (EIGEN_HAS_OPENMP)) || defined (EIGEN_USE_BLAS)
82  // FIXME the transpose variable is only needed to properly split
83  // the matrix product when multithreading is enabled. This is a temporary
84  // fix to support row-major destination matrices. This whole
85  // parallelizer mechanism has to be redisigned anyway.
86  EIGEN_UNUSED_VARIABLE(depth);
87  EIGEN_UNUSED_VARIABLE(transpose);
88  func(0, rows, 0, cols);
89 #else
90 
91  // Dynamically check whether we should enable or disable OpenMP.
92  // The conditions are:
93  // - the max number of threads we can create is greater than 1
94  // - we are not already in a parallel code
95  // - the sizes are large enough
96 
97  // compute the maximal number of threads from the size of the product:
98  // This first heuristic takes into account that the product kernel is fully optimized when working with nr columns at once.
99  Index size = transpose ? rows : cols;
100  Index pb_max_threads = std::max<Index>(1, size / Functor::Traits::nr);
101 
102  // compute the maximal number of threads from the total amount of work:
103  double work = static_cast<double> (rows) * static_cast<double> (cols) *
104  static_cast<double> (depth);
105  double kMinTaskSize = 50000; // FIXME improve this heuristic.
106  pb_max_threads = std::max<Index>(1, std::min<Index>(pb_max_threads, work / kMinTaskSize));
107 
108  // compute the number of threads we are going to use
109  Index threads = std::min<Index>(nbThreads(), pb_max_threads);
110 
111  // if multi-threading is explicitely disabled, not useful, or if we already are in a parallel session,
112  // then abort multi-threading
113  // FIXME omp_get_num_threads()>1 only works for openmp, what if the user does not use openmp?
114  if ((!Condition) || (threads == 1) || (omp_get_num_threads() > 1))
115  return func(0, rows, 0, cols);
116 
117  Eigen::initParallel();
118  func.initParallelSession(threads);
119 
120  if (transpose)
121  std::swap(rows, cols);
122 
123  ei_declare_aligned_stack_constructed_variable(GemmParallelInfo<Index>, info, threads, 0);
124 
125 #pragma omp parallel num_threads(threads)
126  {
127  Index i = omp_get_thread_num();
128  // Note that the actual number of threads might be lower than the number of request ones.
129  Index actual_threads = omp_get_num_threads();
130 
131  Index blockCols = (cols / actual_threads) & ~Index(0x3);
132  Index blockRows = (rows / actual_threads);
133  blockRows = (blockRows / Functor::Traits::mr) * Functor::Traits::mr;
134 
135  Index r0 = i*blockRows;
136  Index actualBlockRows = (i + 1 == actual_threads) ? rows - r0 : blockRows;
137 
138  Index c0 = i*blockCols;
139  Index actualBlockCols = (i + 1 == actual_threads) ? cols - c0 : blockCols;
140 
141  info[i].lhs_start = r0;
142  info[i].lhs_length = actualBlockRows;
143 
144  if (transpose) func(c0, actualBlockCols, 0, rows, info);
145  else func(0, rows, c0, actualBlockCols, info);
146  }
147 #endif
148  }
