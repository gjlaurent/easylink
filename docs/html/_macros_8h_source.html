637 //------------------------------------------------------------------------------------------
638 // Static and dynamic alignment control
639 //
640 // The main purpose of this section is to define EIGEN_MAX_ALIGN_BYTES and EIGEN_MAX_STATIC_ALIGN_BYTES
641 // as the maximal boundary in bytes on which dynamically and statically allocated data may be alignment respectively.
642 // The values of EIGEN_MAX_ALIGN_BYTES and EIGEN_MAX_STATIC_ALIGN_BYTES can be specified by the user. If not,
643 // a default value is automatically computed based on architecture, compiler, and OS.
644 //
645 // This section also defines macros EIGEN_ALIGN_TO_BOUNDARY(N) and the shortcuts EIGEN_ALIGN{8,16,32,_MAX}
646 // to be used to declare statically aligned buffers.
647 //------------------------------------------------------------------------------------------
648 
649 
650 /* EIGEN_ALIGN_TO_BOUNDARY(n) forces data to be n-byte aligned. This is used to satisfy SIMD requirements.
651  * However, we do that EVEN if vectorization (EIGEN_VECTORIZE) is disabled,
652  * so that vectorization doesn't affect binary compatibility.
653  *
654  * If we made alignment depend on whether or not EIGEN_VECTORIZE is defined, it would be impossible to link
655  * vectorized and non-vectorized code.
656  */
657 #if (defined __CUDACC__)
658 #define EIGEN_ALIGN_TO_BOUNDARY(n) __align__(n)
659 #elif EIGEN_COMP_GNUC || EIGEN_COMP_PGI || EIGEN_COMP_IBM || EIGEN_COMP_ARM
660 #define EIGEN_ALIGN_TO_BOUNDARY(n) __attribute__((aligned(n)))
661 #elif EIGEN_COMP_MSVC
662 #define EIGEN_ALIGN_TO_BOUNDARY(n) __declspec(align(n))
663 #elif EIGEN_COMP_SUNCC
664 // FIXME not sure about this one:
665 #define EIGEN_ALIGN_TO_BOUNDARY(n) __attribute__((aligned(n)))
666 #else
667 #error Please tell me what is the equivalent of __attribute__((aligned(n))) for your compiler
668 #endif

886 // EIGEN_SIZE_MIN_PREFER_DYNAMIC gives the min between compile-time sizes. 0 has absolute priority, followed by 1,
887 // followed by Dynamic, followed by other finite values. The reason for giving Dynamic the priority over
888 // finite values is that min(3, Dynamic) should be Dynamic, since that could be anything between 0 and 3.
889 #define EIGEN_SIZE_MIN_PREFER_DYNAMIC(a,b) (((int)a == 0 || (int)b == 0) ? 0 \
890  : ((int)a == 1 || (int)b == 1) ? 1 \
891  : ((int)a == Dynamic || (int)b == Dynamic) ? Dynamic \
892  : ((int)a <= (int)b) ? (int)a : (int)b)
893 
894 // EIGEN_SIZE_MIN_PREFER_FIXED is a variant of EIGEN_SIZE_MIN_PREFER_DYNAMIC comparing MaxSizes. The difference is that finite values
895 // now have priority over Dynamic, so that min(3, Dynamic) gives 3. Indeed, whatever the actual value is
896 // (between 0 and 3), it is not more than 3.
897 #define EIGEN_SIZE_MIN_PREFER_FIXED(a,b) (((int)a == 0 || (int)b == 0) ? 0 \
898  : ((int)a == 1 || (int)b == 1) ? 1 \
899  : ((int)a == Dynamic && (int)b == Dynamic) ? Dynamic \
900  : ((int)a == Dynamic) ? (int)b \
901  : ((int)b == Dynamic) ? (int)a \
902  : ((int)a <= (int)b) ? (int)a : (int)b)
