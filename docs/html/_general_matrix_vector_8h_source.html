17  /* Optimized col-major matrix * vector product:
18  * This algorithm processes 4 columns at onces that allows to both reduce
19  * the number of load/stores of the result by a factor 4 and to reduce
20  * the instruction dependency. Moreover, we know that all bands have the
21  * same alignment pattern.
22  *
23  * Mixing type logic: C += alpha * A * B
24  * | A | B |alpha| comments
25  * |real |cplx |cplx | no vectorization
26  * |real |cplx |real | alpha is converted to a cplx when calling the run function, no vectorization
27  * |cplx |real |cplx | invalid, the caller has to do tmp: = A * B; C += alpha*tmp
28  * |cplx |real |real | optimal case, vectorization possible via real-cplx mul
29  *
30  * Accesses to the matrix coefficients follow the following logic:
31  *
32  * - if all columns have the same alignment then
33  * - if the columns have the same alignment as the result vector, then easy! (-> AllAligned case)
34  * - otherwise perform unaligned loads only (-> NoneAligned case)
35  * - otherwise
36  * - if even columns have the same alignment then
37  * // odd columns are guaranteed to have the same alignment too
38  * - if even or odd columns have the same alignment as the result, then
39  * // for a register size of 2 scalars, this is guarantee to be the case (e.g., SSE with double)
40  * - perform half aligned and half unaligned loads (-> EvenAligned case)
41  * - otherwise perform unaligned loads only (-> NoneAligned case)
42  * - otherwise, if the register size is 4 scalars (e.g., SSE with float) then
43  * - one over 4 consecutive columns is guaranteed to be aligned with the result vector,
44  * perform simple aligned loads for this column and aligned loads plus re-alignment for the other. (-> FirstAligned case)
45  * // this re-alignment is done by the palign function implemented for SSE in Eigen/src/Core/arch/SSE/PacketMath.h
46  * - otherwise,
47  * // if we get here, this means the register size is greater than 4 (e.g., AVX with floats),
48  * // we currently fall back to the NoneAligned case
49  *
50  * The same reasoning apply for the transposed case.
51  *
52  * The last case (PacketSize>4) could probably be improved by generalizing the FirstAligned case, but since we do not support AVX yet...
53  * One might also wonder why in the EvenAligned case we perform unaligned loads instead of using the aligned-loads plus re-alignment
54  * strategy as in the FirstAligned case. The reason is that we observed that unaligned loads on a 8 byte boundary are not too slow
55  * compared to unaligned loads on a 4 byte boundary.
56  *
57  */

86  template<typename Index, typename LhsScalar, typename LhsMapper, bool ConjugateLhs, typename RhsScalar, typename RhsMapper, bool ConjugateRhs, int Version>
87  EIGEN_DONT_INLINE void general_matrix_vector_product<Index, LhsScalar, LhsMapper, ColMajor, ConjugateLhs, RhsScalar, RhsMapper, ConjugateRhs, Version>::run(
88  Index rows, Index cols,
89  const LhsMapper& lhs,
90  const RhsMapper& rhs,
91  ResScalar* res, Index resIncr,
92  RhsScalar alpha) {
93  EIGEN_UNUSED_VARIABLE(resIncr);
94  eigen_internal_assert(resIncr == 1);
95 #ifdef _EIGEN_ACCUMULATE_PACKETS
96 #error _EIGEN_ACCUMULATE_PACKETS has already been defined
97 #endif
98 #define _EIGEN_ACCUMULATE_PACKETS(Alignment0,Alignment13,Alignment2) \
99  pstore(&res[j], \
100  padd(pload<ResPacket>(&res[j]), \
101  padd( \
102  padd(pcj.pmul(lhs0.template load<LhsPacket, Alignment0>(j), ptmp0), \
103  pcj.pmul(lhs1.template load<LhsPacket, Alignment13>(j), ptmp1)), \
104  padd(pcj.pmul(lhs2.template load<LhsPacket, Alignment2>(j), ptmp2), \
105  pcj.pmul(lhs3.template load<LhsPacket, Alignment13>(j), ptmp3)) )))
106 
107  typedef typename LhsMapper::VectorMapper LhsScalars;
108 
109  conj_helper<LhsScalar, RhsScalar, ConjugateLhs, ConjugateRhs> cj;
110  conj_helper<LhsPacket, RhsPacket, ConjugateLhs, ConjugateRhs> pcj;
111  if (ConjugateRhs)
112  alpha = numext::conj(alpha);
113 
114  enum {
115  AllAligned = 0, EvenAligned, FirstAligned, NoneAligned
116  };
117  const Index columnsAtOnce = 4;
118  const Index peels = 2;
119  const Index LhsPacketAlignedMask = LhsPacketSize - 1;
120  const Index ResPacketAlignedMask = ResPacketSize - 1;
121  // const Index PeelAlignedMask = ResPacketSize*peels-1;
122  const Index size = rows;
123 
124  const Index lhsStride = lhs.stride();
125 
126  // How many coeffs of the result do we have to skip to be aligned.
127  // Here we assume data are at least aligned on the base scalar type.
128  Index alignedStart = internal::first_default_aligned(res, size);
129  Index alignedSize = ResPacketSize > 1 ? alignedStart + ((size - alignedStart) & ~ResPacketAlignedMask) : 0;
130  const Index peeledSize = alignedSize - RhsPacketSize * peels - RhsPacketSize + 1;
131 
132  const Index alignmentStep = LhsPacketSize > 1 ? (LhsPacketSize - lhsStride % LhsPacketSize) & LhsPacketAlignedMask : 0;
133  Index alignmentPattern = alignmentStep == 0 ? AllAligned
134  : alignmentStep == (LhsPacketSize / 2) ? EvenAligned
135  : FirstAligned;
136 
137  // we cannot assume the first element is aligned because of sub-matrices
138  const Index lhsAlignmentOffset = lhs.firstAligned(size);
139 
140  // find how many columns do we have to skip to be aligned with the result (if possible)
141  Index skipColumns = 0;
142  // if the data cannot be aligned (TODO add some compile time tests when possible, e.g. for floats)
143  if ((lhsAlignmentOffset < 0) || (lhsAlignmentOffset == size) || (UIntPtr(res) % sizeof (ResScalar))) {
144  alignedSize = 0;
145  alignedStart = 0;
146  alignmentPattern = NoneAligned;
147  } else if (LhsPacketSize > 4) {
148  // TODO: extend the code to support aligned loads whenever possible when LhsPacketSize > 4.
149  // Currently, it seems to be better to perform unaligned loads anyway
150  alignmentPattern = NoneAligned;
151  } else if (LhsPacketSize > 1) {
152  // eigen_internal_assert(size_t(firstLhs+lhsAlignmentOffset)%sizeof(LhsPacket)==0 || size<LhsPacketSize);
153 
154  while (skipColumns < LhsPacketSize &&
155  alignedStart != ((lhsAlignmentOffset + alignmentStep * skipColumns) % LhsPacketSize))
156  ++skipColumns;
157  if (skipColumns == LhsPacketSize) {
158  // nothing can be aligned, no need to skip any column
159  alignmentPattern = NoneAligned;
160  skipColumns = 0;
161  } else {
162  skipColumns = (std::min)(skipColumns, cols);
163  // note that the skiped columns are processed later.
164  }
165 
166  /* eigen_internal_assert( (alignmentPattern==NoneAligned)
167  || (skipColumns + columnsAtOnce >= cols)
168  || LhsPacketSize > size
169  || (size_t(firstLhs+alignedStart+lhsStride*skipColumns)%sizeof(LhsPacket))==0);*/
170  } else if (Vectorizable) {
171  alignedStart = 0;
172  alignedSize = size;
173  alignmentPattern = AllAligned;
174  }
175 
176  const Index offset1 = (alignmentPattern == FirstAligned && alignmentStep == 1) ? 3 : 1;
177  const Index offset3 = (alignmentPattern == FirstAligned && alignmentStep == 1) ? 1 : 3;
178 
179  Index columnBound = ((cols - skipColumns) / columnsAtOnce) * columnsAtOnce + skipColumns;
180  for (Index i = skipColumns; i < columnBound; i += columnsAtOnce) {
181  RhsPacket ptmp0 = pset1<RhsPacket>(alpha * rhs(i, 0)),
182  ptmp1 = pset1<RhsPacket>(alpha * rhs(i + offset1, 0)),
183  ptmp2 = pset1<RhsPacket>(alpha * rhs(i + 2, 0)),
184  ptmp3 = pset1<RhsPacket>(alpha * rhs(i + offset3, 0));
185 
186  // this helps a lot generating better binary code
187  const LhsScalars lhs0 = lhs.getVectorMapper(0, i + 0), lhs1 = lhs.getVectorMapper(0, i + offset1),
188  lhs2 = lhs.getVectorMapper(0, i + 2), lhs3 = lhs.getVectorMapper(0, i + offset3);
189 
190  if (Vectorizable) {
191  /* explicit vectorization */
192  // process initial unaligned coeffs
193  for (Index j = 0; j < alignedStart; ++j) {
194  res[j] = cj.pmadd(lhs0(j), pfirst(ptmp0), res[j]);
195  res[j] = cj.pmadd(lhs1(j), pfirst(ptmp1), res[j]);
196  res[j] = cj.pmadd(lhs2(j), pfirst(ptmp2), res[j]);
197  res[j] = cj.pmadd(lhs3(j), pfirst(ptmp3), res[j]);
198  }
199 
200  if (alignedSize > alignedStart) {
201  switch (alignmentPattern) {
202  case AllAligned:
203  for (Index j = alignedStart; j < alignedSize; j += ResPacketSize)
204  _EIGEN_ACCUMULATE_PACKETS(Aligned, Aligned, Aligned);
205  break;
206  case EvenAligned:
207  for (Index j = alignedStart; j < alignedSize; j += ResPacketSize)
208  _EIGEN_ACCUMULATE_PACKETS(Aligned, Unaligned, Aligned);
209  break;
210  case FirstAligned:
211  {
212  Index j = alignedStart;
213  if (peels > 1) {
214  LhsPacket A00, A01, A02, A03, A10, A11, A12, A13;
215  ResPacket T0, T1;
216 
217  A01 = lhs1.template load<LhsPacket, Aligned>(alignedStart - 1);
218  A02 = lhs2.template load<LhsPacket, Aligned>(alignedStart - 2);
219  A03 = lhs3.template load<LhsPacket, Aligned>(alignedStart - 3);
220 
221  for (; j < peeledSize; j += peels * ResPacketSize) {
222  A11 = lhs1.template load<LhsPacket, Aligned>(j - 1 + LhsPacketSize);
223  palign<1>(A01, A11);
224  A12 = lhs2.template load<LhsPacket, Aligned>(j - 2 + LhsPacketSize);
225  palign<2>(A02, A12);
226  A13 = lhs3.template load<LhsPacket, Aligned>(j - 3 + LhsPacketSize);
227  palign<3>(A03, A13);
228 
229  A00 = lhs0.template load<LhsPacket, Aligned>(j);
230  A10 = lhs0.template load<LhsPacket, Aligned>(j + LhsPacketSize);
231  T0 = pcj.pmadd(A00, ptmp0, pload<ResPacket>(&res[j]));
232  T1 = pcj.pmadd(A10, ptmp0, pload<ResPacket>(&res[j + ResPacketSize]));
233 
234  T0 = pcj.pmadd(A01, ptmp1, T0);
235  A01 = lhs1.template load<LhsPacket, Aligned>(j - 1 + 2 * LhsPacketSize);
236  palign<1>(A11, A01);
237  T0 = pcj.pmadd(A02, ptmp2, T0);
238  A02 = lhs2.template load<LhsPacket, Aligned>(j - 2 + 2 * LhsPacketSize);
239  palign<2>(A12, A02);
240  T0 = pcj.pmadd(A03, ptmp3, T0);
241  pstore(&res[j], T0);
242  A03 = lhs3.template load<LhsPacket, Aligned>(j - 3 + 2 * LhsPacketSize);
243  palign<3>(A13, A03);
244  T1 = pcj.pmadd(A11, ptmp1, T1);
245  T1 = pcj.pmadd(A12, ptmp2, T1);
246  T1 = pcj.pmadd(A13, ptmp3, T1);
247  pstore(&res[j + ResPacketSize], T1);
248  }
249  }
250  for (; j < alignedSize; j += ResPacketSize)
251  _EIGEN_ACCUMULATE_PACKETS(Aligned, Unaligned, Unaligned);
252  break;
253  }
254  default:
255  for (Index j = alignedStart; j < alignedSize; j += ResPacketSize)
256  _EIGEN_ACCUMULATE_PACKETS(Unaligned, Unaligned, Unaligned);
257  break;
258  }
259  }
260  } // end explicit vectorization
261 
262  /* process remaining coeffs (or all if there is no explicit vectorization) */
263  for (Index j = alignedSize; j < size; ++j) {
264  res[j] = cj.pmadd(lhs0(j), pfirst(ptmp0), res[j]);
265  res[j] = cj.pmadd(lhs1(j), pfirst(ptmp1), res[j]);
266  res[j] = cj.pmadd(lhs2(j), pfirst(ptmp2), res[j]);
267  res[j] = cj.pmadd(lhs3(j), pfirst(ptmp3), res[j]);
268  }
269  }
270 
271  // process remaining first and last columns (at most columnsAtOnce-1)
272  Index end = cols;
273  Index start = columnBound;
274  do {
275  for (Index k = start; k < end; ++k) {
276  RhsPacket ptmp0 = pset1<RhsPacket>(alpha * rhs(k, 0));
277  const LhsScalars lhs0 = lhs.getVectorMapper(0, k);
278 
279  if (Vectorizable) {
280  /* explicit vectorization */
281  // process first unaligned result's coeffs
282  for (Index j = 0; j < alignedStart; ++j)
283  res[j] += cj.pmul(lhs0(j), pfirst(ptmp0));
284  // process aligned result's coeffs
285  if (lhs0.template aligned<LhsPacket>(alignedStart))
286  for (Index i = alignedStart; i < alignedSize; i += ResPacketSize)
287  pstore(&res[i], pcj.pmadd(lhs0.template load<LhsPacket, Aligned>(i), ptmp0, pload<ResPacket>(&res[i])));
288  else
289  for (Index i = alignedStart; i < alignedSize; i += ResPacketSize)
290  pstore(&res[i], pcj.pmadd(lhs0.template load<LhsPacket, Unaligned>(i), ptmp0, pload<ResPacket>(&res[i])));
291  }
292 
293  // process remaining scalars (or all if no explicit vectorization)
294  for (Index i = alignedSize; i < size; ++i)
295  res[i] += cj.pmul(lhs0(i), pfirst(ptmp0));
296  }
297  if (skipColumns) {
298  start = 0;
299  end = skipColumns;
300  skipColumns = 0;
301  } else
302  break;
303  } while (Vectorizable);
304 #undef _EIGEN_ACCUMULATE_PACKETS
305  }

370  enum {
371  AllAligned = 0, EvenAligned = 1, FirstAligned = 2, NoneAligned = 3
372  };
373  const Index rowsAtOnce = 4;
374  const Index peels = 2;
375  const Index RhsPacketAlignedMask = RhsPacketSize - 1;
376  const Index LhsPacketAlignedMask = LhsPacketSize - 1;
377  const Index depth = cols;
378  const Index lhsStride = lhs.stride();
379 
380  // How many coeffs of the result do we have to skip to be aligned.
381  // Here we assume data are at least aligned on the base scalar type
382  // if that's not the case then vectorization is discarded, see below.
383  Index alignedStart = rhs.firstAligned(depth);
384  Index alignedSize = RhsPacketSize > 1 ? alignedStart + ((depth - alignedStart) & ~RhsPacketAlignedMask) : 0;
385  const Index peeledSize = alignedSize - RhsPacketSize * peels - RhsPacketSize + 1;
386 
387  const Index alignmentStep = LhsPacketSize > 1 ? (LhsPacketSize - lhsStride % LhsPacketSize) & LhsPacketAlignedMask : 0;
388  Index alignmentPattern = alignmentStep == 0 ? AllAligned
389  : alignmentStep == (LhsPacketSize / 2) ? EvenAligned
390  : FirstAligned;
391 
392  // we cannot assume the first element is aligned because of sub-matrices
393  const Index lhsAlignmentOffset = lhs.firstAligned(depth);
394  const Index rhsAlignmentOffset = rhs.firstAligned(rows);
395 
396  // find how many rows do we have to skip to be aligned with rhs (if possible)
397  Index skipRows = 0;
398  // if the data cannot be aligned (TODO add some compile time tests when possible, e.g. for floats)
399  if ((sizeof (LhsScalar) != sizeof (RhsScalar)) ||
400  (lhsAlignmentOffset < 0) || (lhsAlignmentOffset == depth) ||
401  (rhsAlignmentOffset < 0) || (rhsAlignmentOffset == rows)) {
402  alignedSize = 0;
403  alignedStart = 0;
404  alignmentPattern = NoneAligned;
405  } else if (LhsPacketSize > 4) {
406  // TODO: extend the code to support aligned loads whenever possible when LhsPacketSize > 4.
407  alignmentPattern = NoneAligned;
408  } else if (LhsPacketSize > 1) {
409  // eigen_internal_assert(size_t(firstLhs+lhsAlignmentOffset)%sizeof(LhsPacket)==0 || depth<LhsPacketSize);
410 
411  while (skipRows < LhsPacketSize &&
412  alignedStart != ((lhsAlignmentOffset + alignmentStep * skipRows) % LhsPacketSize))
413  ++skipRows;
414  if (skipRows == LhsPacketSize) {
415  // nothing can be aligned, no need to skip any column
416  alignmentPattern = NoneAligned;
417  skipRows = 0;
418  } else {
419  skipRows = (std::min)(skipRows, Index(rows));
420  // note that the skiped columns are processed later.
421  }
422  /* eigen_internal_assert( alignmentPattern==NoneAligned
423  || LhsPacketSize==1
424  || (skipRows + rowsAtOnce >= rows)
425  || LhsPacketSize > depth
426  || (size_t(firstLhs+alignedStart+lhsStride*skipRows)%sizeof(LhsPacket))==0);*/
427  } else if (Vectorizable) {
428  alignedStart = 0;
429  alignedSize = depth;
430  alignmentPattern = AllAligned;
431  }
432 
433  const Index offset1 = (alignmentPattern == FirstAligned && alignmentStep == 1) ? 3 : 1;
434  const Index offset3 = (alignmentPattern == FirstAligned && alignmentStep == 1) ? 1 : 3;
435 
436  Index rowBound = ((rows - skipRows) / rowsAtOnce) * rowsAtOnce + skipRows;
437  for (Index i = skipRows; i < rowBound; i += rowsAtOnce) {
438  // FIXME: what is the purpose of this EIGEN_ALIGN_DEFAULT ??
439  EIGEN_ALIGN_MAX ResScalar tmp0 = ResScalar(0);
440  ResScalar tmp1 = ResScalar(0), tmp2 = ResScalar(0), tmp3 = ResScalar(0);
441 
442  // this helps the compiler generating good binary code
443  const LhsScalars lhs0 = lhs.getVectorMapper(i + 0, 0), lhs1 = lhs.getVectorMapper(i + offset1, 0),
444  lhs2 = lhs.getVectorMapper(i + 2, 0), lhs3 = lhs.getVectorMapper(i + offset3, 0);
445 
446  if (Vectorizable) {
447  /* explicit vectorization */
448  ResPacket ptmp0 = pset1<ResPacket>(ResScalar(0)), ptmp1 = pset1<ResPacket>(ResScalar(0)),
449  ptmp2 = pset1<ResPacket>(ResScalar(0)), ptmp3 = pset1<ResPacket>(ResScalar(0));
450 
451  // process initial unaligned coeffs
452  // FIXME this loop get vectorized by the compiler !
453  for (Index j = 0; j < alignedStart; ++j) {
454  RhsScalar b = rhs(j, 0);
455  tmp0 += cj.pmul(lhs0(j), b);
456  tmp1 += cj.pmul(lhs1(j), b);
457  tmp2 += cj.pmul(lhs2(j), b);
458  tmp3 += cj.pmul(lhs3(j), b);
459  }
460 
461  if (alignedSize > alignedStart) {
462  switch (alignmentPattern) {
463  case AllAligned:
464  for (Index j = alignedStart; j < alignedSize; j += RhsPacketSize)
465  _EIGEN_ACCUMULATE_PACKETS(Aligned, Aligned, Aligned);
466  break;
467  case EvenAligned:
468  for (Index j = alignedStart; j < alignedSize; j += RhsPacketSize)
469  _EIGEN_ACCUMULATE_PACKETS(Aligned, Unaligned, Aligned);
470  break;
471  case FirstAligned:
472  {
473  Index j = alignedStart;
474  if (peels > 1) {
475  /* Here we proccess 4 rows with with two peeled iterations to hide
476  * the overhead of unaligned loads. Moreover unaligned loads are handled
477  * using special shift/move operations between the two aligned packets
478  * overlaping the desired unaligned packet. This is *much* more efficient
479  * than basic unaligned loads.
480  */
481  LhsPacket A01, A02, A03, A11, A12, A13;
482  A01 = lhs1.template load<LhsPacket, Aligned>(alignedStart - 1);
483  A02 = lhs2.template load<LhsPacket, Aligned>(alignedStart - 2);
484  A03 = lhs3.template load<LhsPacket, Aligned>(alignedStart - 3);
485 
486  for (; j < peeledSize; j += peels * RhsPacketSize) {
487  RhsPacket b = rhs.getVectorMapper(j, 0).template load<RhsPacket, Aligned>(0);
488  A11 = lhs1.template load<LhsPacket, Aligned>(j - 1 + LhsPacketSize);
489  palign<1>(A01, A11);
490  A12 = lhs2.template load<LhsPacket, Aligned>(j - 2 + LhsPacketSize);
491  palign<2>(A02, A12);
492  A13 = lhs3.template load<LhsPacket, Aligned>(j - 3 + LhsPacketSize);
493  palign<3>(A03, A13);
494 
495  ptmp0 = pcj.pmadd(lhs0.template load<LhsPacket, Aligned>(j), b, ptmp0);
496  ptmp1 = pcj.pmadd(A01, b, ptmp1);
497  A01 = lhs1.template load<LhsPacket, Aligned>(j - 1 + 2 * LhsPacketSize);
498  palign<1>(A11, A01);
499  ptmp2 = pcj.pmadd(A02, b, ptmp2);
500  A02 = lhs2.template load<LhsPacket, Aligned>(j - 2 + 2 * LhsPacketSize);
501  palign<2>(A12, A02);
502  ptmp3 = pcj.pmadd(A03, b, ptmp3);
503  A03 = lhs3.template load<LhsPacket, Aligned>(j - 3 + 2 * LhsPacketSize);
504  palign<3>(A13, A03);
505 
506  b = rhs.getVectorMapper(j + RhsPacketSize, 0).template load<RhsPacket, Aligned>(0);
507  ptmp0 = pcj.pmadd(lhs0.template load<LhsPacket, Aligned>(j + LhsPacketSize), b, ptmp0);
508  ptmp1 = pcj.pmadd(A11, b, ptmp1);
509  ptmp2 = pcj.pmadd(A12, b, ptmp2);
510  ptmp3 = pcj.pmadd(A13, b, ptmp3);
511  }
512  }
513  for (; j < alignedSize; j += RhsPacketSize)
514  _EIGEN_ACCUMULATE_PACKETS(Aligned, Unaligned, Unaligned);
515  break;
516  }
517  default:
518  for (Index j = alignedStart; j < alignedSize; j += RhsPacketSize)
519  _EIGEN_ACCUMULATE_PACKETS(Unaligned, Unaligned, Unaligned);
520  break;
521  }
522  tmp0 += predux(ptmp0);
523  tmp1 += predux(ptmp1);
524  tmp2 += predux(ptmp2);
525  tmp3 += predux(ptmp3);
526  }
527  } // end explicit vectorization
528 
529  // process remaining coeffs (or all if no explicit vectorization)
530  // FIXME this loop get vectorized by the compiler !
531  for (Index j = alignedSize; j < depth; ++j) {
532  RhsScalar b = rhs(j, 0);
533  tmp0 += cj.pmul(lhs0(j), b);
534  tmp1 += cj.pmul(lhs1(j), b);
535  tmp2 += cj.pmul(lhs2(j), b);
536  tmp3 += cj.pmul(lhs3(j), b);
537  }
538  res[i * resIncr] += alpha*tmp0;
539  res[(i + offset1) * resIncr] += alpha*tmp1;
540  res[(i + 2) * resIncr] += alpha*tmp2;
541  res[(i + offset3) * resIncr] += alpha*tmp3;
542  }
543 
544  // process remaining first and last rows (at most columnsAtOnce-1)
545  Index end = rows;
546  Index start = rowBound;
547  do {
548  for (Index i = start; i < end; ++i) {
549  EIGEN_ALIGN_MAX ResScalar tmp0 = ResScalar(0);
550  ResPacket ptmp0 = pset1<ResPacket>(tmp0);
551  const LhsScalars lhs0 = lhs.getVectorMapper(i, 0);
552  // process first unaligned result's coeffs
553  // FIXME this loop get vectorized by the compiler !
554  for (Index j = 0; j < alignedStart; ++j)
555  tmp0 += cj.pmul(lhs0(j), rhs(j, 0));
556 
557  if (alignedSize > alignedStart) {
558  // process aligned rhs coeffs
559  if (lhs0.template aligned<LhsPacket>(alignedStart))
560  for (Index j = alignedStart; j < alignedSize; j += RhsPacketSize)
561  ptmp0 = pcj.pmadd(lhs0.template load<LhsPacket, Aligned>(j), rhs.getVectorMapper(j, 0).template load<RhsPacket, Aligned>(0), ptmp0);
562  else
563  for (Index j = alignedStart; j < alignedSize; j += RhsPacketSize)
564  ptmp0 = pcj.pmadd(lhs0.template load<LhsPacket, Unaligned>(j), rhs.getVectorMapper(j, 0).template load<RhsPacket, Aligned>(0), ptmp0);
565  tmp0 += predux(ptmp0);
566  }
567 
568  // process remaining scalars
569  // FIXME this loop get vectorized by the compiler !
570  for (Index j = alignedSize; j < depth; ++j)
571  tmp0 += cj.pmul(lhs0(j), rhs(j, 0));
572  res[i * resIncr] += alpha*tmp0;
573  }
574  if (skipRows) {
575  start = 0;
576  end = skipRows;
577  skipRows = 0;
578  } else
579  break;
580  } while (Vectorizable);

583  }
