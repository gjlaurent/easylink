340  template<typename MatrixType, typename CoeffVectorType>
341  void tridiagonalization_inplace(MatrixType& matA, CoeffVectorType& hCoeffs) {
342  using numext::conj;
343  typedef typename MatrixType::Scalar Scalar;
344  typedef typename MatrixType::RealScalar RealScalar;
345  Index n = matA.rows();
346  eigen_assert(n == matA.cols());
347  eigen_assert(n == hCoeffs.size() + 1 || n == 1);
348 
349  for (Index i = 0; i < n - 1; ++i) {
350  Index remainingSize = n - i - 1;
351  RealScalar beta;
352  Scalar h;
353  matA.col(i).tail(remainingSize).makeHouseholderInPlace(h, beta);
354 
355  // Apply similarity transformation to remaining columns,
356  // i.e., A = H A H' where H = I - h v v' and v = matA.col(i).tail(n-i-1)
357  matA.col(i).coeffRef(i + 1) = 1;
358 
359  hCoeffs.tail(n - i - 1).noalias() = (matA.bottomRightCorner(remainingSize, remainingSize).template selfadjointView<Lower>()
360  * (conj(h) * matA.col(i).tail(remainingSize)));
361 
362  hCoeffs.tail(n - i - 1) += (conj(h) * RealScalar(-0.5)*(hCoeffs.tail(remainingSize).dot(matA.col(i).tail(remainingSize)))) * matA.col(i).tail(n - i - 1);
363 
364  matA.bottomRightCorner(remainingSize, remainingSize).template selfadjointView<Lower>()
365  .rankUpdate(matA.col(i).tail(remainingSize), hCoeffs.tail(remainingSize), Scalar(-1));
366 
367  matA.col(i).coeffRef(i + 1) = beta;
368  hCoeffs.coeffRef(i) = h;
369  }
370  }

427  template<typename MatrixType, int Size, bool IsComplex>
428  struct tridiagonalization_inplace_selector {
429  typedef typename Tridiagonalization<MatrixType>::CoeffVectorType CoeffVectorType;
430  typedef typename Tridiagonalization<MatrixType>::HouseholderSequenceType HouseholderSequenceType;
431 
432  template<typename DiagonalType, typename SubDiagonalType>
433  static void run(MatrixType& mat, DiagonalType& diag, SubDiagonalType& subdiag, bool extractQ) {
434  CoeffVectorType hCoeffs(mat.cols() - 1);
435  tridiagonalization_inplace(mat, hCoeffs);
436  diag = mat.diagonal().real();
437  subdiag = mat.template diagonal<-1 > ().real();
438  if (extractQ)
439  mat = HouseholderSequenceType(mat, hCoeffs.conjugate())
440  .setLength(mat.rows() - 1)
441  .setShift(1);
442  }
443  };

449  template<typename MatrixType>
450  struct tridiagonalization_inplace_selector<MatrixType, 3, false> {
451  typedef typename MatrixType::Scalar Scalar;
452  typedef typename MatrixType::RealScalar RealScalar;
453 
454  template<typename DiagonalType, typename SubDiagonalType>
455  static void run(MatrixType& mat, DiagonalType& diag, SubDiagonalType& subdiag, bool extractQ) {
456  using std::sqrt;
457  const RealScalar tol = (std::numeric_limits<RealScalar>::min)();
458  diag[0] = mat(0, 0);
459  RealScalar v1norm2 = numext::abs2(mat(2, 0));
460  if (v1norm2 <= tol) {
461  diag[1] = mat(1, 1);
462  diag[2] = mat(2, 2);
463  subdiag[0] = mat(1, 0);
464  subdiag[1] = mat(2, 1);
465  if (extractQ)
466  mat.setIdentity();
467  } else {
468  RealScalar beta = sqrt(numext::abs2(mat(1, 0)) + v1norm2);
469  RealScalar invBeta = RealScalar(1) / beta;
470  Scalar m01 = mat(1, 0) * invBeta;
471  Scalar m02 = mat(2, 0) * invBeta;
472  Scalar q = RealScalar(2) * m01 * mat(2, 1) + m02 * (mat(2, 2) - mat(1, 1));
473  diag[1] = mat(1, 1) + m02*q;
474  diag[2] = mat(2, 2) - m02*q;
475  subdiag[0] = beta;
476  subdiag[1] = mat(2, 1) - m01 * q;
477  if (extractQ) {
478  mat << 1, 0, 0,
479  0, m01, m02,
480  0, m02, -m01;
481  }
482  }
483  }
484  };
