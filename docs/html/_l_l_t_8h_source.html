231  template<typename MatrixType, typename VectorType>
232  static Index llt_rank_update_lower(MatrixType& mat, const VectorType& vec, const typename MatrixType::RealScalar& sigma) {
233  using std::sqrt;
234  typedef typename MatrixType::Scalar Scalar;
235  typedef typename MatrixType::RealScalar RealScalar;
236  typedef typename MatrixType::ColXpr ColXpr;
237  typedef typename internal::remove_all<ColXpr>::type ColXprCleaned;
238  typedef typename ColXprCleaned::SegmentReturnType ColXprSegment;
239  typedef Matrix<Scalar, Dynamic, 1> TempVectorType;
240  typedef typename TempVectorType::SegmentReturnType TempVecSegment;
241 
242  Index n = mat.cols();
243  eigen_assert(mat.rows() == n && vec.size() == n);
244 
245  TempVectorType temp;
246 
247  if (sigma > 0) {
248  // This version is based on Givens rotations.
249  // It is faster than the other one below, but only works for updates,
250  // i.e., for sigma > 0
251  temp = sqrt(sigma) * vec;
252 
253  for (Index i = 0; i < n; ++i) {
254  JacobiRotation<Scalar> g;
255  g.makeGivens(mat(i, i), -temp(i), &mat(i, i));
256 
257  Index rs = n - i - 1;
258  if (rs > 0) {
259  ColXprSegment x(mat.col(i).tail(rs));
260  TempVecSegment y(temp.tail(rs));
261  apply_rotation_in_the_plane(x, y, g);
262  }
263  }
264  } else {
265  temp = vec;
266  RealScalar beta = 1;
267  for (Index j = 0; j < n; ++j) {
268  RealScalar Ljj = numext::real(mat.coeff(j, j));
269  RealScalar dj = numext::abs2(Ljj);
270  Scalar wj = temp.coeff(j);
271  RealScalar swj2 = sigma * numext::abs2(wj);
272  RealScalar gamma = dj * beta + swj2;
273 
274  RealScalar x = dj + swj2 / beta;
275  if (x <= RealScalar(0))
276  return j;
277  RealScalar nLjj = sqrt(x);
278  mat.coeffRef(j, j) = nLjj;
279  beta += swj2 / dj;
280 
281  // Update the terms of L
282  Index rs = n - j - 1;
283  if (rs) {
284  temp.tail(rs) -= (wj / Ljj) * mat.col(j).tail(rs);
285  if (gamma != 0)
286  mat.col(j).tail(rs) = (nLjj / Ljj) * mat.col(j).tail(rs) + (nLjj * sigma * numext::conj(wj) / gamma) * temp.tail(rs);
287  }
288  }
289  }
290  return -1;
291  }

296  template<typename MatrixType>
297  static Index unblocked(MatrixType& mat) {
298  using std::sqrt;
299 
300  eigen_assert(mat.rows() == mat.cols());
301  const Index size = mat.rows();
302  for (Index k = 0; k < size; ++k) {
303  Index rs = size - k - 1; // remaining size
304 
305  Block<MatrixType, Dynamic, 1> A21(mat, k + 1, k, rs, 1);
306  Block<MatrixType, 1, Dynamic> A10(mat, k, 0, 1, k);
307  Block<MatrixType, Dynamic, Dynamic> A20(mat, k + 1, 0, rs, k);
308 
309  RealScalar x = numext::real(mat.coeff(k, k));
310  if (k > 0) x -= A10.squaredNorm();
311  if (x <= RealScalar(0))
312  return k;
313  mat.coeffRef(k, k) = x = sqrt(x);
314  if (k > 0 && rs > 0) A21.noalias() -= A20 * A10.adjoint();
315  if (rs > 0) A21 /= x;
316  }
317  return -1;
318  }

320  template<typename MatrixType>
321  static Index blocked(MatrixType& m) {
322  eigen_assert(m.rows() == m.cols());
323  Index size = m.rows();
324  if (size < 32)
325  return unblocked(m);
326 
327  Index blockSize = size / 8;
328  blockSize = (blockSize / 16)*16;
329  blockSize = (std::min)((std::max)(blockSize, Index(8)), Index(128));
330 
331  for (Index k = 0; k < size; k += blockSize) {
332  // partition the matrix:
333  // A00 | - | -
334  // lu = A10 | A11 | -
335  // A20 | A21 | A22
336  Index bs = (std::min)(blockSize, size - k);
337  Index rs = size - k - bs;
338  Block<MatrixType, Dynamic, Dynamic> A11(m, k, k, bs, bs);
339  Block<MatrixType, Dynamic, Dynamic> A21(m, k + bs, k, rs, bs);
340  Block<MatrixType, Dynamic, Dynamic> A22(m, k + bs, k + bs, rs, rs);
341 
342  Index ret;
343  if ((ret = unblocked(A11)) >= 0) return k + ret;
344  if (rs > 0) A11.adjoint().template triangularView<Upper>().template solveInPlace<OnTheRight>(A21);
345  if (rs > 0) A22.template selfadjointView<Lower>().rankUpdate(A21, typename NumTraits<RealScalar>::Literal(-1)); // bottleneck
346  }
347  return -1;
348  }

421  template<typename MatrixType, int _UpLo>
422  template<typename InputType>
423  LLT<MatrixType, _UpLo>& LLT<MatrixType, _UpLo>::compute(const EigenBase<InputType>& a) {
424  check_template_parameters();
425 
426  eigen_assert(a.rows() == a.cols());
427  const Index size = a.rows();
428  m_matrix.resize(size, size);
429  if (!internal::is_same_dense(m_matrix, a.derived()))
430  m_matrix = a.derived();
431 
432  // Compute matrix L1 norm = max abs column sum.
433  m_l1_norm = RealScalar(0);
434  // TODO move this code to SelfAdjointView
435  for (Index col = 0; col < size; ++col) {
436  RealScalar abs_col_sum;
437  if (_UpLo == Lower)
438  abs_col_sum = m_matrix.col(col).tail(size - col).template lpNorm<1>() + m_matrix.row(col).head(col).template lpNorm<1>();
439  else
440  abs_col_sum = m_matrix.col(col).head(col).template lpNorm<1>() + m_matrix.row(col).tail(size - col).template lpNorm<1>();
441  if (abs_col_sum > m_l1_norm)
442  m_l1_norm = abs_col_sum;
443  }
444 
445  m_isInitialized = true;
446  bool ok = Traits::inplace_decomposition(m_matrix);
447  m_info = ok ? Success : NumericalIssue;
448 
449  return *this;
450  }
