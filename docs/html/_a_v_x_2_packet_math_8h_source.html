260 template<> EIGEN_STRONG_INLINE Packet8f pmadd(const Packet8f& a, const Packet8f& b, const Packet8f& c) {
261 #if ( (EIGEN_COMP_GNUC_STRICT && EIGEN_COMP_GNUC<80) || (EIGEN_COMP_CLANG) )
262  // Clang stupidly generates a vfmadd213ps instruction plus some vmovaps on registers,
263  // and even register spilling with clang>=6.0 (bug 1637).
264  // Gcc stupidly generates a vfmadd132ps instruction.
265  // So let's enforce it to generate a vfmadd231ps instruction since the most common use
266  // case is to accumulate the result of the product.
267  Packet8f res = c;
268  __asm__("vfmadd231ps %[a], %[b], %[c]" : [c] "+x" (res) : [a] "x" (a), [b] "x" (b));
269  return res;
270 #else
271  return _mm256_fmadd_ps(a, b, c);
272 #endif
273  }

278  Packet4d res = c;
279  __asm__("vfmadd231pd %[a], %[b], %[c]" : [c] "+x" (res) : [a] "x" (a), [b] "x" (b));

385  template<> EIGEN_STRONG_INLINE Packet8f ploaddup<Packet8f>(const float* from) {
386  // TODO try to find a way to avoid the need of a temporary register
387  // Packet8f tmp = _mm256_castps128_ps256(_mm_loadu_ps(from));
388  // tmp = _mm256_insertf128_ps(tmp, _mm_movehl_ps(_mm256_castps256_ps128(tmp),_mm256_castps256_ps128(tmp)), 1);
389  // return _mm256_unpacklo_ps(tmp,tmp);
390 
391  // _mm256_insertf128_ps is very slow on Haswell, thus:
392  Packet8f tmp = _mm256_broadcast_ps((const __m128*) (const void*) from);
393  // mimic an "inplace" permutation of the lower 128bits using a blend
394  tmp = _mm256_blend_ps(tmp, _mm256_castps128_ps256(_mm_permute_ps(_mm256_castps256_ps128(tmp), _MM_SHUFFLE(1, 0, 1, 0))), 15);
395  // then we can perform a consistent permutation on the global register to get everything in shape:
396  return _mm256_permute_ps(tmp, _MM_SHUFFLE(3, 3, 2, 2));
397  }

448  template<> EIGEN_DEVICE_FUNC inline void pscatter<float, Packet8f>(float* to, const Packet8f& from, Index stride) {
449  __m128 low = _mm256_extractf128_ps(from, 0);
450  to[stride * 0] = _mm_cvtss_f32(low);
451  to[stride * 1] = _mm_cvtss_f32(_mm_shuffle_ps(low, low, 1));
452  to[stride * 2] = _mm_cvtss_f32(_mm_shuffle_ps(low, low, 2));
453  to[stride * 3] = _mm_cvtss_f32(_mm_shuffle_ps(low, low, 3));
454 
455  __m128 high = _mm256_extractf128_ps(from, 1);
456  to[stride * 4] = _mm_cvtss_f32(high);
457  to[stride * 5] = _mm_cvtss_f32(_mm_shuffle_ps(high, high, 1));
458  to[stride * 6] = _mm_cvtss_f32(_mm_shuffle_ps(high, high, 2));
459  to[stride * 7] = _mm_cvtss_f32(_mm_shuffle_ps(high, high, 3));
460  }
