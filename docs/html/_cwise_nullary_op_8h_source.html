687  template<typename Derived>
688  bool MatrixBase<Derived>::isIdentity
689  (const RealScalar& prec) const {
690  typename internal::nested_eval<Derived, 1>::type self(derived());
691  for (Index j = 0; j < cols(); ++j) {
692  for (Index i = 0; i < rows(); ++i) {
693  if (i == j) {
694  if (!internal::isApprox(self.coeff(i, j), static_cast<Scalar> (1), prec))
695  return false;
696  } else {
697  if (!internal::isMuchSmallerThan(self.coeff(i, j), static_cast<RealScalar> (1), prec))
698  return false;
699  }
700  }
701  }
702  return true;
703  }
